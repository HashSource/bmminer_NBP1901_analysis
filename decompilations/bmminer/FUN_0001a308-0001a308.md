#include <string.h>
#include <stdio.h>

#define MAX_HASHBOARDS 4
#define FREQ_ARRAY_SIZE 0x100
#define FREQ_INFO_SIZE 0x400
#define HASHBOARD_DATA_OFFSET FREQ_ARRAY_SIZE * 0x100
#define LOG_BUFFER_SIZE 0x800
#define MAX_LOG_LEVEL 3

// Function Prototypes (Assumed)
int GetMaxFrequency(uint *maxFrequencyPointer);
int ReadHashBoardFrequency(int hashboardIndex);
int VerifyFrequencyAvailability();
int ReportError(int logLevel, char *errorMessage);
int LogMessage(int logLevel, char *logMessage);

// Global Variables (Placeholders)
char GLOBAL_DATA_AREA[FREQ_INFO_SIZE * MAX_HASHBOARDS];
char GLOBAL_VAR_FREQUENCY_OFFSET;

// Function to validate hashboard frequencies
int ValidateHashboardFrequencies(void)
{
  char freqCount;
  int readStatus;
  int validationStatus;
  uint maxFrequency;
  uint boardFrequency;
  uint *frequencyArrayPointer;
  char logBuffer[LOG_BUFFER_SIZE]; // Buffer for log messages
  
  // Try to fetch the maximum allowed frequency set for the board
  readStatus = GetMaxFrequency(&maxFrequency);
  if (readStatus == 0) {
    for (int hashboardIndex = 0; hashboardIndex < MAX_HASHBOARDS; hashboardIndex++) {
      validationStatus = ReadHashBoardFrequency(hashboardIndex);
      
      // If the hashboard's frequency info was read successfully
      if (validationStatus != 0) {
        // Clear the hashboard's frequency data area
        memset(GLOBAL_DATA_AREA + hashboardIndex * HASHBOARD_DATA_OFFSET, 0, FREQ_INFO_SIZE);

        // Populate the hashboard's frequency data area with current values
        validationStatus = ReadFrequencyDataIntoBuffer(hashboardIndex, (unsigned char)GLOBAL_VAR_FREQUENCY_OFFSET, GLOBAL_DATA_AREA + hashboardIndex * HASHBOARD_DATA_OFFSET);
        
        // If the hashboard's frequency data could not be populated
        if (validationStatus != 0) {
          // An error occurred, report it
          ReportError(3, 0xff);
          ReportError(0xc, "Get frequency failed!");
          return -1;
        }

        // Check if the hashboard has available frequencies to operate on
        freqCount = VerifyFrequencyAvailability();
        
        // If there is at least one available frequency
        if (freqCount != '\0') {
          // Retrieve the primary frequency for the hashboard
          boardFrequency = GLOBAL_DATA_AREA[hashboardIndex * HASHBOARD_DATA_OFFSET];
          
          // If the primary frequency exceeds the maximum allowed frequency
          if (maxFrequency < boardFrequency) {
            // Log error message if log level permits
            if (MAX_LOG_LEVEL < GLOBAL_DATA_AREA[0x7eb9c]) {
              snprintf(logBuffer, LOG_BUFFER_SIZE, "Freq(%d) > Max_freq(%d), invalid!\n", boardFrequency, maxFrequency);
              LogMessage(3, logBuffer, 0);
            }
            // Continue to report error and exit
            ReportError(3, 0xff);
            ReportError(0xc, "Get frequency failed!");
            return -1;
          }

          // Iterate through other frequencies if available
          frequencyArrayPointer = GLOBAL_DATA_AREA + HASHBOARD_DATA_OFFSET + 1 + hashboardIndex * FREQ_ARRAY_SIZE;
          while (frequencyArrayPointer != &GLOBAL_DATA_AREA[HASHBOARD_DATA_OFFSET + hashboardIndex * FREQ_ARRAY_SIZE + freqCount]) {
            boardFrequency = *frequencyArrayPointer++;
            
            // Check if current frequency also exceeds maximum allowed frequency
            if (maxFrequency < boardFrequency) {
              // Log error message if log level permits
              if (MAX_LOG_LEVEL < GLOBAL_DATA_AREA[0x7eb9c]) {
                snprintf(logBuffer, LOG_BUFFER_SIZE,"Freq(%d) > Max_freq(%d), invalid!\n", boardFrequency, maxFrequency);
                LogMessage(3, logBuffer, 0);
              }
              // Continue to report error and exit
              ReportError(3, 0xff);
              ReportError(0xc, "Get frequency failed!");
              return -1;
            }
          }
        }
      }
    }
    // Validation succeeded for all hashboards
    return 0;
  } else {
    // Failed to fetch the maximum frequency, log the error
    ReportError(3, 0xff);
    ReportError(0xc, "Get max freq failed!");
    return -1;
  }
}
