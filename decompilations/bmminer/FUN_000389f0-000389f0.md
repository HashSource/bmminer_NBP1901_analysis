void StartMinerThreads(int miningIntensity)
{
  int checkMinerStatus;
  int threadId;
  int *threadParameters;
  pthread_t minerThreads[4]; // Array to store thread IDs for up to 4 miner threads
  int threadStartStatus; // Used to check the status of the thread start
  int minerStateFlags[3]; // Array to hold state of 3 miners?
  undefined localFlag;

  if ((globalMiningFlag == 0) || (*(int *)(globalMiningDataPointer + 4) == 0)) {

      // Copy log message into local variables for printing error
      threadStartStatus = loggingDataHandleError._0_4_;
      minerStateFlags[0] = loggingDataHandleError._4_4_;
      minerStateFlags[1] = loggingDataHandleError._8_4_;
      minerStateFlags[2] = loggingDataHandleError._12_4_;
      localFlag = (undefined)loggingDataHandleErrorByte;
      LogError(3, &threadStartStatus, 0); // Log the handle is NULL error
      return;
    }
  }
  else {
    // Setup for creating miner threads
    threadParameters = &threadStartStatus; // Point to first element of thread parameters
    threadId = 0; // Initialize the thread ID
    // Zero-initialize miner thread IDs and start status
    minerThreads[0] = 0;
    minerThreads[1] = 0;
    minerThreads[2] = 0;
    minerThreads[3] = 0;
    memset(threadParameters, 0, 0x30); // Clear memory for thread parameters
    do {
      // Checks availability/status of a miner to determine if it can start
      checkMinerStatus = CheckMinerAvailability(threadId);
      if ((checkMinerStatus != 0) && (*(int *)(globalMiningDataPointer + (threadId + 2) * 4) == 1)) {

        *threadParameters = threadId;

        // Create the miner thread with the thread ID and default attributes
        pthread_create(&minerThreads[threadId], (pthread_attr_t *)0x0, MinerThreadFunction, threadParameters);
        pthread_detach(minerThreads[threadId]); // Detach the newly created thread
      }
      threadId++; // Move on to the next thread ID
      threadParameters += 3; // Move to the next set of parameters (each set is 3 ints)
    } while (threadId != 4); // Loop until all 4 miners are created or attempted to be created

      while(true) {

        usleep(100000); // Sleep for a moment before trying again
        threadId = 0; // Reset thread ID for another pass through the loop
      }
      threadId++; // Check the next thread
    } while (threadId != 4); // Continue until all thread statuses are checked
  }
  return;
}
