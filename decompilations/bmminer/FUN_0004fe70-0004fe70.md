#define STATUS_SUCCESS 0
#define STATUS_FAILURE -1
#define SOME_CONSTANT_VALUE 5

/**
 * Function to Check and Update Hardware Status
 * @param hardware_status  Variable where the updated hardware status is returned
 * @return 0 if success, -1 if failure
 */
int CheckAndUpdateHardwareStatus(unsigned char hardware_status[8]);

/**
 * Function to Perform Hardware Operation
 * @param operation_param Parameter for the operation to be performed on hardware
 * @param hardware_status Current hardware status needed for the operation
 * @param operation_size  Size of the operation parameter
 * @return 0 if success, -1 if failure
 */
int PerformHardwareOperation(unsigned int operation_param, unsigned char hardware_status[8], int operation_size);

/**
 * Main Function to Update Hardware with Given Parameter
 *
 * This function first checks the hardware status and updates it if check passes,
 * then it performs a hardware operation using the passed parameter and the updated status.
 *
 * @param operation_param The parameter for the operation to perform on hardware
 * @return 0 if success, -1 if failure
 */
unsigned int UpdateHardwareWithParameter(unsigned int operation_param)
{
  int check_status;
  unsigned char updated_status[8];
  unsigned int operation_result;
  
  // Check and update the hardware status
  check_status = CheckAndUpdateHardwareStatus(updated_status);
  
  // If the check is successful, perform the operation with the updated status
  if (check_status == STATUS_SUCCESS) {
    operation_result = PerformHardwareOperation(operation_param, updated_status, SOME_CONSTANT_VALUE);
  } else {
    // Otherwise, report failure
    operation_result = STATUS_FAILURE;
  }
  
  return operation_result;
}
