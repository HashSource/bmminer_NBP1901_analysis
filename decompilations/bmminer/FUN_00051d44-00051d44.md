// Initializes a control structure used to interact with the Antminer hardware.
// The control structure is set to default values and a hardware-specific function is called to
// further initialize it with runtime details.
//
// Parameters:
//   operationMode: A byte-sized value representing the operation mode to set.
//   ctrlStructPtr: Pointer to a control structure to be initialized.
//
// Returns:
//   If ctrlStructPtr is NULL, returns 0xffffffff indicating an error.
//   Otherwise, returns 0 indicating success.

uint32_t InitializeControlStructure(uint8_t operationMode, uint32_t *ctrlStructPtr)
{
  uint8_t hardwareInitStatus;
  uint32_t result;

  // Validate the control structure pointer.
  if (ctrlStructPtr == NULL) {
    // Return 0xffffffff if the control structure pointer is NULL.
    result = 0xffffffff;
  }
  else {
    // Set result to zero (success) and initialize the control structure with default values.
    result = 0;
    *ctrlStructPtr = 0;                               // Clear primary control value.
    *(uint8_t *)(ctrlStructPtr + 1) = 0;              // Clear secondary control value.
    *(uint8_t *)((int)ctrlStructPtr + 2) = operationMode; // Set operation mode.
    *(uint8_t *)((int)ctrlStructPtr + 1) = 5;         // Set some other control code to 5.
    *(uint8_t *)ctrlStructPtr = 0x40;                 // Set control flag to 0x40 (possibly a permission or state flag).

    // Call a hardware-specific function to initialize additional parts of the control structure.
    hardwareInitStatus = InitializeHardware(ctrlStructPtr, 0x20);

    // Update the second byte of the control structure based on the hardware initialization status.
    // Preserve the upper 3 bits and merge in the lower 5 bits from hardwareInitStatus.
    *(uint8_t *)(ctrlStructPtr + 1) = (*(uint8_t *)(ctrlStructPtr + 1) & 0xe0) | (hardwareInitStatus & 0x1f);
  }
  // Return the result of the initialization.
  return result;
}
