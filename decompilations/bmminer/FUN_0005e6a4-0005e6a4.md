void PrintBufferInfo(int *bufferPtr, undefined4 logLevel)
{
  // The actual buffer start pointer is at the first index of the bufferPtr array
  int *actualBufferStart = *bufferPtr;
  
  // The offset for read operations is at the fifth index of the bufferPtr array
  long readOffset = bufferPtr[4];
  
  // The offset for write operations is at the sixth index of the bufferPtr array
  long writeOffset = bufferPtr[5];
  
  // The total buffer size is at the seventh index of the bufferPtr array
  long totalBufferSize = bufferPtr[6];
  
  // The buffer metadata string is placed right after the buffer size in memory i.e., at the eighth index
  char *bufferMetadataStr = (char *)(bufferPtr + 7);
  
  // The difference between the write and read offset gives us the used buffer size
  long usedBufferSize = bufferPtr[1] - actualBufferStart;
  
  // Call the logging function with the formatted buffer information
  LogFormattedBufferInfo(logLevel, "src/buf.c", 0x4a,
                         "---buf[%p][%ld-%ld][%ld][%s][%p:%ld]---",
                         bufferPtr, readOffset, writeOffset, totalBufferSize,
                         bufferMetadataStr, actualBufferStart, usedBufferSize);
  return;
}
