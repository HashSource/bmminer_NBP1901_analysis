void ProcessMinerTemperatureData(int minerIndex, uint temperatureSensorIndex) {
  short rawTemperatureValue;
  uint32_t formattedTemperature;
  int sensorCount;
  uint currentSensorIndex;
  char logBuffer[2052];
  
  // Validate parameters
  if (minerIndex == 0 || temperatureSensorIndex > 3) {
    // Log error if logging is enabled
    if ((isLoggingEnabled != '\0') || (isDebugModeEnabled != '\0') || (logLevel > 2)) {
      snprintf(logBuffer, 0x800, "%s: input bad param\n", functionName);
      LogErrorMessage(3, logBuffer, 0);
    }
    return; // Invalid parameters, exit function
  }
  uint32_t temperatureDataHolder = InitializeTemperatureDataHolder();
  currentSensorIndex = 0;
  
  if ((isSingleTemperatureSensorEnabled == 1) || (configuredSensorCount < 1)) {
    // Use single sensor temperature reading flow
    while (true) {
      sensorCount = GetSingleSensorCount();
      if (sensorCount <= (int)currentSensorIndex) break;
      rawTemperatureValue = ReadSingleTemperatureSensor(temperatureSensorIndex, currentSensorIndex & 0xff);
      currentSensorIndex++;
      formattedTemperature = FormatRawTemperatureData((int)rawTemperatureValue, (int)rawTemperatureValue >> 0x1f);
      StoreFormattedTemperatureData(temperatureDataHolder, formattedTemperature);
    }
  } else {
    // Use multiple sensors temperature reading flow
    while (true) {
      sensorCount = GetMultipleSensorCount();
      if (sensorCount <= (int)currentSensorIndex) break;
      rawTemperatureValue = ReadMultipleTemperatureSensors(temperatureSensorIndex, currentSensorIndex & 0xff);
      formattedTemperature = FormatRawTemperatureData((int)rawTemperatureValue, (int)rawTemperatureValue >> 0x1f);
      StoreFormattedTemperatureData(temperatureDataHolder, formattedTemperature);
      currentSensorIndex++;
    }
  }
  
  // Pass formatted temperature data to the storage function
  StoreTemperatureData(minerIndex, "temp_pcb", temperatureDataHolder);
}
