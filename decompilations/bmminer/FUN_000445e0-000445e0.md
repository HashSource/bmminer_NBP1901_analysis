void WaitForMinerHardwareResponse(int minerId)
{
  int responseStatus;
  int retryCount;
  
  SendHardwareRequest(0x1b, minerId); // Send request to miner hardware with a specific ID
  if (minerId >= 0) {
    CheckHardwareResponse(0x1b, &responseStatus); // Check response status from hardware
    return;
  }
  // If miner ID is invalid, retry sending the request up to 3000 times (3000ms)
  retryCount = 3000;
  do {
    CheckHardwareResponse(0x1b, &responseStatus); // Check response status from hardware
    if (responseStatus >= 0) {
      return; // If response is valid, return
    }
    usleep(1000); // Wait for 1ms before retrying
    retryCount = retryCount - 1;
  } while (retryCount != 0); // Loop until retries are exhausted
  return; // Return if no valid response is received after 3000 retries
}
