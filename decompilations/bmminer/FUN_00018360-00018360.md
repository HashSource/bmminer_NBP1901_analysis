#include <stdio.h>
#include <stdint.h>
#include <string.h>

// External variables, which should be declared somewhere else in the program
extern int loggingLevel;
extern uint32_t *array_of_pointers_to_structs; // Assuming it's an array of pointers to some structs

// Custom error codes
#define SUCCESS 0
#define ERROR_VERSION_INFO_MISSING -1 // Version information is missing
#define ERROR_INVALID_ARGUMENT -2 // Invalid argument passed to the function
#define ERROR_BUFFER_TOO_SMALL -2 // Provided buffer is too small to hold the version

/**
 * Fetches the firmware version string of a specific mining chain.
 *
 * @param chain_index The index number of the chain whose firmware version is to be retrieved.
 * @param version_buffer A buffer where the firmware version string will be written.
 * @param buffer_length The size of the buffer provided for the version string.
 *
 * @return A status code indicating the result of the operation.
 */
uint32_t getFirmwareVersionForChain(int chain_index, char *version_buffer, uint32_t buffer_length) {
  size_t version_length;
  char *firmware_version_pointer;
  char debug_message[2040];

  // Check if version information is available and not empty
  if ((array_of_pointers_to_structs == NULL) || (*(char *)(array_of_pointers_to_structs[chain_index] + 0x1c) == '\0')) {
    if (loggingLevel < 4) {
      return ERROR_VERSION_INFO_MISSING;
    }
    snprintf(debug_message, sizeof(debug_message), "No firmware version, chain = %d\n", chain_index);
    logMessage(3, debug_message); // Assuming the existence of a logMessage function
    return ERROR_VERSION_INFO_MISSING;
  }

  // Validate the provided version buffer
  if (version_buffer == NULL) {
    return ERROR_INVALID_ARGUMENT;
  }

  // Retrieve the firmware version string pointer from the struct associated with the chain
  firmware_version_pointer = *(char **)(*(int *)(array_of_pointers_to_structs[chain_index] + 0x10));

  // Get the length of the firmware version string
  version_length = strlen(firmware_version_pointer);

  // Ensure that the provided buffer is large enough to hold the version string
  if (version_length < buffer_length) {
    snprintf(version_buffer, buffer_length, "%s", firmware_version_pointer);
    return SUCCESS; // Firmware version successfully copied to provided buffer
  }

  // Buffer is too small to hold the version string
  return ERROR_BUFFER_TOO_SMALL;
}
