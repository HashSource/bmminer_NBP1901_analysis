#include <locale.h>
#include <string.h>
#include <errno.h>
#include <stdlib.h>
#include <math.h>

// Convert string to double with locale consideration and strict error checking.
// param_1: Pointer to string buffer and its length.
// param_2: Pointer to a double variable where the result will be stored.
// Returns: 0 on success, or 0xffffffff on failure.
uint32_t ConvertStringToDoubleWithLocale(char **stringBuffer, double *result) {
  char decimalChar;
  struct lconv *localeInfo;
  char *decimalPointPosition;
  int *errnoPointer;
  uint32_t returnCode;
  char *endPtr;
  double convertedValue;

  // Get locale info, specifically the decimal point character.
  localeInfo = localeconv();
  endPtr = *stringBuffer;
  decimalChar = *localeInfo->decimal_point;

  // Replace '.' with locale-specific decimal point if necessary.
  if ((decimalChar != '.') && (decimalPointPosition = strchr(endPtr, '.'), decimalPointPosition != NULL)) {
    *decimalPointPosition = decimalChar;
  }

  // Store the current error number and reset it to zero.
  errnoPointer = __errno_location();
  *errnoPointer = 0;

  // Attempt to convert the string to a double.
  convertedValue = strtod(endPtr, &endPtr);

  // Check if conversion consumed the entire input buffer and that the converted value is within range.
  // Also check for various error conditions including out-of-range and invalid input errors.
  if (endPtr == *stringBuffer + (intptr_t)stringBuffer[1]) {
    if (((convertedValue == DBL_MIN || convertedValue < DBL_MIN != (isnan(convertedValue) || isnan(DBL_MIN))) &&
        (DBL_MAX >= convertedValue)) || (*errnoPointer != ERANGE)) {
      returnCode = 0;
      *result = convertedValue;
    } else {
      returnCode = 0xffffffff; // Indicate failure due to conversion error.
    }
    return returnCode;
  }

  // If we reach here, the string was not fully consumed, which means invalid input format.
  // Trigger an assertion failure.
  __assert_fail("endPtr == stringBuffer->value + stringBuffer->length", "strconv.c", 0x4e, "ConvertStringToDoubleWithLocale");
}
