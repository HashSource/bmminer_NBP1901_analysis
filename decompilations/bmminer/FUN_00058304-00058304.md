{
  void *record;
  size_t nameLength;
  int *errnoPtr;
  
  // Validate name argument
  if (name == NULL) {
    logError(2, "src/record.c", 31, "name is null or zero");
    record = NULL;
  }

  else if (data == 0) {
    logError(2, "src/record.c", 32, "data is null or zero");
    record = NULL;
  }
  else {
    // Allocate memory for the record
    record = calloc(1, 0x408); // 0x408 is the size of the record structure
    if (record == NULL) {
      errnoPtr = __errno_location(); // Get the last error number
      logError(2, "src/record.c", 36, "calloc failed, errno[%d]", *errnoPtr);
    }
    else {
      nameLength = strlen(name);
      if (nameLength < 0x401) { // Ensure the name fits in the record 0x401 buffer size
        // Copy name into the record
        memcpy(record, name, nameLength + 1);
        // Set data in the record
        *(int *)((int)record + 0x404) = data;
        // Initialize the rest of the record with a custom function (assumed to be zeroing)
        initializeRecord(record, 0);
      }
      else {
        // Name too long to fit in the record
        logError(2, "src/record.c", 41, "name[%s] is too long", name);

        freeRecord(record);

      }
    }
  }
  return record;
}
