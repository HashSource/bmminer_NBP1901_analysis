{
  int currentTemperature;
  int minimumTemperature;
  int nextIndex;
  
  minimumTemperature = 0xff; // Initialize minimum temperature to highest possible value (255 in decimal)
  currentTemperature = 0; // Start with the first temperature sensor
  
  do {
    nextIndex = currentTemperature + 1; // Prepare to check the next temperature sensor
    currentTemperature = readTemperatureSensor(currentTemperature); // Read the current temperature sensor
    if (currentTemperature < minimumTemperature) {
      minimumTemperature = currentTemperature; // Update minimum temperature if a new minimum is found
    }
    minimumTemperature = (int)(short)minimumTemperature; // Cast the minimum temperature to a short to ensure it's in a valid range
    currentTemperature = nextIndex; // Move to the next temperature sensor

}
