double CalculateAverageTemperature(int sensorRow, int sensorColumn)
{
  double *sensorDataPointer;
  int indexDifference;
  bool isNegative;
  bool isEqual;
  double singleTemperature;
  double accumulatedTemperature;
  
  // Check if the sensorRow is less than 4, adjusting checks for sensorColumn if necessary
  isEqual = sensorRow == 3;
  isNegative = SBORROW4(sensorRow, 3);
  indexDifference = sensorRow - 3;
  if (sensorRow < 4) {
    isEqual = sensorColumn == 0xff;
    isNegative = SBORROW4(sensorColumn, 0xff);
    indexDifference = sensorColumn - 0xff; 
  }
  
  // If the sensor index is not equal or not within the valid range, return a default temperature
  if (!isEqual && indexDifference < 0 == isNegative) {
    return DEFAULT_TEMPERATURE_VALUE; // Placeholder for actual default temperature value
  }
  
  // Lock the mutex before accessing shared temperature data
  pthread_mutex_lock(TEMPERATURE_DATA_MUTEX);
  
  // Calculate the pointer to the temperature data for the given sensor
  sensorDataPointer = (double *)(&TEMPERATURE_DATA_BASE + (sensorColumn + sensorRow * 0x100) * 0x20);
  
  // Initialize loop variables
  int sensorDataCount = 4;
  accumulatedTemperature = DEFAULT_TEMPERATURE_VALUE; // Placeholder for actual default temperature value

  // Sum temperatures from four data points
  do {
    singleTemperature = *sensorDataPointer;
    sensorDataPointer = sensorDataPointer + 1;
    sensorDataCount = sensorDataCount - 1;
    accumulatedTemperature = accumulatedTemperature + singleTemperature;
  } while (sensorDataCount != 0);
  
  // Unlock the mutex after accessing shared temperature data
  pthread_mutex_unlock(TEMPERATURE_DATA_MUTEX);
  
  // Return the average temperature value
  return accumulatedTemperature * 0.25; // Each temperature contributes to 1/4 of the average
}
