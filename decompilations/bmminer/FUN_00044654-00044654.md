// GetMinerStatusValue - Retrieves a specific miner status value from a global status array.
//
// param index: The index of the miner within the global array to retrieve the status from.
//
// Returns an 8-byte (64-bit) value from the status structure of the specified miner.

uint64_t GetMinerStatusValue(int index)
{
  // Assuming that 'globalMinerStatusArray' is an already defined global variable
  // pointing to the base address of an array containing miner status structures,
  // where each structure is assumed to be 0x40 (64) bytes in size.
  // The status value is assumed to be at an offset of 8 bytes within the structure.

  return *(uint64_t *)(globalMinerStatusArray + index * 0x40 + 8);
}
