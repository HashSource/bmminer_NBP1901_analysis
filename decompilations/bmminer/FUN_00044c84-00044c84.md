void SetASICFrequency(uint frequency_code) 
{
  undefined frequency_config_buffer[12];
  
  // Send frequency code to the ASIC with the high bit set to indicate a command
  SendASICCommand(0x14, frequency_code & 0x1ffff | 0x80000000);

  // Read the response from ASIC into the buffer
  ReadASICResponse(0x14, frequency_config_buffer);

  // Function ends without explicitly using the response
  return;
}
