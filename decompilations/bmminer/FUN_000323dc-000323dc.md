void PrintChainVersions(void)
{
  int chainIndex; // chainIndex is used to iterate over possible chains
  int chainFound; // chainFound will be 1 if the chain is found, 0 otherwise
  unsigned int pcbVersion; // Holds PCB version
  unsigned int bomVersion; // Holds BOM version
  char outputBuffer[2052]; // Buffer for output strings

  chainIndex = 0; // Start with the first chain
  do {
    // Check if the chain is found for the current index
    chainFound = CheckChainPresence(chainIndex);
    // If chain is found and logging level is higher than 3
    if ((chainFound != 0) && (3 < loggingLevel)) {
      pcbVersion = GetPCBVersion(chainIndex);
      // Prepare and output the message with the chain's PCB Version info
      snprintf(outputBuffer, sizeof(outputBuffer), "Chain [%d] PCB Version: 0x%04x\n", chainIndex, pcbVersion);
      LogEvent(3, outputBuffer, 0);
      
      // If the logging level is still set high enough, get and log BOM version
      if (3 < loggingLevel) {
        bomVersion = GetBOMVersion(chainIndex);
        // Prepare and output the message with the chain's BOM Version info
        snprintf(outputBuffer, sizeof(outputBuffer), "Chain [%d] BOM Version: 0x%04x\n", chainIndex, bomVersion);
        LogEvent(3, outputBuffer, 0);
      }
    }
    chainIndex = chainIndex + 1; // Move to the next chain index
  } while (chainIndex != 4); // There are 4 chains, indexed 0 to 3
  return;
}
