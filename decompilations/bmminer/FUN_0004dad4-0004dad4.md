// Function: PopByteFromBuffer
// Description: This function pops a single byte from a custom buffer structure.
//              The buffer seems to be defined by an array, with the first element
//              being a pointer to the actual buffer content, and the second
//              element keeping track of the size or the index to the next
//              byte that needs to be popped.
// Arguments:
//     bufferPtr - a pointer to an array representing the buffer. 
//                 bufferPtr[0] is a pointer to the buffer content, and
//                 bufferPtr[1] is the index/size.
// Returns:
//     The byte that was popped from the buffer, or 0 if the buffer is empty.
char PopByteFromBuffer(int *bufferPtr)
{
  char poppedByte;

  // Check if the buffer is empty
  if (bufferPtr[1] == 0) {
    // Buffer is empty, return 0
    poppedByte = 0;
  } else {
    // Pop the top-most byte from the buffer
    bufferPtr[1]--;  // Decrement size/index
    poppedByte = *(char *)(*bufferPtr + bufferPtr[1]);  // Get the byte
    *(char *)(*bufferPtr + bufferPtr[1]) = 0;  // Clear the slot where the byte was
  }

  return poppedByte;
}
