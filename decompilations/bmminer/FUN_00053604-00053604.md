void DeinitializeLoggingSystem(void)
{
  int lockResult;
  
  LogDebugMessage(0,"src/zlog.c",0x130,"------Logging system deinitialization start------");
  lockResult = pthread_rwlock_wrlock((pthread_rwlock_t *)&GlobalLoggingLock);
  if (lockResult == 0) {
    if (IsLoggingInitialized == 0) {
      LogErrorMessage(2,"src/zlog.c",0x138,"Before finishing, must call InitializeLogging() or InitializeDefaultLogging() first");
    }
    else {
      ReleaseLoggingResources();
      IsLoggingInitialized = 0;
    }
    LogDebugMessage(0,"src/zlog.c",0x140,"------Logging system deinitialization end------");
    lockResult = pthread_rwlock_unlock(&LoggingLock);
    if (lockResult != 0) {
      LogErrorMessage(2,"src/zlog.c",0x143,"pthread_rwlock_unlock failed, rc=[%d]",lockResult);
    }
    return;
  }
  LogErrorMessage(2,"src/zlog.c",0x133,"pthread_rwlock_wrlock failed, rc[%d]",lockResult);
  return;
}
