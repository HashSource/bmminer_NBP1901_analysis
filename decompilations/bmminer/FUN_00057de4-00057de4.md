int FindLevelIndex(int *levelListPtr, char *levelName)
{
  int i;
  int numLevels;
  int *currentLevelPtr;
  int levelIndex;

  // Check if the level name is NULL or an empty string
  if ((levelName == NULL) || (*levelName == '\0')) {
    LogError(2, "src/level_list.c", 0x84, "Level name is [%s], can't find level", levelName);
  }
  else {
    // Get the number of levels
    numLevels = levelListPtr[1];
    if (0 < numLevels) {
      // Get the pointer to the first level entry, subtracting 4 for the offset
      currentLevelPtr = (int *)(*levelListPtr + -4);
      levelIndex = 0;
      do {
        currentLevelPtr = currentLevelPtr + 1;  // Move to the next level entry
        // Check if it's a valid entry and compare the level name with the current level entry name
        if ((*currentLevelPtr != 0) && (i = strcasecmp(levelName, (char *)(*currentLevelPtr + 4)), i == 0)) {
          return levelIndex;  // Level was found. Return the index.
        }
        levelIndex = levelIndex + 1;  // Move to the next index
      } while (levelIndex != numLevels);  // Continue until all levels are checked
    }
    // If the level was not found, log an error indicating it could not be found.
    LogError(2, "src/level_list.c", 0x8e, "Level name [%s] can't be found in level list", levelName);
  }
  return -1;  // Return -1 if the level is not found or if an error occurred.
}
