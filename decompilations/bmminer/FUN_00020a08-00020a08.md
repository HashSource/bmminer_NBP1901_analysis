// Set the status of the Antminer system
void SetAntminerStatus(byte newStatus)
{
  // Update the global status variable with the new status, preserving other bits
  // We assume DAT_0007f8f4 is a global variable representing system status,
  // where the least significant bit (LSB) toggles a specific status flag.
  globalSystemStatus._0_1_ = (byte)globalSystemStatus & 0xFE | newStatus & 0x01;
  // The "& 0xFE" operation clears the LSB (making it 0), and "| newStatus & 0x01"
  // sets the LSB to the value of the LSB of newStatus, effectively changing the flag
  // without altering the other bits in the status byte.
  return;
}
