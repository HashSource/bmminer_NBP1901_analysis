#include <sys/mman.h>
#include <fcntl.h>
#include <stdio.h>
#include <unistd.h>

#define FPGA_DEVICE "/dev/axi_fpga_dev"
#define FPGA_MEM_DEVICE "/dev/fpga_mem"
#define FPGA_MMAP_SIZE 0x1200
#define FPGA_MEM_MMAP_SIZE 0x1000000
#define ERROR_OPEN_FAILED -1
#define ERROR_MMAP_FAILED -2

// Global variables to keep track of FPGA device descriptors and memory mappings
int fpgaDeviceFd = -1;
void* fpgaRegBaseAddr = MAP_FAILED;
int fpgaMemFd = -1;
void* fpgaMemBaseAddr = MAP_FAILED;
int fpgaInitialized = 0;

// InitializeFpga - Initializes FPGA by mapping its registers and memory into the process space
int InitializeFpga(void) {
  char logBuffer[2048]; // Buffer to store logs
  
  // Open FPGA device file
  fpgaDeviceFd = open(FPGA_DEVICE, O_RDWR);
  if (fpgaDeviceFd < 0) {
    // Log error message if opening the FPGA device fails
    snprintf(logBuffer, sizeof(logBuffer), "Failed to open %s. Error: %d\n", FPGA_DEVICE, fpgaDeviceFd);
    LogError(logBuffer);
    return ERROR_OPEN_FAILED;
  } else {
    // Map FPGA registers into the process's address space
    fpgaRegBaseAddr = mmap(NULL, FPGA_MMAP_SIZE, PROT_READ | PROT_WRITE, MAP_SHARED, fpgaDeviceFd, 0);
    if (fpgaRegBaseAddr == MAP_FAILED) {
      // Log error message if memory-mapping the FPGA registers fails
      snprintf(logBuffer, sizeof(logBuffer), "Failed to map FPGA registers. Error: %p\n", fpgaRegBaseAddr);
      LogError(logBuffer);
      close(fpgaDeviceFd);
      fpgaDeviceFd = -1;
      return ERROR_MMAP_FAILED;
    } else {
      // Open FPGA memory device file
      fpgaMemFd = open(FPGA_MEM_DEVICE, O_RDWR);
      if (fpgaMemFd < 0) {
        // Log error message if opening the FPGA memory device fails
        snprintf(logBuffer, sizeof(logBuffer), "Failed to open %s. Error: %d\n", FPGA_MEM_DEVICE, fpgaMemFd);
        LogError(logBuffer);
        perror("open");
        munmap(fpgaRegBaseAddr, FPGA_MMAP_SIZE);
        close(fpgaDeviceFd);
        fpgaDeviceFd = -1;
        return ERROR_OPEN_FAILED;
      } else {
        // Map FPGA memory into the process's address space
        fpgaMemBaseAddr = mmap(NULL, FPGA_MEM_MMAP_SIZE, PROT_READ | PROT_WRITE, MAP_SHARED, fpgaMemFd, 0);
        if (fpgaMemBaseAddr == MAP_FAILED) {
          // Log error message if memory-mapping the FPGA memory fails
          snprintf(logBuffer, sizeof(logBuffer), "Failed to map FPGA memory. Error: %p\n", fpgaMemBaseAddr);
          LogError(logBuffer);
          munmap(fpgaRegBaseAddr, FPGA_MMAP_SIZE);
          close(fpgaDeviceFd);
          close(fpgaMemFd);
          fpgaDeviceFd = -1;
          fpgaMemFd = -1;
          return ERROR_MMAP_FAILED;
        } else {
          // Successfully initialized FPGA, log the memory base address
          snprintf(logBuffer, sizeof(logBuffer), "Mapped FPGA memory at address: %p\n", fpgaMemBaseAddr);
          LogInfo(logBuffer);
          fpgaInitialized = 1;
          return 0; // Success
        }
      }
    }
  }
}
