```c
// Function: freePointerArray
// Description: Frees a dynamically allocated array of pointers and the array itself.
// Parameters:
//     - pointerArray: A pointer to an array of pointers to be freed.
void freePointerArray(void **pointerArray)
{
  // Check if the pointer array is NULL to avoid dereferencing a NULL pointer
  if (pointerArray == NULL) {
    return;
  }
  // Free the memory block pointed to by the first element in the array
  free(*pointerArray);
  // Free the memory block for the array itself
  free(pointerArray);
}
```
