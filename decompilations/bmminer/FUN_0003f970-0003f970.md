double CalculateValuesSum(int rowIndex, int columnIndex)
{
  // Declare pointers and variables for the calculation
  double *valuePtr;
  int compResult;
  bool lowerThanRowIndexBoundary;
  bool lowerThanColumnIndexBoundary;
  double sum;
  double currentValue;
  
  // Check if the provided rowIndex is less than 4
  lowerThanRowIndexBoundary = (rowIndex - 3) < 0;
  compResult = rowIndex - 3;
  lowerThanColumnIndexBoundary = rowIndex == 3;
  if (rowIndex < 4) {
    // Check if the provided columnIndex is less than 255
    lowerThanColumnIndexBoundary = (columnIndex - 0xFF) < 0;
    compResult = columnIndex - 0xFF;
    lowerThanColumnIndexBoundary = columnIndex == 0xFF;
  }
  // If either boundary check fails, return a default value
  if (!lowerThanColumnIndexBoundary && compResult < 0 == lowerThanRowIndexBoundary) {
    return DEFAULT_RETURN_VALUE;
  }
  // Lock mutex before accessing shared resources
  pthread_mutex_lock(MUTEX_FOR_GLOBAL_DATASET);
  
  // Calculate the starting pointer based on rowIndex and columnIndex
  valuePtr = (double *)(GLOBAL_DATASET_BASE_PTR + (columnIndex + rowIndex * 0x100) * 0x20);
  
  // Initialize sum with a default value
  sum = DEFAULT_RETURN_VALUE;
  // Loop through the dataset to accumulate the sum of 4 values
  for (int i = 0; i < 4; ++i) {
    currentValue = *valuePtr;
    valuePtr = valuePtr + 1;
    sum = sum + currentValue;
  }
  
  // Unlock the mutex after accessing shared resources
  pthread_mutex_unlock(MUTEX_FOR_GLOBAL_DATASET);
  // Return the calculated sum
  return sum;
}
