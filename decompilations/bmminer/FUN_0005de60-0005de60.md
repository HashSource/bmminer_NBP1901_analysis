// Function: FindFirstNonZeroValue
// Description: Finds and returns the first non-zero integer from an array. If all values are zero,
// or the array length is zero, the function returns 0.
//
// Parameters:
//   arrayInfo - An integer pointer to a structure holding array information.
//               The structure is expected to be:
//               [0] - Pointer to the array.
//               [1] - Length of the array (number of elements).
//
// Returns:
//   The first non-zero value in the array, or 0 if all values are zero or the array is empty.

int FindFirstNonZeroValue(int *arrayInfo)
{
  int *arrayPtr;  // Pointer to the start of the array.
  int arrayLength;  // Length of the array.
  int index;  // Index used for iterating through the array.
  
  // Check if the array length is zero by accessing the value at (arrayInfo + 8)
  // which corresponds to the length member of the structure.
  arrayLength = *(arrayInfo + 2); // Adjusted to reflect index rather than byte offset
  if (arrayLength == 0) {
    return 0;
  }
  
  // Set the array pointer to the start of the array, which is stored at (arrayInfo + 4)
  arrayPtr = *(int **)(arrayInfo + 1); // Adjusted to reflect index rather than byte offset
  // Check if the first element of the array is non-zero and return it if true
  if (*arrayPtr != 0) {
    return *arrayPtr;
  }
  
  // Initialize index for iterating through the array
  index = 0;
  do {
    // Increment the index
    index = index + 1;
    // If we've checked all elements, return 0 because all values are zero
    if (index == arrayLength) {
      return 0;
    }
    // Move the array pointer to the next element
    arrayPtr = arrayPtr + 1;
  // Continue until a non-zero element is found
  } while (*arrayPtr == 0);
  
  // Return the first non-zero value found in the array
  return *arrayPtr;
}
