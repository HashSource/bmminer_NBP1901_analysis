void DumpNonceInfoForAllCores(void)
{
  int numAsics;
  int chainIndex;
  int totalNoncesByChain;
  int noncesBufferOffset;
  int totalNonceCount;
  int asicIndex;
  int nonceCountForASIC;
  int *nonceArrayPointer;
  int totalAsicsInChain;
  char *logMessage;
  char nonceInfoBuffer[260]; // Buffer to store the log messages
  
  noncesBufferOffset = 0;
  chainIndex = 0;
  totalNoncesByChain = 0;
  // Clear out the buffer to hold the textual nonce information
  memset(nonceInfoBuffer,0,sizeof(nonceInfoBuffer));
  do {
    // Check if any ASICs are present for this chain
    numAsics = GetNumAsicsInChain(chainIndex);
    if (numAsics != 0) {
      // Lazily initialize the logger named "bmminer_nonce"
      if (logDescriptor == 0) {
        logDescriptor = InitializeLogger("bmminer_nonce");
      }
      // Log chain information
      LogInfo(logDescriptor, "asic_response.c", 0xf, "dump_nonce_info_all_core", 0x18, 0x21a, 0x14,
              "\n ASIC nonce number for chain[%d]...\n", chainIndex);

      // Get total nonces found by all ASICs in this chain (total nonces counter)
      totalAsicsInChain = GetTotalAsicsInChain();
      if (0 < totalAsicsInChain) {
        // Get a pointer to the nonce counter array for the current chain
        nonceArrayPointer = (int *)(nonceCounterArrayBase + chainIndex * 0x400);
        asicIndex = 0;
        do {
          nonceArrayPointer++; // Move to the next ASIC's nonce count
          asicIndex++;
          totalNoncesByChain += *nonceArrayPointer;
          // Write nonce information for the current ASIC into the buffer
          noncesBufferOffset += snprintf(nonceInfoBuffer + noncesBufferOffset, sizeof(nonceInfoBuffer) - noncesBufferOffset, "ASIC[%03d]=%-8d", asicIndex, *nonceArrayPointer);
          
          // Retrieve the current domain number (may be related to work distribution or current job)
          GetDomainNumber();
          UpdateDomainNonceCounter(asicIndex);
          // Check if the buffer is ready to be logged
          if (extraout_r1 == 0) {
            if (logDescriptor == 0) {
              logDescriptor = InitializeLogger("bmminer_nonce");
            }
            // Log the nonce information that was buffered
            logMessage = nonceInfoBuffer;
            domainNumber = GetDomainNumber();
            LogInfoFormatted(asicIndex, domainNumber, "ASIC domain %02d %s total=%-8u", domainNumber, logMessage, totalNoncesByChain);
            // Reset counters and buffer offset for next iteration
            totalNoncesByChain = 0;
            noncesBufferOffset = 0;
          }
          // Increment nonce-index
          asicIndex = asicIndex;
        } while (asicIndex != totalAsicsInChain); // Continue until all ASICs for the chain have been processed
      }
    }
    chainIndex++; // Move to next chain
  } while (chainIndex != 4); // There are 4 chains on the system
  return;
}
