```c
#include <stdbool.h> // Include standard boolean definitions
#include <string.h>  // Include string handling functions

/**
 * Check if two strings are identical.
 *
 * This function compares two strings to determine if they are exactly the same.
 *
 * @param str1 The first string to compare.
 * @param str2 The second string to compare.
 *
 * @return True if the strings are identical, false otherwise.
 */
bool areStringsIdentical(char *str1, char *str2)
{
  int comparisonResult;

  // Compare the two strings and store the result
  comparisonResult = strcmp(str1, str2);

  // If the result is zero, strings are identical
  return comparisonResult == 0;
}
```
