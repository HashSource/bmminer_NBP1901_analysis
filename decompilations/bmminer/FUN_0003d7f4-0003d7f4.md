char * GetNextToken(int dataBufferIndex, size_t *parsedLength)
{
  byte currentByte;
  size_t tokenLength;
  char *tokenStartPtr;
  
  currentByte = *(byte *)(dataBufferIndex + *parsedLength);
  // Check if the current byte is not an ASCII letter or it is an equal sign ('=')
  if ((currentByte & 0xdf) == 0 || currentByte == '=') {
    tokenStartPtr = NULL;
  }
  else {
    // Point to the start of the next token, skipping two characters
    tokenStartPtr = (char *)(dataBufferIndex + *parsedLength + 2);
    // Find the length of the token up to any of the delimiter characters ('|', '=', or ' ')
    tokenLength = strcspn(tokenStartPtr, "|= ");
    // Update the parsed length to the length of the token found
    *parsedLength = tokenLength;
  }
  // Return the start pointer of the token, or NULL if no valid token is found
  return tokenStartPtr;
}
