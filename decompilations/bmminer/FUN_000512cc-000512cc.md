void SetAntminerControlFlag(uint32_t minerID, uint controlFlag)
{
  uint32_t controlValue;

  // Read the current control value from the miner
  ReadMinerControlValue(0, minerID, 0, 0x18, &controlValue);
  
  // Modify the control flag in the control value
  // Maintain all bits except for the target flag bit (assumed to be at position 3) 
  // If controlFlag is 1, set bit 3; else clear bit 3.
  controlValue = (controlValue & 0xffffff07) | ((controlFlag & 1) << 3);
  
  // Write the new control value to the miner
  WriteMinerControlValue(minerID, 1, 0, 0x18, controlValue);

  // Function complete, return to caller
  return;
}
