#include <stdio.h>

/* Global declarations (previously _DAT_0007ede0, DAT_00080fa8, DAT_00080fac, DAT_0007eb9c) */
extern unsigned int globalPulseConfig;
extern unsigned char globalPulseWidthSel;
extern unsigned char globalCcDlySel;
extern unsigned char globalDebugLevel;

/* Function to update and log pulse configuration */
unsigned int UpdateAndLogPulseConfig(void) {
  char logBuffer[2052]; // Buffer for logging
  
  // Update global pulse configuration
  globalPulseConfig = (globalPulseConfig & 0xFFFF00FF) | ((unsigned int)globalPulseWidthSel << 8);
  globalPulseConfig = (globalPulseConfig & 0xFF00FFFF) | ((unsigned int)globalCcDlySel << 16);
  
  // If the debug level is greater than 3, log the pulse configuration
  if (globalDebugLevel > 3) {
    // Prepare the log message
    snprintf(logBuffer, 0x800, "pulse_mode = %d, ccdly_sel = %d, pwth_sel = %d\n",
             globalPulseConfig & 0xFF, globalCcDlySel & 0xFF, globalPulseWidthSel & 0xFF);
    
    // Log the message (previously FUN_0002e584)
    LogDebugMessage(3, logBuffer, 0);
  }
  
  // Return the updated pulse configuration (excluding the top byte)
  return globalPulseConfig & 0xFFFFFF;
}
