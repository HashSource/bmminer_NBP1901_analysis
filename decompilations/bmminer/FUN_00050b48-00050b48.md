void SetASICFrequency(undefined4 chip_id, uint frequency)
{
  byte frequency_trim_value;
  undefined4 frequency_register_value;
  
  // Read current frequency register value
  ReadASICRegister(0, chip_id, 0, 0x58, &frequency_register_value);
  
  // Extract the frequency trim value (lower 4 bits of the second byte)
  frequency_trim_value = ((byte)frequency_register_value >> 8) & 0x0F;
  
  // Combine the new frequency with the old trim value and update the register
  UpdateASICFrequency(chip_id, 1, 0, 0x58, frequency_register_value);
  
  return;
}
