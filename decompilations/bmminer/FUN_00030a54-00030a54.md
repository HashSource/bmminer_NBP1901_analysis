```c
#include <sys/prctl.h>

void setThreadName(unsigned int thread_id)
{
  char thread_name_buffer[24];
  
  // Create a thread name string using the provided thread ID
  // The thread name will have a format like "cg@1", "cg@2", etc.
  snprintf(thread_name_buffer, sizeof(thread_name_buffer), "cg@%u", thread_id);

  // Rename the thread using prctl() system call
  // PR_SET_NAME (0xf) is used to set the name of the calling thread
  prctl(PR_SET_NAME, thread_name_buffer, 0, 0, 0);

  return;
}
```
