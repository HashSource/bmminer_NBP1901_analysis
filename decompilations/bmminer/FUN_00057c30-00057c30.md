// Initializes logging levels for the system
int InitializeLoggingLevels(void)
{
  int logLevelsList;
  int setupResult;
  
  // Create a new array list for logging levels using a custom destructor
  logLevelsList = CreateArrayList(CustomLogLevelsDestructor);
  if (logLevelsList == 0) {
    LogError(2, "src/level_list.c", 0x3e, "ArrayList creation failed");
  }
  else {
    // Add logging levels to the list with corresponding syslog constants
    setupResult = AddLogLevel(logLevelsList, "* = 0, LOG_INFO");
    if ((((((setupResult != 0) || (setupResult = AddLogLevel(logLevelsList, "DEBUG = 20, LOG_DEBUG"), setupResult != 0)) ||
          (setupResult = AddLogLevel(logLevelsList, "INFO = 40, LOG_INFO"), setupResult != 0)) ||
         ((setupResult = AddLogLevel(logLevelsList, "NOTICE = 60, LOG_NOTICE"), setupResult != 0 ||
          (setupResult = AddLogLevel(logLevelsList, "WARN = 80, LOG_WARNING"), setupResult != 0)))) ||
        ((setupResult = AddLogLevel(logLevelsList, "ERROR = 100, LOG_ERR"), setupResult != 0 ||
         ((setupResult = AddLogLevel(logLevelsList, "FATAL = 120, LOG_ALERT"), setupResult != 0 ||
          (setupResult = AddLogLevel(logLevelsList, "UNKNOWN = 254, LOG_ERR"), setupResult != 0)))))) ||
       (setupResult = AddLogLevel(logLevelsList, "! = 255, LOG_INFO"), setupResult != 0)) {
      // If any of these additions fail, log an error and clean up
      LogError(2, "src/level_list.c", 0x43, "Default log level setup failed");
      DestroyArrayList(logLevelsList);
      logLevelsList = 0;
    }
  }
  // Return the handle to the list; or 0 if there was an error
  return logLevelsList;
}
