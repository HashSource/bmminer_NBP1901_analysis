bool CheckIfReadRegComplete(int chain_index)
{
  int reg_size;
  char *reg_start_ptr;
  char *reg_end_ptr;
  char *current_ptr;
  int populated_reg_count; // Number of non-zero entries in the register
  char debug_message[2048];
  
  // Get the size of the register for the given chain
  reg_size = *(int *)(GlobalRegisterBaseAddress + 4);
  
  // If DEBUG level is higher than a certain threshold, log a message
  if (DebugLevel > 4) {
    snprintf(debug_message, 0x800,
             "[DEBUG] Check if read reg done, chain = %d, done_threshold = %d.\n",
             chain_index, 1);
    LogDebugMessage(4, debug_message, 0);
  }

  // Get the size of the populated part of the register
  int populated_size = GetPopulatedRegSize();
  
  // If the register has data populated in it
  if (populated_size != 0) {
    // Calculate the pointer to the start and end of the populated section
    reg_start_ptr = (char *)(GlobalRegisterBaseAddress + chain_index * 0x100);
    reg_end_ptr = reg_start_ptr + populated_size;

    // Count all non-zero entries in the populated part of the register
    populated_reg_count = 0;
    current_ptr = reg_start_ptr;
    do {
      if (*current_ptr != '\0') {
        populated_reg_count++;
      }
      current_ptr++;
    } while (current_ptr != reg_end_ptr);
  }

  // If populated size and the number of non-zero entries match, return true
  return populated_size == populated_reg_count;
}
