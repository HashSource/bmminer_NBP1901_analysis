uint CheckIfNegativeRule(char *ruleString)
{
  uint isNegativeRule;
  
  // Check if the input pointer is NULL
  if (ruleString == NULL) {
    LogError(2, "src/rule.c", 0x3ea, "ruleString is null"); // Log an error if pointer is null
    return 0xffffffff; // Return an error code
  }
  // Check if the rule string starts with '!' and is followed by a null terminator
  // which would indicate a negative rule
  if (*ruleString == '!') {
    isNegativeRule = (uint)(ruleString[1] == '\0');
  }
  else {
    isNegativeRule = 0; // Not a negative rule if it doesn't start with '!'
  }
  return isNegativeRule; // Return the result
}
