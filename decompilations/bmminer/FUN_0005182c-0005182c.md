void SetFanControlMode(uint32_t fanControllerAddress, int enableAutomaticMode)
{
  uint8_t fanControlRegisterValue;
  uint32_t fanControlRegister;
  
  // Read the current value of the fan control register
  ReadFromHardware(fanControllerAddress, 0, 0, 0x34, &fanControlRegister);
  
  // Modify the fan control mode bit based on the enableAutomaticMode flag
  if (enableAutomaticMode == 0) {
    // Disable automatic mode (the 4th bit of the register should be cleared)
    fanControlRegisterValue = (uint8_t)fanControlRegister & 0xF7;
  }
  else {
    // Enable automatic mode (the 4th bit of the register should be set)
    fanControlRegisterValue = (uint8_t)fanControlRegister | 0x08;
  }
  
  // Update the fan control register with the new value
  fanControlRegister = (fanControlRegister & 0xFFFFFF00) | fanControlRegisterValue;
  
  // Write the updated value back to the hardware
  WriteToHardware(fanControllerAddress, 1, 0, 0x34, fanControlRegister);
  return;
}
