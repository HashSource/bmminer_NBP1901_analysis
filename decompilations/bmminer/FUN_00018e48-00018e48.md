void WriteToEEPROM(uint32_t address, uint32_t data, int isVerifiedWrite)
{
  uint32_t EEPROMWriteMessageBuffer[14]; // Buffer for constructing the message for EEPROM write
  
  // Only allow writes that are aligned to 32 bits and optionally check for a verified write
  if ((data & 3) == 0) {
    if (isVerifiedWrite == 0) {
      PerformVerifiedWriteToEEPROM();
      return;
    }
    InitiateEEPROMWriteSequence(address, data, 1);
    return;
  }
  
  // Logging unaligned write attempt if verbosity level is 4 or higher
  if (verbosityLevel < 4) {
    return;
  }
  
  // Fill the buffer with parts of the "EEPROM write data is not 32-bit aligned" message
  EEPROMWriteMessageBuffer[0] = EEPROM_UNALIGNED_WRITE_MSG[0];
  EEPROMWriteMessageBuffer[1] = EEPROM_UNALIGNED_WRITE_MSG[1];
  EEPROMWriteMessageBuffer[2] = EEPROM_UNALIGNED_WRITE_MSG[2];
  EEPROMWriteMessageBuffer[3] = EEPROM_UNALIGNED_WRITE_MSG[3];
  EEPROMWriteMessageBuffer[4] = EEPROM_UNALIGNED_WRITE_MSG[4];
  EEPROMWriteMessageBuffer[5] = EEPROM_UNALIGNED_WRITE_MSG[5];
  EEPROMWriteMessageBuffer[6] = EEPROM_UNALIGNED_WRITE_MSG[6];
  EEPROMWriteMessageBuffer[7] = EEPROM_UNALIGNED_WRITE_MSG[7];
  EEPROMWriteMessageBuffer[8] = EEPROM_UNALIGNED_WRITE_MSG[8];
  EEPROMWriteMessageBuffer[9] = EEPROM_UNALIGNED_WRITE_MSG[9];
  EEPROMWriteMessageBuffer[10] = EEPROM_UNALIGNED_WRITE_MSG[10];
  EEPROMWriteMessageBuffer[11] = EEPROM_UNALIGNED_WRITE_MSG[11];
  EEPROMWriteMessageBuffer[12] = EEPROM_UNALIGNED_WRITE_MSG[12];
  EEPROMWriteMessageBuffer[13] = EEPROM_UNALIGNED_WRITE_MSG[13];
  
  LogError(3, EEPROMWriteMessageBuffer, 0); // Log the unaligned write attempt as an error
  return;
}
