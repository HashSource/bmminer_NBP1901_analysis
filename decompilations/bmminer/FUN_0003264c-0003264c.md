void UpdateASICStatusAndLogging(void)
{
  int chainIndex;
  uint ASICIndex;
  undefined4 ASICDataHigh;
  uint ASICCount;
  int exitCode;
  int validNonces;
  uint ASICDataLow;
  undefined4 *ASICDataPointer;
  undefined4 *ASICBaseDataPointer;
  int maxNonceIndex;
  int currentNonceIndex;
  char logBuffer[2052];
  
  chainIndex = 0;
  do {
    exitCode = CheckIfChainExists(chainIndex);
    if (exitCode != 0) {
      ResetASIC(chainIndex);
      ASICIndex = 0;
      ASICDataPointer = &ASICBaseArray + chainIndex * 0x200;
      while( true ) {
        ASICCount = GetASICCount();
        ASICDataLow = LoggerDataVariable;
        if (ASICCount <= ASICIndex) break;
        ASICDataHigh = ASICDataPointer[1];
        exitCode = (chainIndex * 0x3c00 + ASICIndex * 0x3c + GlobalNonceIndex % 0x3c) * 8;
        *(undefined4 *)(&GlobalASICStatusArray + exitCode) = *ASICDataPointer;
        *(undefined4 *)(&GlobalASICStatusArray + exitCode + 4) = ASICDataHigh;
        if (4 < ASICDataLow) {
          snprintf(logBuffer, 0x800, "chain %u asic %u asic_nonce_num %llu\n", chainIndex, ASICIndex);
          LogData(4, logBuffer, 0);
        }
        ASICIndex = ASICIndex + 1;
        ASICDataPointer = ASICDataPointer + 2;
      }
    }
    chainIndex = chainIndex + 1;
  } while (chainIndex != 4);
  GlobalNonceIndex = GlobalNonceIndex + 1;
  
  memset(&NonceStatisticsArray1, 0, 0x2000);
  memset(&NonceStatisticsArray2, 0, 0x2000);
  memset(&NonceStatisticsArray3, 0, 0x2000);
  ASICDataPointer = &GlobalErrorCounterArray;
  ResetNonceStatistics(&NonceStatisticsArray1,10);
  ResetNonceStatistics(&NonceStatisticsArray2,30);
  chainIndex = 0;
  ResetNonceStatistics(&NonceStatisticsArray3,60);
  
  do {
    exitCode = CheckIfChainExists(chainIndex);
    if (exitCode != 0) {
      validNonces = 0;
      ASICIndex = 0;
      ASICBaseDataPointer = &ASICBaseArray + chainIndex * 0x200;
      while( true ) {
        ASICDataLow = GetASICCount();
        maxNonceIndex = validNonces + ASICIndex;
        if (ASICDataLow <= ASICIndex) break;
        ASICDataHigh = GetASICErrors();
        RecordASICErrors(ASICIndex, ASICDataHigh);
        currentNonceIndex = maxNonceIndex;
        if (exitCode == 0) {
          validNonces = validNonces + 1;
          currentNonceIndex = validNonces + ASICIndex;
          MarkNonceValid(chainIndex, maxNonceIndex, 0x20);
        }
        maxNonceIndex = GlobalNonceIndex % 0x3c + -1;
        if (maxNonceIndex < 0) {
          maxNonceIndex = GlobalNonceIndex % 0x3c + 0x3b;
        }
        if (*(int *)(&GlobalASICStatusArray + (chainIndex * 0x3c00 + ASICIndex * 0x3c + maxNonceIndex) * 8) < 2) {
          MarkNonceInvalid(chainIndex, currentNonceIndex, 0x78);
          ASICDataPointer[ASICIndex] = ASICDataPointer[ASICIndex] + 1;
        }
        else {
          MarkNonceValid(chainIndex, currentNonceIndex, 0x6f);
        }
        ASICIndex = ASICIndex + 1;
        *ASICBaseDataPointer = 0;
        ASICBaseDataPointer[1] = 0;
        ASICBaseDataPointer = ASICBaseDataPointer + 2;
      }
      ResetChainStatus(chainIndex, maxNonceIndex, 0);
    }
    chainIndex = chainIndex + 1;
    ASICDataPointer = ASICDataPointer + 0x100;
  } while (chainIndex != 4);
  return;
}
