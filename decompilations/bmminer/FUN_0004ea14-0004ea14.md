// Function to initialize a hash rate data structure if the provided value is valid
hash_rate_data_t * InitHashRateData(uint32_t low32bits, uint32_t high32bits)
{
  hash_rate_data_t *hashRateDataPtr;
  double hashRateValue;
  
  // Check if the given 64-bit hash rate value is not NaN and has valid magnitude
  if ((!NAN((double)CONCAT44(high32bits, low32bits)) && 
       !NAN((double)CONCAT44(high32bits, low32bits))) &&
     (hashRateValue = ABS((double)CONCAT44(high32bits, low32bits)),
     hashRateValue == THRESHOLD_HASH_RATE || 
      hashRateValue < THRESHOLD_HASH_RATE != (NAN(hashRateValue) || NAN(THRESHOLD_HASH_RATE)))) {
      
    // Allocate memory for the hash rate data structure
    hashRateDataPtr = (hash_rate_data_t *)AllocateMemory(sizeof(hash_rate_data_t));
    
    if (hashRateDataPtr != (hash_rate_data_t *)0x0) {
      // Initialize the hash rate data structure with provided values
      hashRateDataPtr->low32bits = low32bits;
      hashRateDataPtr->high32bits = high32bits;
      hashRateDataPtr->dataType = HASH_RATE_DATA_TYPE; // Set the data type to hash rate
      hashRateDataPtr->valid = HASH_RATE_VALID; // Mark the hash rate data as valid
    }
    
    return hashRateDataPtr;
  }
  
  // Return NULL if the provided hash rate value is not valid
  return (hash_rate_data_t *)0x0;
}
