#include <errno.h>
#include <stdio.h>

// Function to log initialization failure of a pthread mutex
void LogMutexInitFailure(char *sourceFile, int functionNameAddress)
{
  int *errnoPointer;
  char errorMessageBuffer[2052]; // Buffer to hold the error message

  errnoPointer = __errno_location(); // Get the location of the errno
  // Format the error message with the errno value, source file, and location
  snprintf(errorMessageBuffer, sizeof(errorMessageBuffer),
           "Failed to pthread_mutex_init errno=%d in %s %p():%d",
           *errnoPointer, sourceFile, functionNameAddress);
  // Log the error message with a severity level of 3 (Error) and exitFlag set to 1
  LogError(3, errorMessageBuffer, 1);
  return;
}
