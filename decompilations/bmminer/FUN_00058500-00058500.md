int InitializeRecordTable(void)
{
  int recordTableHandle;
  
  // Attempt to create a new hash table for records
  recordTableHandle = CreateHashTable(0x14, &RecordHashFunction, RecordCompareFunction, 0, RecordDestructorFunction);
  
  // If hash table creation fails, log an error message
  if (recordTableHandle == 0) {
    LogError(2, "src/record_table.c", 0x31, "Failed to create new record hash table");
  }
  else {
    // If successful, initialize record table with default values
    InitializeRecordTableWithDefaults(recordTableHandle, 0);
  }
  
  // Return handle to the new (or failed) record table
  return recordTableHandle;
}
