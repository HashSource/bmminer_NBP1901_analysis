// Function that checks and updates system settings
void UpdateSystemSettings(uint32_t setting_id, int32_t setting_handle) {
  int32_t operation_status;
  uint32_t setting_value;
  
  // If we have a valid setting handle, proceed to update the setting
  if ((setting_handle != 0) && (setting_value = setting_id, operation_status = CheckSettingValidity(setting_handle, &setting_value), operation_status == 1)) {
    ApplySettingUpdate(setting_handle, &setting_value);
    // Setting updated successfully, so we exit the function
    return;
  }
  // If the setting handle was not valid, or the setting was not updatable, do nothing
  return;
}
