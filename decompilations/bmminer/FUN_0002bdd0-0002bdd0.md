```c
#include <pthread.h>

// Global mutex for synchronization
pthread_mutex_t global_mutex = PTHREAD_MUTEX_INITIALIZER;

// Shared global variables
int total_success_count = 0;
int total_failure_count = 0;
int total_retries_count = 0;

// Initializes shared global counters to zero
void InitializeCounters(void)
{
  // Lock the mutex before making changes to the shared variables
  pthread_mutex_lock(&global_mutex);
  
  // Reset counters to zero
  total_success_count = 0;
  total_failure_count = 0;
  total_retries_count = 0;
  
  // Unlock the mutex after changes are made
  pthread_mutex_unlock(&global_mutex);
}
```
