void ConvertBinaryToHexadecimalString(int binaryDataPtr, int dataSize)
{
  // Initialize variables for the conversion process
  char *hexStringPtr;
  uint outputSize;
  char *hexDigitPtr;
  byte *binaryDataCurrent;
  byte *binaryDataEnd;
  uint paddingSize;
  
  // Calculate the size needed for the output string with padding to align to 4 bytes
  outputSize = dataSize * 2 + 1; // Each byte of data becomes two hex digits, plus the null-terminator
  paddingSize = outputSize & 3; // Calculate padding for 4-byte alignment
  if ((int)outputSize < 1) {
    paddingSize = -(-outputSize & 3);
  }
  
  // Allocate memory for the hex string with proper padding
  hexStringPtr = (char *)AllocateMemoryForString((4 - paddingSize) + outputSize, 1, "util.c", GLOBAL_DATA_HEXSTRING_FILE_IDENTIFIER, 0x315);
  
  // If there is data to convert, process each byte of binary data
  if (0 < dataSize) {
    binaryDataCurrent = (byte *)(binaryDataPtr - 1); // Set the pointer to the byte before the binary data
    binaryDataEnd = binaryDataCurrent + dataSize; // Calculate the end of the binary data
    hexDigitPtr = hexStringPtr; // Start of destination for the hexadecimal string
    do {
      binaryDataCurrent++; // Move to the next byte of binary data
      // Convert the high nibble (4 bits) to a hexadecimal digit and store it in the string
      *hexDigitPtr = "0123456789abcdef"[*binaryDataCurrent >> 4];
      // Convert the low nibble to a hexadecimal digit and store it in the string
      hexDigitPtr[1] = "0123456789abcdef"[*binaryDataCurrent & 0xf];
      // Move to the next pair of hex digits in the string
      hexDigitPtr += 2;
    } while (binaryDataCurrent != binaryDataEnd); // Continue until all binary data is converted
    
    // Move the hex string pointer to the end of the converted data
    hexStringPtr += dataSize * 2;
  }
  
  // Null-terminate the hexadecimal string
  *hexStringPtr = '\0';
  
  return;
}
