/*
 * Function Purpose: Update status message based on the mining hardware condition.
 * param_1: Buffer to store the status message.
 * param_2: Size of the buffer.
 * param_3: Error code to include in the message if check fails.
 *
 * Returns: Always returns 0 indicating success on updating the status message.
 */
 
uint32_t UpdateMiningStatus(char *statusBuffer, size_t bufferLength, uint32_t errorCode)
{
  int minerStatusCheck;
  uint32_t lastError;
  
  // Check the status of the miner hardware
  minerStatusCheck = CheckMinerHardwareStatus();
  
  // If miner hardware is in a good state, format the output with the given error code
  if (minerStatusCheck == 0) {
    snprintf(statusBuffer, bufferLength, "%d", errorCode);
    return 0;
  }
  
  // If miner hardware status is in a specific failing state, format the output accordingly
  if (minerStatusCheck != 1) {
    snprintf(statusBuffer, bufferLength, "searching");
    return 0;
  }
  
  // Retrieve the last error message from the miner hardware
  lastError = GetLastMinerHardwareError();
  
  // Format the status buffer with the "searchfailed" string and the last error
  snprintf(statusBuffer, bufferLength, "%s:%s", "searchfailed", lastError);
  return 0;
}
