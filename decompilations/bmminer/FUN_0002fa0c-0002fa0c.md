int CompareHashToTarget(uint *hash, uint *target) {
  void *hash_str;
  void *target_str;
  uint *hash_ptr;
  uint *target_ptr;
  char *result_str;
  int result;
  uint hash_value;
  uint target_value;
  uint hash_be[8]; // BE stands for Big Endian
  char compare_result_buffer[2052];
  
  // Compare hash to the target
  hash_ptr = hash + 8;
  target_ptr = target + 8;
  do {
    hash_ptr--;
    target_ptr--;
    if (*target_ptr < *hash_ptr) {
      result = 0;
      goto DONE;
    }
  } while ((*target_ptr <= *hash_ptr) && (hash != hash_ptr));
  result = 1;
DONE:
  // If debugging is enabled
  if (DEBUG_FLAG != '\0') {
    // Convert hash and target from little endian to big endian for display
    for (int i = 0; i < 8; i++) {
      hash_value = hash[i];
      hash_be[i] = hash_value >> 24 | (hash_value & 0xff) << 24 | (hash_value & 0xff00) << 8 | (hash_value >> 8 & 0xff00);
      target_value = target[i];
      hash_be[i + 8] = target_value >> 24 | (target_value & 0xff) << 24 | (target_value & 0xff00) << 8 | (target_value >> 8 & 0xff00);
    }
  
    // Call a function to convert big endian hash and target to strings
    hash_str = ConvertToHexString(&hash_be[0], 0x20);
    target_str = ConvertToHexString(&hash_be[8], 0x20);

    // If additional debugging is enabled, output result
    if ((DEBUG_VERBOSE_FLAG != '\0') && ((SOME_STATUS_FLAG != '\0') || (CRITICAL_VALUE > 6))) {
      result_str = "YES (hash <= target)";
      if (result == 0) {
        result_str = "no (false positive; hash > target)";
      }
      // Log the comparison and results
      snprintf(compare_result_buffer, 0x800, "Proof: %s\nTarget: %s\nTrgVal? %s", hash_str, target_str, result_str);
      LogOutput(7, compare_result_buffer, 0);
    }

    // Free the allocated strings
    free(hash_str);
    free(target_str);
    return result;
  }
  return result;
}
