uint32_t check_and_init_hardware(int *hardware_status_ptr)
{
  uint32_t init_result;
  
  // Check if the hardware status pointer is not NULL and the status is zero (not initialized)
  if ((hardware_status_ptr != NULL) && (*hardware_status_ptr == 0)) {
    // Initialize hardware and store the result
    init_result = initialize_hardware(hardware_status_ptr + 2);
    // Return the initialization result
    return init_result;
  }
  // Return 0 if the hardware status pointer was NULL or if the hardware was already initialized
  return 0;
}
