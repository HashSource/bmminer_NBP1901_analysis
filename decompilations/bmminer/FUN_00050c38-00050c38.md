void ConfigureChipFrequencyMode(uint32_t chipAddress, int frequencyMode)
{
  uint32_t chipConfigData[2];
  
  // Read chip configuration into chipConfigData
  ReadChipConfiguration(0, chipAddress, 0, 0x18, chipConfigData);

  // If frequencyMode is 0, set the chip to low-frequency mode
  if (frequencyMode == 0) {
    // Set bits 28-31 to 1111 and clear bit 22
    chipConfigData[0] = (chipConfigData[0] & 0xFFBFFFFF) | 0xF0000000;
  } else { // Else, set the chip to high-frequency mode
    // Clear bits 0-27 and set bit 22
    chipConfigData[0] = (chipConfigData[0] & 0xFFFFFFF) | 0x400000;
  }

  // Write the modified configuration back to the chip
  WriteChipConfiguration(chipAddress, 1, 0, 0x18, chipConfigData[0]);

  return;
}
