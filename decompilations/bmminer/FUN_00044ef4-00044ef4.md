bool checkHashboardPresence(uint hashboardIndex)
{
  uint hashboardStatus;
  
  getHashboardStatus(3, &hashboardStatus); // Retrieve the status of all connected hashboards
  // Check if the particular hashboard indexed by 'hashboardIndex' is present
  // by checking if the corresponding bit is set in the 'hashboardStatus' variable.
  return (hashboardStatus & (1 << (hashboardIndex & 0xff))) != 0;
}
