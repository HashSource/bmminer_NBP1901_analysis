void FreeAllAllocatedMemory(void)
{
  int totalAllocations;
  void *memoryTrackerPtr;
  void *memoryToFree;
  int *memoryOffsetPtr;
  int allocationIndex;
  
  // Get the total number of memory allocations made, stored in a global variable or provided by another function
  totalAllocations = GetTotalAllocations();
  
  // Retrieve the global pointer tracking all allocated memory blocks
  memoryTrackerPtr = g_AllocatedMemoryTracker;
  
  if (totalAllocations != 0) { // If there are allocations to free
    allocationIndex = 0; // Start from the first allocation
    do {
      // Calculate the offset pointer for the current allocation
      memoryOffsetPtr = g_MemoryOffsetsArray + allocationIndex;
      allocationIndex = allocationIndex + 1;

      // Acquire the pointer to the memory block that needs to be freed
      memoryToFree = *(void **)((int)memoryTrackerPtr + (int)memoryOffsetPtr * 4 + 4);
      
      if (memoryToFree != NULL) { // If the memory block pointer is not null
        // Free the allocated memory block
        free(memoryToFree);
        // Update the memory tracker as the memory has been freed
        memoryTrackerPtr = g_AllocatedMemoryTracker;
        // Set the corresponding memory block pointer in the tracker to null
        *(void **)((int)g_AllocatedMemoryTracker + (int)memoryOffsetPtr * 4 + 4) = NULL;
      }
    } while (totalAllocations != allocationIndex); // Continue until all allocations have been processed
  }

  // Free the memory tracker itself
  free(memoryTrackerPtr);
  // Set the global memory tracker pointer to null as it's been freed
  g_AllocatedMemoryTracker = NULL;
  
  return;
}
