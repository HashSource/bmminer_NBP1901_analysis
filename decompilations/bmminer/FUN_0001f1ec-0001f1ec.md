void ConfigureMinerHardware(undefined4 hwConfigParam, int enableMonitoring)
{
  int initializationStatus;
  uint minerIndex;
  
  minerIndex = 0;
  initializationStatus = InitializeHardware(0); // Initialize the first miner hardware
  while( true ) {
    if (initializationStatus != 0) {
      SetHardwarePowerState(minerIndex & 0xff, POWER_ON); // Power on the miner if initialization was successful
      if (enableMonitoring == 0) {
        DisableHardwareMonitoring(minerIndex & 0xff); // Disable monitoring if requested
      }
      else {
        EnableHardwareMonitoring(); // Enable monitoring otherwise
      }
    }
    minerIndex += 1; // Move to the next miner index
    if (minerIndex == 4) break; // We assume there are 4 miners to configure; exit loop after the last one
    initializationStatus = InitializeHardware(minerIndex); // Initialize the next miner hardware
  }
  return;
}
