void ConfigureMinerHardwareErrorRegulation(uint32_t config_register_address, uint32_t error_register_address, int should_enable_error_regulation)
{
  uint32_t register_value;
  
  // Read the value from the hardware error configuration register
  ReadASICConfiguration(1, config_register_address, error_register_address, 0x18, &register_value);
  
  // Depending on the should_enable_error_regulation input, set or clear the error regulation bits
  if (should_enable_error_regulation == 0) {
    // Disable error regulation
    register_value = (register_value & 0xffbfffff) | 0xf0000000;
  }
  else {
    // Enable error regulation
    register_value = (register_value & 0xfffffff) | 0x400000;
  }
  
  // Write the modified value back to the hardware error configuration register
  WriteASICConfiguration(config_register_address, 0, error_register_address, 0x18, register_value);
  return;
}
