```c
// Function to compare two integers by subtracting the value pointed by the second integer pointer
// from the value pointed by the first integer pointer.
// This is usually used for comparison purposes in sorting algorithms like qsort.
// Returns the resulting difference.
// Inputs:
//   pointerToFirstInt - pointer to the first integer
//   pointerToSecondInt - pointer to the second integer
// Outputs:
//   Result of the subtraction
int CompareIntegers(int *pointerToFirstInt, int *pointerToSecondInt) {
  // Return the difference between the two integer values pointed by the input pointers.
  return *pointerToFirstInt - *pointerToSecondInt;
}
```
