void RemoveTailElementFromList(struct List *list, char *functionName, char *fileName, int lineNumber)
{
  char errorLog[2052]; // Buffer for the error log message

  // If the list's do_tail is not set to true, log an error and exit
  if (!list->do_tail) {
    snprintf(
      errorLog,
      sizeof(errorLog),
      "List %s can't %s() - do_tail is false - from %s %s() line %d in %s %s():%d",
      list->name,
      list->operationName,
      functionName,
      fileName,
      lineNumber,
      "klist.c",
      list->operationName,
      0xb5
    );
    LogError(3, errorLog); // Log error with severity level 3
    ExitApplication(1); // Exit the application with status code 1
  }

  // Remove the tail element from the list if it's not empty
  int tailNodePtr = list->tail;
  if (tailNodePtr != 0) {
    int prevNodePtr = *(int *)(tailNodePtr + 4);
    list->tail = prevNodePtr;
    if (prevNodePtr == 0) {
      list->head = 0; // If the list is empty after removal, set head to null
    } else {
      *(int *)(prevNodePtr + 8) = 0; // Update previous node's next pointer to null
    }
    *(int *)(tailNodePtr + 8) = 0; // Clear the removed node's next pointer
    *(int *)(tailNodePtr + 4) = 0; // Clear the removed node's previous pointer
    list->nodeCount--; // Decrement the number of nodes in the list
  }
  return;
}
