/* Global variables */
extern int GLOBAL_DEBUG_LEVEL;
extern int GLOBAL_MINER_HANDLE;
extern char GLOBAL_DEBUG_CLEAR_CONTEXT_STR[];
extern char GLOBAL_HANDLE_IS_NULL_STR[];
extern char ram0x00067bd8;
extern char ram0x00067efc;
extern void logMessage(int, char *, int);

/* Function: ClearMinerContext */
void clearMinerContext(void)
{
  int handle;
  char logMessageBuffer[2048]; // Allocate buffer for logging message

  // Only log if the debug level is higher than 4
  if (GLOBAL_DEBUG_LEVEL > 4) {
    // Prepare log message "Clear ctxt" from a global debug string
    logMessageBuffer[0] = GLOBAL_DEBUG_CLEAR_CONTEXT_STR[0];
    logMessageBuffer[1] = GLOBAL_DEBUG_CLEAR_CONTEXT_STR[1];
    logMessageBuffer[2] = GLOBAL_DEBUG_CLEAR_CONTEXT_STR[2];
    logMessageBuffer[3] = GLOBAL_DEBUG_CLEAR_CONTEXT_STR[3];
    logMessageBuffer[4] = GLOBAL_DEBUG_CLEAR_CONTEXT_STR[4];
    logMessageBuffer[5] = GLOBAL_DEBUG_CLEAR_CONTEXT_STR[5];
    logMessageBuffer[6] = GLOBAL_DEBUG_CLEAR_CONTEXT_STR[6];
    logMessageBuffer[7] = GLOBAL_DEBUG_CLEAR_CONTEXT_STR[7];
    logMessageBuffer[8] = GLOBAL_DEBUG_CLEAR_CONTEXT_STR[8];
    logMessageBuffer[9] = GLOBAL_DEBUG_CLEAR_CONTEXT_STR[9];
    logMessageBuffer[10] = GLOBAL_DEBUG_CLEAR_CONTEXT_STR[10];
    logMessageBuffer[11] = GLOBAL_DEBUG_CLEAR_CONTEXT_STR[11];
    logMessageBuffer[12] = GLOBAL_DEBUG_CLEAR_CONTEXT_STR[12];
    logMessageBuffer[13] = GLOBAL_DEBUG_CLEAR_CONTEXT_STR[13];
    logMessageBuffer[14] = GLOBAL_DEBUG_CLEAR_CONTEXT_STR[14];
    logMessageBuffer[15] = GLOBAL_DEBUG_CLEAR_CONTEXT_STR[15];
    logMessageBuffer[16] = GLOBAL_DEBUG_CLEAR_CONTEXT_STR[16];
    logMessageBuffer[17] = ram0x00067efc; // Additional byte from global data

    // Log the message with the debug level and message buffer
    logMessage(4, logMessageBuffer, 0);
  }

  handle = GLOBAL_MINER_HANDLE;
  // Check if the global miner handle is null or points to null
  if ((GLOBAL_MINER_HANDLE == 0) || (*(int *)(GLOBAL_MINER_HANDLE + 4) == 0)) {
    // Only log if the debug level is higher than 3
    if (GLOBAL_DEBUG_LEVEL > 3) {
      // Prepare log message "Handle is NULL" from a global error string
      for (int i = 0; i < 16; ++i) {
        logMessageBuffer[i] = GLOBAL_HANDLE_IS_NULL_STR[i];
      }
      logMessageBuffer[16] = ram0x00067bd8; // Additional byte from global data

      // Log the message with the debug level and message buffer
      logMessage(3, logMessageBuffer, 0);
      return;
    }
  }
  else {
    // Clear the context in the miner handle by setting all fields to 0
    for (int offset = 0x18; offset <= 0x3c; offset += 4) {
      *(int *)(handle + offset) = 0;
    }
  }
  return;
}
