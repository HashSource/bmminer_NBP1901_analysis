void SetFanSpeed(undefined4 fan_controller, uint fan_index, uint speed_percent)
{
  // Configure fan speed by sending appropriate command to fan controller
  // fan_controller: The reference to the fan controller object/device
  // fan_index: The index of the fan whose speed is to be adjusted (0, 1, 2...)
  // speed_percent: The desired speed of the fan expressed as a percentage (0-100)

  // Send fan speed control command:
  // - The first argument is the fan controller.
  // - The second argument set to 1 means we're configuring fan speed.
  // - Third and fourth arguments set to 0 are not used here.
  // - The last argument constructs the command to set the fan speed:
  //    - 0x80008700 is the base command.
  //    - (fan_index & 3) << 4 places the fan index into the command.
  //    - (speed_percent & 7) adds the speed percentage to the command.
  //    - The bitwise operations ensure that the values fit within expected bit ranges.
  SendFanControlCommand(fan_controller, 1, 0, 0x3c, (fan_index & 3) << 4 | speed_percent & 7 | 0x80008700);
  return;
}
