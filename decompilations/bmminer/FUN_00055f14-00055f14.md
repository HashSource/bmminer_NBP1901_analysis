void deleteCategoryEntries(int categoryArrayPtr)
{
  int currentEntryPtr;
  
  // If the given category array pointer is NULL, log an error and exit function
  if (categoryArrayPtr == 0) {
    logErrorToFileWithLine(2,"src/category_table.c",0x5b,"categories is null or 0");
    return;
  }

  // Get the first category entry to start the deletion process
  currentEntryPtr = getFirstCategoryEntry();

  // If there's no entry, no further action is needed
  if (currentEntryPtr == 0) {
    return;
  }

  // Loop through all category entries and delete each one
  do {
    deleteEntryByPtr(*(undefined4 *)(currentEntryPtr + 8));
    currentEntryPtr = getNextCategoryEntry(categoryArrayPtr, currentEntryPtr);
  } while (currentEntryPtr != 0); // Continue until there are no more entries

  return;
}
