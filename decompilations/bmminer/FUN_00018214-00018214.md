#include <stdio.h>

// Define error code for temperature read failure
#define TEMP_READ_FAILURE 0xffffffff

// Log a temperature reading error for a specific chain on the mining hardware
// Param: chain_id - Identifier for the specific mining hardware chain
// Returns: Error code for the failure
unsigned int logTemperatureError(unsigned int chain_id)
{
  // Allocate a buffer to hold the error message
  char errorMessageBuffer[2056];
  
  // Format the error message with the chain identifier
  snprintf(errorMessageBuffer, sizeof(errorMessageBuffer), "No temp, chain = %d.\n", chain_id);
  
  // Log the error message with a severity level of 4
  logErrorMessage(4, errorMessageBuffer, 0);
  
  // Return the error code for temperature read failure
  return TEMP_READ_FAILURE;
}
