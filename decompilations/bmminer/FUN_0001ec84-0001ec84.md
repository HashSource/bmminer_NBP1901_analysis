void sendCommandToSelectedMinerModule(uint8_t selectedModule, uint32_t command, uint32_t parameter1, uint32_t parameter2)
{
  uint32_t responseHigh;
  uint32_t minerModuleIdx;
  uint64_t response;
  
  minerModuleIdx = 0;                     // Initialize miner module index
  responseHigh = command;
  do {
    response = getMinerModuleStatus(minerModuleIdx, responseHigh);   // Get the status of the miner module
    // The higher 32 bits of the response contain additional data or status codes, 
    // whereas the lower 32 bits contain the main response information.
    responseHigh = (uint32_t)(response >> 32);              
    if (((int32_t)response != 0) &&             // If there is a non-zero response
        (selectedModule == minerModuleIdx ||    // And if specified module index matches the current index,
        selectedModule == 0xff)) {              // or if the command is to be sent to all modules (0xff)
      // Send a command to the miner module and pass all provided parameters
      sendCommandToMinerModule(minerModuleIdx & 0xff, command, parameter1, parameter2);  
      responseHigh = command;   // Prepare command for the next iteration/response check
    }
    minerModuleIdx++;  // Move to the next miner module index
  } while (minerModuleIdx != 4);  // Loop through all 4 miner modules
  return;
}
