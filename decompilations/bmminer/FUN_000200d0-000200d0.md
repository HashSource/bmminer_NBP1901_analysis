void processTemperatureData(int sensorType, uint chipIndex) {
  short temperatureRaw;
  undefined4 formattedTemperature;
  undefined4 timestamp;
  int maxChipCount;
  uint currentChip;
  char logBuffer[2052];
  
  // Check for invalid sensor type or chip index (index should be less than 4).
  if (sensorType == 0 || chipIndex > 3) {
    // Log error if specific debugging flags are set.
    if (((debugFlag1 != '\0') || (debugFlag2 != '\0')) || (debugThreshold < 2)) {
      snprintf(logBuffer, 0x800, "%s: input bad param\n", functionName);
      writeToLog(3, logBuffer, 0);
    }
    return;
  }
  
  // Get the current timestamp.
  timestamp = getCurrentTimestamp();

  // Initialize the current chip index based on conditions.
  currentChip = 0;
  if ((globalFlag == 1) || (maxChipCount = 0, totalChipCount < 1)) {
    // Process each chip's temperature for single chip configuration.
    while (true) {
      maxChipCount = getMaxChipCountSingle();
      if (maxChipCount <= (int)currentChip) break;
      temperatureRaw = getTemperatureRawSingle(chipIndex, currentChip & 0xff);
      currentChip += 1;
      formattedTemperature = formatTemperature(temperatureRaw, temperatureRaw >> 0x1f);
      timestampedTemperatureData(timestamp, formattedTemperature);
    }
  }
  else {
    // Process each chip's temperature for multiple chip configuration.
    while (true) {
      maxChipCount = getMaxChipCountMultiple();
      if (maxChipCount <= (int)currentChip) break;
      temperatureRaw = getTemperatureRawMultiple(chipIndex, currentChip & 0xff);
      formattedTemperature = formatTemperature(temperatureRaw, temperatureRaw >> 0x1f);
      timestampedTemperatureData(timestamp, formattedTemperature);
      currentChip += 1;
    }
  }
  
  // Write final temperature data to the appropriate sensor.
  writeTemperatureData(sensorType, "temp_chip", timestamp);
  return;
}
