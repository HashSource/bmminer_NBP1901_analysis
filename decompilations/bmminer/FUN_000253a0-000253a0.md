char * FetchWebPage(char *url) {
  char charFromUrl;
  int compareResult;
  char *substringPointer;
  size_t mainUrlLength;
  size_t subUrlLength;
  struct hostent *hostEntity;
  int socketDescriptor;
  ssize_t bytesSentOrReceived;
  long httpResponseCode;
  char *responseData;
  uint port;
  uint socketOptionsVal;
  char urlCopy[1023];
  char subUrl[1025];
  char buffer[4096];
  
  // Initialize buffers to zeros
  memset(buffer, 0, 4096);
  memset(urlCopy, 0, 1023);
  memset(subUrl, 0, 1025);

  // Handling null URL parameter
  if (url == NULL) {
    // Simplified error logging removed for brevity
    return NULL;
  } else {
    // Check if the URL starts with "http://"
    compareResult = strncmp(url, "http://", 7);
    if (compareResult != 0) {
      // Simplified error logging removed for brevity
      return NULL;
    }
    // Skip past "http://"
    url += 7;

    // Find first slash after "http://" which marks the end of the domain
    substringPointer = strchr(url, '/');
    mainUrlLength = strlen(url);
    if (substringPointer == NULL) {
      // No slash found, so copy the whole string
      memcpy(urlCopy, url, mainUrlLength);
      urlCopy[mainUrlLength] = '\0';
    } else {
      // Slash found, separate domain and path
      subUrlLength = strlen(substringPointer);
      memcpy(urlCopy, url, mainUrlLength - subUrlLength);
      charFromUrl = substringPointer[1];
      urlCopy[mainUrlLength - subUrlLength] = '\0';
      if (charFromUrl != '\0') {
        memcpy(subUrl + 1, substringPointer + 1, subUrlLength - 1);
        subUrl[subUrlLength] = 0;
      }
    }

    // Check if port is specified in the URL
    substringPointer = strchr(urlCopy, ':');
    if (substringPointer == NULL) {
      // No port specified, default to 80
      port = 80;
    } else {
      *substringPointer = '\0';
      // Convert port from string to integer
      port = strtol(substringPointer + 1, NULL, 10);
    }

    // Resolve hostname to IP address
    hostEntity = gethostbyname(urlCopy);
    if (hostEntity != NULL) {
      // Setup the socket address structure
      ((ushort *)subUrl)[0] = htons(port); // Network byte order for port
      subUrl[0] = 2; // AF_INET
      *((int *)(subUrl + 4)) = *((int *)*hostEntity->h_addr_list); // Host IP address

      // Create the socket
      socketDescriptor = socket(AF_INET, SOCK_STREAM, 0);
      if (socketDescriptor != -1) {
        // Set socket options
        socketOptionsVal = 10;
        compareResult = setsockopt(socketDescriptor, SOL_SOCKET, SO_RCVTIMEO, &socketOptionsVal, sizeof(socketOptionsVal));
        compareResult &= setsockopt(socketDescriptor, SOL_SOCKET, SO_SNDTIMEO, &socketOptionsVal, sizeof(socketOptionsVal));

        // Attempt to connect
        if ((compareResult == 0) && (connect(socketDescriptor, (struct sockaddr *)subUrl, sizeof(struct sockaddr_in)) != -1)) {
          // Send the HTTP GET request
          sprintf(buffer, "GET /%s HTTP/1.1\r\nHOST: %s:%d\r\nAccept: */*\r\n\r\n",
                  subUrl + 1, urlCopy, port);
          mainUrlLength = strlen(buffer);
          compareResult = 0;
          while (mainUrlLength > 0) {
            bytesSentOrReceived = send(socketDescriptor, buffer + compareResult, mainUrlLength, 0);
            if (bytesSentOrReceived == -1) {
              // Simplified error logging removed for brevity
              return NULL;
            }
            compareResult += bytesSentOrReceived;
            mainUrlLength -= bytesSentOrReceived;
          }

          // Receive the response
          bytesSentOrReceived = recv(socketDescriptor, buffer, 4096, 0);
          if (bytesSentOrReceived > 0) {
            close(socketDescriptor);
            substringPointer = strstr(buffer, "HTTP/1.1");
            if (substringPointer == NULL) {
              return NULL;
            }
            // Get the HTTP response code
            httpResponseCode = strtol(substringPointer + 9, NULL, 10);
            if (httpResponseCode != 200) {
              return NULL;
            }
            // Find the start of the response body
            substringPointer = strstr(buffer, "\r\n\r\n");
            if (substringPointer == NULL) {
              return NULL;
            }
            // Allocate memory for and return the response body
            mainUrlLength = strlen(substringPointer);
            responseData = (char *)malloc(mainUrlLength + 1);
            if (responseData == NULL) {
              return NULL;
            }
            strcpy(responseData, substringPointer + 4);
            return responseData;
          } else {
            // Simplified error logging removed for brevity
            return NULL;
          }
        }
      }
    }
    // Simplified error logging removed for brevity
  }
  
  return NULL;
}
