int getMinerTemperature(uint sensorId)
{
  int temperatureValue;
  uint tempVariable;
  uint sensorData[2];
  
  tempVariable = 0;
  temperatureValue = 6;
  readSensorStatus(0, sensorData);
  sensorData[0] = sensorData[0] & 0xffff; // Mask higher bits to consider only lower 16 bits
  
  do {
    readSensorStatus(1, &tempVariable);
    if ((tempVariable << 0x15) >> 0x1d == sensorId) {
      if (sensorData[0] == 0xb025) { // Check for a specific sensor model ID
        temperatureValue = (tempVariable & 0xff) * 0xf0; // Calculate temperature for model ID 0xb025
      }
      else {
        temperatureValue = (tempVariable & 0xff) * 0x78; // Calculate temperature for other models
      }
      return temperatureValue;
    }
    usleep(10000); // Wait for 10ms before next read
    temperatureValue = temperatureValue - 1;
  } while (temperatureValue != 0);
  
  return -1; // Return -1 if sensorId is not found after 6 attempts
}
