#include <stdlib.h> // Include standard library for the `free` function

// Function Prototype
void clearMinerDiagnosticContext(int *diagnosticContextPtr);

// Function to clear the diagnostic context for a Bitmain Antminer
void clearMinerDiagnosticContext(int *diagnosticContextPtr)
{
  // Check if the pointer to the diagnostic context is not null
  if (diagnosticContextPtr != NULL)
  {
    // Check if the diagnostic context pointed by the pointer is not zero
    if (*diagnosticContextPtr != 0)
    {
      // Call a function (presumably to clean up resources)
      cleanupResources();
    }
    // Log the deletion of the diagnostic context with a pointer value
    logEvent(0, "src/mdc.c", 0x25, "zlog_mdc_del[%p]", diagnosticContextPtr);
    // Free the memory allocated to the diagnostic context
    free(diagnosticContextPtr);
    return;
  }
  // Log an error if the diagnostic context pointer is null or points to zero
  logError(2, "src/mdc.c", 0x23, "diagnosticContextPtr is null or points to 0");
  return;
}
