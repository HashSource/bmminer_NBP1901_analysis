#include <pthread.h> // Include pthread header for pthread_rwlock_*

// Profiling and log global variables
// These are most likely used to keep state and control access between threads
extern pthread_rwlock_t g_logLock; 
extern int g_isLogInitialized;
extern int g_logInitVersion;
extern int g_defaultCategory;
extern int *g_categoryList;
extern int *g_ruleList;
extern int *g_bufferList;
extern void *g_rwlockAttribute; // Possibly a pthread_rwlockattr_t or similar

// Function declarations for logging and other operations
void log_message(int, const char *, int, const char *, ...);
void process_category_list(int *, int);
void process_rule_list(int *, int);
void process_buffer_list(int *, int);
void log_default_category(int, int);

void display_log_profile(void)
{
  int lockResult;
  
  // Attempt to acquire read lock
  lockResult = pthread_rwlock_rdlock(&g_logLock);
  if (lockResult != 0) {
    // Failed to acquire read lock, log error with details
    log_message(2,"src/zlog.c",0x3b4,"pthread_rwlock_wrlock fail, rc[%d]",lockResult);
    return;
  }
  
  // Begin logging the profile data
  log_message(1,"src/zlog.c",0x3b7,"------zlog_profile start------ ");
  log_message(1,"src/zlog.c",0x3b8,"is init:[%d]", g_isLogInitialized);
  log_message(1,"src/zlog.c",0x3b9,"init version:[%d]", g_logInitVersion);
  
  // Process and log details of category, rule, and buffer lists
  process_category_list(g_categoryList, 1);
  process_rule_list(g_ruleList, 1);
  process_buffer_list(g_bufferList, 1);
  
  // If default category is set, log details for it
  if (g_defaultCategory != 0) {
    log_message(1,"src/zlog.c",0x3be,"-default_category-", g_logInitVersion);
    log_default_category(g_defaultCategory, 1);
  }
  
  // Indicate end of log profile
  log_message(1,"src/zlog.c",0x3c1,"------zlog_profile end------ ");
  
  // Release the read lock
  lockResult = pthread_rwlock_unlock(g_rwlockAttribute);
  if (lockResult == 0) {
    return;
  }
  
  // Failed to release read lock, log error with details
  log_message(2,"src/zlog.c",0x3c4,"pthread_rwlock_unlock fail, rc=[%d]",lockResult);
  return;
}
