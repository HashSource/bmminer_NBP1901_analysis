// Function: GetNextNode
// Description: This function retrieves the next node in a data structure, 
// such as a linked list, where the current node is given as an argument.
// It returns NULL if there is no next node.
// 
// Parameters:
// - currentNodeAddr: Memory address of the current node.
//
// Returns: Memory address of the next node or NULL if at the end of the list.
int GetNextNode(int currentNodeAddr)
{
  int nextNodeAddr;
  
  // Get the address stored at offset 0x18 relative to the currentNodeAddr
  // This address points to the next node
  nextNodeAddr = *(int *)(currentNodeAddr + 0x18);
  
  // Check if the next node address is just pointing back to the current node address with offset 0x14
  // which may mean that this is the end of the list. If so, return NULL (i.e., 0)
  if (nextNodeAddr == currentNodeAddr + 0x14) {
    nextNodeAddr = 0;
  }
  
  return nextNodeAddr; // Return the address of the next node, or NULL if no more nodes follow
}
