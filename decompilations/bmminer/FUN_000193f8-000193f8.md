{
  double defaultVoltage;
  int chipIndex;
  int chipOperational;
  double currentVoltage;
  double minimalVoltage;
  double tempVoltage;
  char logBuffer[2048];
  
  defaultVoltage = DEFAULT_VOLTAGE; // A reference voltage value, probably declared globally
  chipIndex = 0; // Start from the first chip
  minimalVoltage = MIN_VOLTAGE; // A minimal voltage value, probably declared globally
  do {
    chipOperational = IsChipOperational(chipIndex); // Check if the chip at the index is operational
    tempVoltage = minimalVoltage;
    if (chipOperational != 0) {
      currentVoltage = (double)ReadChipVoltage(chipIndex); // Read the voltage of the current chip
      // If the current voltage is different from the default and is less than the default (checking for NaN in the values)
      if (currentVoltage != defaultVoltage && currentVoltage < defaultVoltage == (!isnan(currentVoltage) || !isnan(defaultVoltage))) {
        // If the minimal voltage is less or equal to the current voltage, and minimal voltage is less than the default voltage
        if (minimalVoltage <= currentVoltage && minimalVoltage < defaultVoltage) {
          tempVoltage = minimalVoltage;
        } else {
          // Update the minimal voltage if current voltage is less
          tempVoltage = currentVoltage;
        }
      }
    }
    // Move to the next chip
    chipIndex = chipIndex + 1;
    // Update the minimal voltage
    minimalVoltage = tempVoltage;
  } while (chipIndex != 4); // Assuming there are 4 chips to check

  }
  return minimalVoltage; // Return the minimal voltage found
}
