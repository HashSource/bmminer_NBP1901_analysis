// Function prototype for the log error
void logError(int level, char *file, int line, char *message);

// Function prototype for getting the first category from the category table.
int getFirstCategory();

// Function prototype for getting the next category from the category table.
int getNextCategory(int categoryTable, int currentCategory);

// Function prototype for updating rules for a given category.
int updateCategoryRules(int category, uint32_t someParam);

// Error code constant for failing operations.
#define ERROR_CODE (-1)

/**
 * Function to update rules for categories.
 * @param categoryTable A reference to the category table.
 * @param rulesParam A parameter defining rules.
 * @return A status code, 0 for success, -1 (ERROR_CODE) for failure.
 */
uint32_t updateCategoriesRules(int categoryTable, uint32_t rulesParam) {
  int currentCategory;
  int result;
  
  // Check if the category table reference is NULL or 0; if so, log error and return error code.
  if (categoryTable == 0) {
    logError(2, "src/category_table.c", 0x3e, "Category table is null or empty");
    return ERROR_CODE;
  }

  // Get the first category.
  currentCategory = getFirstCategory();

  // Iterate over categories and update rules.
  while (true) {
    if (currentCategory == 0) {
      return 0; // Success code 0 if all categories are traversed without issues.
    }

    // Call function to update rules for the current category.
    result = updateCategoryRules(*(uint32_t *)(currentCategory + 8), rulesParam);
    if (result != 0) {
      break; // Break loop if updating rules fails.
    }

    // Get the next category using current category as reference point.
    currentCategory = getNextCategory(categoryTable, currentCategory);
  }

  // If any category rule update fails, log error attempt to roll back.
  logError(2, "src/category_table.c", 0x42, "Failed to update category rules, attempt to roll back");
  return ERROR_CODE; // Return error code upon failure.
}
