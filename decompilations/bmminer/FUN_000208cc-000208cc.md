// SetMinerFrequency: Sets the minimum frequency for the Bitmain Antminer based on the EEPROM settings.
void SetMinerFrequency(uint32_t frequencyParam, uint32_t configParam) {
  uint32_t freqValue;
  
  // SetConfigParam: Sends configuration parameter to the mining hardware.
  SetConfigParam(configParam, 2);

  // ReadEEPROMFreq: Reads the frequency value from the EEPROM.
  freqValue = ReadEEPROMFreq(0, 0);

  // ApplyFrequencySetting: Applies the frequency setting to the mining hardware.
  ApplyFrequencySetting(frequencyParam, "eeprom_freq", freqValue);
  return;
}
