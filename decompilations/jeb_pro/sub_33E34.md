int InitializeMiningHardware(int initParam, int configParam, size_t configSize) {
    int logMessageBuffer;
    PerformInitialSetup();
    InitializeASICCommunication();
    int initResult = AllocateMemoryForMining(4000, 1, configSize);
    int finalResult = initResult;
    if(finalResult == 0) {
        // If the hardware revision is lower or equal than 3
        if(globalHardwareRevision <= 3) {
            finalResult = ConfigureASIC(initResult, configParam, configSize);
            if(finalResult != 0) {
                goto handleInitFailure;
            }
        }
        else {
            // Hardcoded check values for hardware, likely for verification
            logMessageBuffer = 544104774;
            int checkValue1 = 0x63656863;
            int checkValue2 = 1634738283;
            int checkValue3 = 0x64657373;
            short checkValue4 = 2606;
            char checkValue5 = 0;
            int checkResult = PerformHardwareCheck(3, &logMessageBuffer, 0);
            finalResult = ConfigureASIC(checkResult, configParam, configSize);
            if(finalResult != 0) {
            handleInitFailure:
                int hwCheckResult = PerformHardwareHealthCheck();
                if(hwCheckResult == 0 || globalHardwareCheckFlag == 0) {
                    return finalResult;
                }
            }
        }
        
        // If initialization failed, perform clean-up
        TerminateMiningThreads();
        ReleaseMiningMemory();
        if(globalMinerTypeFlag == 0) {
            int boardTypeParameter = (int)globalBoardType;
            if(boardTypeParameter > 0) {
                finalResult = ConfigureMiningBoard(boardTypeParameter, configParam, configSize);
                goto checkFinalResult;
            }
        }
        else {
            uint32_t modelIdentifier = GetHardwareModelIdentifier();
            int isNotModelBHB28601 = strcmp((char*)modelIdentifier, "BHB28601");
            finalResult = ConfigureMiningBoard(isNotModelBHB28601 != 0 ? 1500 : 1900, configParam, configSize);
        checkFinalResult:
            if(finalResult != 0) {
                LogError(12, 0);
                LogMessage(9, "get power type version failed!", configSize);
                return finalResult;
            }
        }

        // If hardware revision is higher than 3, log additional information
        if(globalHardwareRevision > 3) {
            int minTempValue = GetMinimumStartupTemperature();
            snprintf(&logMessageBuffer, 0x800, "start up min temp by 75a = %d", minTempValue);
            PerformHardwareCheck(3, &logMessageBuffer, 0);
        }

        // Perform remaining initialization steps
        InitializeTemperatureSensors();
        InitializeVoltageRegulators();
        InitializeCoolingSystem();
        DelayExecution(10);
        SetASICInitializationRoutine(&loc_1C200);
        DelayExecution(10);
        finalResult = FinalizeASICConfiguration();
        
        // If finalization is successful, reset flags and return success
        if(finalResult == 0) {
        resetAndReturnSuccess:
            ResetMiningState();
            finalResult = 0;
            globalInitializationFlag1 = 0;
            globalInitializationFlag2 = 0;
            globalInitializationFlag3 = 0;
        }
        // If there is a health check flag and it fails, retry
        else {
            int healthCheckRetryFlag = PerformHardwareHealthCheck();
            if(healthCheckRetryFlag != 0 && globalHardwareCheckFlag != 0) {
                goto resetAndReturnSuccess;
            }
        }
    }
    return finalResult;
}
