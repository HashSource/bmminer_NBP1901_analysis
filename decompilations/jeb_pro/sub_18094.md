short GetPCBVersion(int chainIndex) {
    char logBuffer;
    uint32_t gMinerConfigPtr = gvar_7F814; // Global pointer to Miner Config structure
    if (gMinerConfigPtr == 0 || *(char*)(gMinerConfigPtr + 28) == 0) {
        if (gvar_7EB9C <= 3) {
            return 0xffff;
        }
        snprintf(&logBuffer, 0x800, "No pcb version, chain = %d.\n", chainIndex);
        LogWarning(3, &logBuffer, 0);
        return 0xffff;
    }
    uint32_t chainConfigPtr = *(uint32_t*)(chainIndex * 4 + gMinerConfigPtr);
    return (short)(((uint16_t)*(char*)(chainConfigPtr + 28)) | ((uint16_t)*(char*)(chainConfigPtr + 27) << 8));
}
