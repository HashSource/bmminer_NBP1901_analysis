void ProcessMiningStatus(int isStatusCheck) {
  unsigned int now;
  int minersCount;
  int minerIndex;
  unsigned int minerStatus;
  int nextMinerIndex;
  char logBuffer[2052]; // Buffer to hold logs
  
  if (isStatusCheck == 0) { // If this is not a status check call, perform error logging
    // Check if mining has stopped or an error has occurred
    if ((MiningStoppedFlag != '\0') || (ErrorOccurredFlag != '\0') || (CriticalErrorCount > 2)) {
      // Log an error message indicating a bad JSON parameter was detected
      snprintf(logBuffer, 0x800, "%s: input bad json param\n", MiningOperationName);
      LogError(3, logBuffer, 0);
      return;
    }
  } else { // If this is a status check call, perform status update
    now = GetCurrentTime();
    minersCount = GetConnectedMinersCount();

    if (minersCount > 0) {
      minerIndex = 0;
      do {
        nextMinerIndex = minerIndex + 1;
        minerIndex = GetNextMinerIndex(minerIndex);
        minerStatus = GetMinerStatus(minerIndex, minerIndex >> 0x1f);
        UpdateMinerStatus(now, minerStatus);
        minerIndex = nextMinerIndex;
      } while (minersCount != nextMinerIndex);
    }
    
    SendStatusUpdate(isStatusCheck, &StatusUpdateEndpoint, now); // Send status update to the server
  }
  return;
}
