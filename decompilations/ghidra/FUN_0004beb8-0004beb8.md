// GetMinerErrorCode: Returns the error code from a miner status structure
// param_1: Pointer to a miner status structure
uint32_t GetMinerErrorCode(int minerStatusPtr)
{
  // The error code is assumed to be at offset 12 (0xc) within the miner status structure
  return *(uint32_t *)(minerStatusPtr + 0xc);
}
