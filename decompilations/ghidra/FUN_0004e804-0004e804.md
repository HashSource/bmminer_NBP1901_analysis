#include <stddef.h> // Include standard definitions for size_t
#include <string.h> // Include string library for strlen

#define ERROR_CODE 0xFFFFFFFF // Define an error code constant

// Function prototypes
uint32_t SendDataWithStringLength(uint32_t socketFd, const char *data, size_t dataLength);

// Send a string message through a given socket
// Params:
// - uint32_t socketFd: The file descriptor of the socket through which to send the message
// - char *messageToSend: The string message to send
// Returns:
// - uint32_t: The result of the send operation; ERROR_CODE if the message is NULL
uint32_t SendStringMessage(uint32_t socketFd, char *messageToSend)
{
    size_t messageLength;
    uint32_t sendResult;
    
    // Check if the input message is not NULL
    if (messageToSend != NULL)
    {
        // Get the length of the input message
        messageLength = strlen(messageToSend);
        // Send the message and its length to the socket
        sendResult = SendDataWithStringLength(socketFd, messageToSend, messageLength);
        return sendResult;
    }
    // Return an error code if messageToSend is NULL
    return ERROR_CODE;
}
