void HeartbeatMonitoringThread(void)
{
  double initialVoltage;
  int heartBeatFailureCount;
  uint chainIndex;
  double averageVoltage;
  double voltageChange;
  double currentVoltage;
  double targetVoltage;
  int failureCounterArray[4];
  char logBuffer[2048];
  
  initialVoltage = VoltageThreshold;
  failureCounterArray[0] = 0;
  failureCounterArray[1] = 0;
  failureCounterArray[2] = 0;
  failureCounterArray[3] = 0;
  prctl(PR_SET_NAME,"heart_beat",0,0,0); // set the name of the thread
  targetVoltage = -1.0;
  do {
    averageVoltage = VoltageBase;
    chainIndex = 0;
    pthread_setcancelstate(PTHREAD_CANCEL_ENABLE,NULL);
    do {
      heartBeatFailureCount = CheckChainHeartbeat(chainIndex);
      if (heartBeatFailureCount != 0) {
        DeliberateDelay(10);
        heartBeatFailureCount = ResetChain(chainIndex);
        if (heartBeatFailureCount == 0) {
          heartBeatFailureCount = failureCounterArray[chainIndex];
          failureCounterArray[chainIndex] = heartBeatFailureCount + 1;
          if (1 < VerboseLevel) { // VerboseLevel is a global debug level
            snprintf(logBuffer, 0x800, "chain[%d] heart beat fail %d times.\n", chainIndex, heartBeatFailureCount + 1);
            Log(1, logBuffer);
          }
        }
        else {
          failureCounterArray[chainIndex] = 0;
        }
        DeliberateDelay(10);
        currentVoltage = (double)GetChainVoltage(chainIndex);
        averageVoltage += currentVoltage;
      }
      chainIndex++;
    } while (chainIndex != 4);
    heartBeatFailureCount = GetWorkingChainCount();
    if (heartBeatFailureCount != 0) {
      heartBeatFailureCount = GetWorkingChainCount();
      averageVoltage /= (double)(longlong)heartBeatFailureCount;
    }
    if (targetVoltage < initialVoltage) {
      targetVoltage = averageVoltage;
    }
    if (4 < VerboseLevel) {
      snprintf(logBuffer, 0x800, "last avg = %.2f, current voltage = %.2f", VerboseLevel, targetVoltage, averageVoltage);
      Log(4, logBuffer);
    }
    voltageChange = (double)(longlong)(VoltageAdjustmentFactor / 100) * 0.75;
    if ((voltageChange < targetVoltage &&
        ((voltageChange = targetVoltage * 0.75, voltageChange != averageVoltage) && (voltageChange < averageVoltage) ||
         isnan(voltageChange) || isnan(averageVoltage))) ||
        (voltageChange < averageVoltage &&
        ((voltageChange = averageVoltage * 0.75, voltageChange != targetVoltage) && (voltageChange < targetVoltage) ||
         isnan(voltageChange) || isnan(targetVoltage)))) &&
        (3 < VerboseLevel)) {
      voltageChange = averageVoltage - targetVoltage;
      const char* trend = (voltageChange != 0.0 && voltageChange < 0.0) ? "decreased" : "increased";
      snprintf(logBuffer, 0x800, "chain avg voltage %s to %.2f", trend, averageVoltage);
      Log(3, logBuffer);
    }
    pthread_setcancelstate(PTHREAD_CANCEL_DISABLE,NULL);
    pthread_testcancel();
    sleep(10); // Wait for 10 seconds before repeating the cycle
    targetVoltage = averageVoltage; // Update targetVoltage for the next cycle
  } while(true);
}
