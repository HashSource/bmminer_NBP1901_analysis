// ResizeArrayList - Resizes an array list to at least the specified capacity

// requested_capacity: Minimum required capacity to resize the array list
// Returns: Status code (0 for success, 0xffffffff for failure)
uint32_t ResizeArrayList(void ***array_list, int *current_capacity, int requested_capacity)
{
  void *new_memory_block;
  uint32_t status_code;
  int *error_number;
  int current_size;
  int required_size;
  int size_difference;
  
  // Calculate the new required size, which should be the larger of the current capacity doubled or the requested capacity
  required_size = *current_capacity * 2;
  if (*current_capacity * 2 < requested_capacity) {
    required_size = requested_capacity;
  }

  // Attempt to reallocate memory for the new size
  new_memory_block = realloc(*array_list, required_size * sizeof(int));
  if (new_memory_block == NULL) {
    // If realloc fails, get the error number and log it
    error_number = __errno_location();
    LogError(2, "src/zc_arraylist.c", 0x43, "realloc fail, errno[%d]", *error_number);
    status_code = 0xffffffff; // Indicate failure
  }
  else {
    // If realloc succeeds, set the new memory block to the array list
    current_size = *current_capacity;
    *array_list = new_memory_block;
    
    // Calculate the size difference for the new and old sizes
    size_difference = required_size - current_size;
    if (size_difference != 0) {
      // If there is a size difference, initialize the new memory area with zeros
      memset((char *)new_memory_block + current_size * sizeof(int), 0, size_difference * sizeof(int));
    }
    
    status_code = 0; // Indicate success
    *current_capacity = required_size; // Update the current capacity
  }
  
  // Return the status code indicating the result of resizing
  return status_code;
}
