/* Global flag indicating if the device (e.g., LCD screen) is open */
extern int gDeviceOpenFlag;

/* Function to safely close the device and handle errors */
int closeDeviceSafely(int fileDescriptor)
{
  int closeStatus;
  char errorMessageBuffer[1024];
  const char *closeFailedMsg = "Close failed!";

  // Check if the device is open
  if (gDeviceOpenFlag != 0) {
    // Attempt to close the device
    closeStatus = close(fileDescriptor);
    
    // If closing the device failed, log an error message
    if (closeStatus != 0) {
      // Prepare the error message in the buffer
      memcpy(errorMessageBuffer, closeFailedMsg, strlen(closeFailedMsg) + 1); // +1 for the null-terminator
      // Log the error (Assuming FUN_0002e584 is a logging function)
      logErrorMessage(0, errorMessageBuffer);
      // Set the status to -1 indicating an error
      closeStatus = -1;
    }

    // Reset the device open flag
    gDeviceOpenFlag = 0;
    
    // Return the status of the close operation
    return closeStatus;
  }
  
  // If the device was not open, return 0 indicating "nothing to do"
  return 0;
}
