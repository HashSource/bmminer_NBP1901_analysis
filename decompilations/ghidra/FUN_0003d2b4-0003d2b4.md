char * generateFormattedString(char *format, char *inputString)
{ 
  size_t formatLength;
  size_t inputStringLength;
  char *resultString;
  
  // Calculate the length of the format string
  formatLength = strlen(format);
  // Calculate the length of the input string
  inputStringLength = strlen(inputString);
  // Allocate memory for the resultant formatted string
  resultString = (char *)malloc(formatLength + inputStringLength);
  // Generate the formatted string using sprintf
  sprintf(resultString, format, inputString);
  
  // Return the newly created formatted string
  return resultString;
}
