int getTotalHashRate(int mode)
{
  int hashRatePrimary;
  int hashRateSecondary;
  
  // If mode is 0, get the primary hash rate only.
  if (mode == 0) {
    hashRatePrimary = getHashRateComponent(0x3ba, 1);
    return hashRatePrimary;
  }
  // If mode is 1, get the secondary hash rate only.
  if (mode == 1) {
    hashRateSecondary = getHashRateComponent(0x3bb);
    return hashRateSecondary;
  }
  // For any other mode, get the total hash rate (sum of primary and secondary).
  hashRatePrimary = getHashRateComponent(0x3ba, 1);
  hashRateSecondary = getHashRateComponent(0x3bb, 1);
  return hashRatePrimary + hashRateSecondary;
}
