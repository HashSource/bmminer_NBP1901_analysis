int CheckI2CDeviceStatus(uint32_t deviceAddress, uint32_t registerAddress)
{
  int attemptResult;
  int retryCount;
  
  retryCount = 8; // retry up to 8 times
  do {
    // Attempt to communicate with the I2C device
    attemptResult = AttemptI2CCommunication(0x1c, deviceAddress, registerAddress);
    
    // If the communication is unsuccessful, return the error code
    if (attemptResult != 0) {
      return attemptResult;
    }
    
    // Wait for 1 millisecond before the next attempt
    usleep(1000);
    
    // Decrement the retry count
    retryCount = retryCount - 1;
  } while (retryCount != 0); // Keep retrying until the count reaches zero
  
  return 0; // If all attempts are unsuccessful, return 0 indicating no error
}
