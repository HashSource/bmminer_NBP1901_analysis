void EnablePrivilegedMode(undefined4 networkSocket)
{
  uint privilegeFlag;
  
  // Get current privileges for the network socket
  GetSocketPrivileges(0, networkSocket, 0, 0x90, &privilegeFlag);
  
  // Modify the privilege flag to enable privileged mode
  privilegeFlag = privilegeFlag | 0x800;
  
  // Set the new privileges back to the network socket
  SetSocketPrivileges(networkSocket, 1, 0, 0x90, privilegeFlag);
  
  return;
}
