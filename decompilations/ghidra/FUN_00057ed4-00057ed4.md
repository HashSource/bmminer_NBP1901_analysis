void PrintMinerConfig(void *minerConfigPtr, undefined4 logLevel)
{
  int minerConfigEntryPtr;
  int minerConfigKeyValPtr;
  
  // Check if the miner configuration pointer is not NULL
  if (minerConfigPtr != (void *)0x0) {
    // Log the miner configuration object address
    LogPrint(logLevel, "src/mdc.c", 0x16, "---mdc[%p]---", minerConfigPtr);
    
    // Iterate over the miner configuration entries using a utility function
    for (minerConfigEntryPtr = GetFirstConfigEntry(*((undefined4 *)minerConfigPtr)); 
         minerConfigEntryPtr != 0; 
         minerConfigEntryPtr = GetNextConfigEntry(*((undefined4 *)minerConfigPtr), minerConfigEntryPtr)) {
      // Get the address of the key-value pair structure within the configuration entry
      minerConfigKeyValPtr = *(int *)(minerConfigEntryPtr + 8);
      
      // Log the key-value pair address along with the key and value strings
      LogPrint(logLevel, "src/mdc.c", 0x1c, "----mdc_kv[%p][%s]-[%s]----",
               minerConfigKeyValPtr, minerConfigKeyValPtr, minerConfigKeyValPtr + 0x401);
    }
    return;
  }
  // If the miner configuration pointer is null, log an error
  LogPrint(2, "src/mdc.c", 0x15, "a_mdc is null or 0");
  return;
}
