void ClearStringBuffer(char *buffer)
{
  size_t bufferSize;
  
  // If the buffer pointer is NULL, there's nothing to clear, so return immediately
  if (buffer == NULL) {
    return;
  }
  
  // Get the length of the string in the buffer
  bufferSize = strlen(buffer);
  
  // Clear the buffer by setting all its bytes to 0
  // Assuming FUN_0004e700 is a memset-like function that clears a given buffer
  ClearBuffer(buffer, bufferSize);
  return;
}
