// Function responsible for handling log request and updating log configuration if needed.
void HandleLogRequest(int logCategory, undefined4 logMessage, undefined4 srcFile, undefined4 srcLine,
                      undefined4 function, undefined4 userMsg, uint logLevel, undefined4 timeInfo)
{
  int *threadLogBuffer;
  int setSpecificRetVal;
  uint calculatedIndex;
  uint bitOffset;
  
  // If the category is specified, check if the log level is enabled for this category.
  if (logCategory != 0) {
    calculatedIndex = logLevel + 7;
    if (logLevel >= 0) { // if logLevel is non-negative
      calculatedIndex = logLevel;
    }
    bitOffset = logLevel & 7; // bitwise AND logLevel with 7 to get the bit offset
    if ((int)logLevel < 0) { // if logLevel is negative
      bitOffset = -(-logLevel & 7);
    }
    // Check if the specified log level is allowed by the category's configuration.
    if ((((unsigned int)*(unsigned char *)(logCategory + (calculatedIndex >> 3) + 0x408) >> 
        (7 - bitOffset & 0xFF)) & 1U) == 0)
    {
      return; // Log level not enabled, exit the function.
    }
  }
  
  // Obtain read lock on global configuration singleton
  pthread_rwlock_rdlock((pthread_rwlock_t *)&GlobalConfigLock);
  
  if (GlobalConfigInitFlag == 0) {
    LogError(2, "Never call zlog_init() or dzlog_init() before");
    goto UnlockAndReturn;
  }
  
  // Retrieve the thread-specific data
  threadLogBuffer = (int *)pthread_getspecific(ThreadLogBufferKey);
  if (threadLogBuffer == NULL) {
    // Allocate a new log buffer for the thread if not already present
    threadLogBuffer = (int *)AllocateThreadLogBuffer(GlobalCategoryBufferSize,
                                                     *(undefined4 *)(GlobalConfig + 0x41c),
                                                     *(undefined4 *)(GlobalConfig + 0x420),
                                                     *(undefined4 *)(GlobalConfig + 0x244c));
    if (threadLogBuffer == NULL) {
      LogError(2, "Thread log buffer allocation failed");
      goto UnlockAndReturn;
    }
    // Set thread-specific data
    setSpecificRetVal = pthread_setspecific(ThreadLogBufferKey, threadLogBuffer);
    if (setSpecificRetVal != 0) {
      FreeThreadLogBuffer(threadLogBuffer);
      LogError(2, "pthread_setspecific failed, error code [%d]", setSpecificRetVal);
      goto UnlockAndReturn;
    }
  }
  
  // Resize buffer if necessary.
  if (*threadLogBuffer != GlobalCategoryBufferSize) {
    setSpecificRetVal = ResizeThreadLogBuffer(threadLogBuffer,
                                              *(undefined4 *)(GlobalConfig + 0x41c),
                                              *(undefined4 *)(GlobalConfig + 0x420));
    if ((setSpecificRetVal != 0) ||
       (setSpecificRetVal = ResizeThreadMsgBuffer(threadLogBuffer,
                                                  *(undefined4 *)(GlobalConfig + 0x244c)),
       setSpecificRetVal != 0)) {
      LogError(2, "Thread log buffer resize failed, error code [%d]", setSpecificRetVal);
      goto UnlockAndReturn;
    }
    *threadLogBuffer = GlobalCategoryBufferSize;
  }
  
  // Format and write the log data.
  FormatAndWriteLog(threadLogBuffer[2], logCategory, *(undefined4 *)(logCategory + 0x404), logMessage,
                    srcFile, srcLine, function, userMsg, logLevel, timeInfo, &stack0x00000010);
  
  // Try to write log to output.
  int outputRetVal = WriteLogToOutput(logCategory, threadLogBuffer);
  if (outputRetVal == 0) {
    // If reload period is reached, attempt to reload config.
    if ((*(uint *)(GlobalConfig + 0x243c) != 0) &&
        (ReloadConfigCounter = ReloadConfigCounter + 1, 
        *(uint *)(GlobalConfig + 0x243c) < ReloadConfigCounter)) {
      pthread_rwlock_unlock(&GlobalConfigLock);
      int reloadRetVal = ReloadLogConfig(0xffffffff);
      if (reloadRetVal == 0) {
        return; // Configuration reload successful.
      }
      // Configuration reload failed, log error.
      LogError(2, "Reached reload period but failed to reload configuration, check config file details");
      return;
    }
  }
  else {
    LogError(2, "Log output failed, source file [%s], source line [%ld]", srcFile, userMsg);
  }

UnlockAndReturn:
  // Release the read lock.
  pthread_rwlock_unlock(&GlobalConfigLock);
  return;
}
