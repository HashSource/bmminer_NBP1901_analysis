/* WARNING: Globals starting with '_' overlap smaller symbols at the same address */

int InitializeMinerHardware(void) {
    int initializationStatus;
    int powerTypeStatus;
    uint32_t fanCheckMessage[8];
    char *hardwareIdentifier;
    uint32_t startupTempValue;

    InitializeSystem();
    InitializeASIC();
    initializationStatus = ConfigureMinerSettings(4000, 1);
    if (initializationStatus != 0) {
        return initializationStatus;
    }
    if (global_MinerVersion < 4) {
        initializationStatus = SetupNetwork();
    } else {
        // Populate fan check message with "Fan check passed" string
        memcpy(fanCheckMessage, "Fan check passed", sizeof(fanCheckMessage));
        SendMessageToLCD(3, fanCheckMessage, 0);
        initializationStatus = SetupNetwork();
    }
    if (initializationStatus != 0) {
        powerTypeStatus = CheckPowerType();
        if (powerTypeStatus == 0) {
            return initializationStatus;
        }
        if (global_PowerTypeChecked == 0) {
            return initializationStatus;
        }
    }
    CheckTemperatureSensors();
    ConfigureASICFrequency();
    if (global_miningFlag == '\0') {
        if (global_totalASICS < 1) goto EndInitialization;
        initializationStatus = GetASICVersion();
    } else {
        hardwareIdentifier = GetHardwareIdentifier();
        initializationStatus = strcmp(hardwareIdentifier,"BHB28601");
        if (initializationStatus == 0) {
            startupTempValue = 0x76c;
        } else {
            startupTempValue = 0x5dc;
        }
        initializationStatus = GetASICVersion(startupTempValue);
    }
    if (initializationStatus != 0) {
        SetLEDStatus(0xc, 0);
        LogMessage(9, "get power type version failed!");
        return initializationStatus;
    }
EndInitialization:
    if (global_MinerVersion > 3) {
        startupTempValue = GetStartupTemperatureValue();
        snprintf((char *)&fanCheckMessage, 0x800, "start up min temp by 75a = %d", startupTempValue);
        SendMessageToLCD(3, fanCheckMessage, 0);
    }
    ResetHashboard();
    InitializeVoltageRegulators();
    InitializeASICChips();
    DelayExecution(10);
    SetFansSpeed(0x1c200);
    DelayExecution(10);
    initializationStatus = TestASICChips();
    if ((initializationStatus == 0) || ((powerTypeStatus = CheckPowerType(), powerTypeStatus != 0 && (global_PowerTypeChecked != 0)))) {
        FinalizeInitialization();
        initializationStatus = 0;
        global_currentTemp = 0;
        global_maxTemp = 0;
        global_minTemp = 0;
    }
    return initializationStatus;
}
