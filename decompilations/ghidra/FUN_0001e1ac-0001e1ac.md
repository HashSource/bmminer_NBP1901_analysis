// RetrieveTemperatureValue - Retrieves the temperature value for a specific sensor index.
// param_1: Sensor index for which the temperature should be retrieved.
// Returns the temperature value as a 32-bit integer.

uint32_t RetrieveTemperatureValue(int sensorIndex)
{
  // The base temperature sensor data memory address is assumed to be stored in a global variable.
  // The temperature for each sensor is stored at a fixed offset from the base address.
  // The offset for each sensor seems to be sensorIndex * 12 bytes (0xC).
  // Within that offset, the temperature value itself seems to be 4 bytes into it.
  return *(uint32_t *)(gTemperatureSensorBaseAddress + sensorIndex * 0xC + 4);
}
