```c
#include <pthread.h>

// Global variables for data synchronization
pthread_mutex_t dataMutex = PTHREAD_MUTEX_INITIALIZER;
int dataReadyFlag = 0;
int processDataFlag = 0;
int shutdownFlag = 0;

// Resets synchronization flags to indicate that new data is not ready and should not be processed
void resetSynchronizationFlags(void)
{
  // Lock the mutex to ensure thread-safe access to flags
  pthread_mutex_lock(&dataMutex);

  // Reset flags
  dataReadyFlag = 0;
  processDataFlag = 0;
  shutdownFlag = 0;

  // Unlock the mutex
  pthread_mutex_unlock(&dataMutex);
}
```
