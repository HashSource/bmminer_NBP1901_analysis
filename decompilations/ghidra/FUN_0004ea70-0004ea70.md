uint64_t GetConfigValueFromStruct(int *configStruct)
{
  // Validate that the provided pointer is not NULL and points to the expected value.
  if ((configStruct != NULL) && (*configStruct == EXPECTED_VALUE)) {
    // Return the 64-bit value located two integers into the structure.
    return *(uint64_t *)(configStruct + 2);

  return ERROR_VALUE;
}
