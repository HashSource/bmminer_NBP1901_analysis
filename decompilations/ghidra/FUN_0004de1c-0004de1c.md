#include <stdint.h>

// Function: CheckErrorStatus
// Description: Analyses a status code from Bitmain Antminer and returns an error category.
// Parameters:
// - statusCode: The status code to analyze.
// Returns:
// - An error category code based on the status code provided.

uint32_t CheckErrorStatus(uint32_t statusCode) {
  // Check if the error indicator bit is not set (bit 7)
  if ((statusCode & 0x80) == 0) {
    return 1; // No error
  }

  // XOR status code with 0x80 and check if the result is less than 0x42
  if ((statusCode ^ 0x80) < 0x42) {
    return 0; // Error category 0
  }

  // Add 0x3e to status code, apply 0xff mask, and check if the result is less than 0x1e
  if ((statusCode + 0x3e & 0xff) < 0x1e) {
    return 2; // Error category 2
  }

  // Add 0x20 to status code, apply 0xff mask, and check if the result is less than 0x10
  if ((statusCode + 0x20 & 0xff) < 0x10) {
    return 3; // Error category 3
  }

  // Add 0x10 to status code, apply 0xff mask, and check if the result is less than 5
  if ((statusCode + 0x10 & 0xff) < 5) {
    return 4; // Error category 4
  }

  // If none of the above conditions are met, return no specific error category
  return 0; // No specific category
}
