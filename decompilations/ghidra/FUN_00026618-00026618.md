void CheckSystemBalanceAndRebootIfNecessary(void)
{
  int currentBalanceValue;
  double scaledBalance;
  
  currentBalanceValue = GetCurrentBalanceReading();  // Get the current balance value
  scaledBalance = (double)(longlong)currentBalanceValue * balanceScalingFactor;  // Scale the balance reading
  if (scaledBalance == lastKnownBalance || scaledBalance < lastKnownBalance != (IsNaN(scaledBalance) || IsNaN(lastKnownBalance))) {
    // If the scaled balance hasn't changed or isn't NaN, exit without doing anything
    return;
  }
  currentBalanceValue = CheckIfRebootAllowed();  // Check if the conditions for reboot are met
  if (currentBalanceValue == 0) {
    // If not allowed to reboot, exit without doing anything
    return;
  }
  // Log the unbalance event and initiate reboot
  LogEventAndReboot(0xb, "Unbalance happened, reboot!\n");
  return;
}
