/* Function to unmap and close FPGA memory */

void UnmapAndCloseFPGAMemory(void)
{
  int operationResult;
  char errorMessageUnmapFailed[] = "munmap failed";
  char errorMessageUnmapFPGAFailed[] = "munmap FPGA mem addr hal failed";
  
  // Check if FPGA memory is already mapped
  if (_isFPGAMemMapped == 0) {
    return; // Return if not mapped
  }
  
  // Unmap the FPGA control memory region
  operationResult = munmap(_fpgaControlMemAddress, 0x1200);
  if (operationResult < 0) {
    LogError(errorMessageUnmapFailed);
  }
  
  // Unmap the FPGA hash memory region
  operationResult = munmap(_fpgaHashMemAddress, 0x1000000);
  if (operationResult < 0) {
    LogError(errorMessageUnmapFPGAFailed);
  }

  // Mark the FPGA memory as unmapped
  _isFPGAMemMapped = 0;

  // Close the file descriptors for the FPGA memory regions
  close(_fpgaControlMemFileDescriptor);
  close(_fpgaHashMemFileDescriptor);
  
  return;
}
