void DumpMemory(int startAddress, uint numberOfBytes)
{
  byte *currentBytePointer;
  uint bytesPrinted;
  uint printableChar;
  char *dataTypeIndicator;
  int baseAddress;
  char outputBuffer[0x800]; // Buffer to hold string outputs
  
  dataTypeIndicator = "read return data";
  baseAddress = startAddress;
  snprintf(outputBuffer, sizeof(outputBuffer), "Dumping %u %s bytes from %p:\n", numberOfBytes, dataTypeIndicator, startAddress);
  LogMessage(2, outputBuffer, 0); // Log the message
  
  if ((int)numberOfBytes < 1) {
    return; // No bytes to dump, return immediately
  }
  
  do {
    bytesPrinted = 0;
    do {
      while (bytesPrinted < numberOfBytes) {
        currentBytePointer = (byte *)(startAddress + bytesPrinted);
        bytesPrinted += 1;
        snprintf(outputBuffer, sizeof(outputBuffer), "%02x ", (uint)*currentBytePointer, dataTypeIndicator, baseAddress);
        LogMessage(2, outputBuffer, 0); // Log the hex byte
        if (bytesPrinted == 0x10) goto PrintNewline; // Check if 16 bytes were printed
      }
      bytesPrinted += 1; // Increment for next iteration
      snprintf(outputBuffer, sizeof(outputBuffer), "   "); // Append spaces if less than 16 bytes
      LogMessage(2, outputBuffer, 0);
    } while (bytesPrinted != 0x10);
    
PrintNewline:
    bytesPrinted = 0;
    do {
      while (bytesPrinted < numberOfBytes) {
        printableChar = (uint)*(byte *)(startAddress + bytesPrinted);
        bytesPrinted += 1;
        if (printableChar < 0x20 || printableChar > 0x7e) {
          printableChar = 0x2e; // Replace non-printable characters with a dot
        }
        snprintf(outputBuffer, sizeof(outputBuffer), "%c", printableChar);
        LogMessage(2, outputBuffer, 0); // Log the ASCII character
        if (bytesPrinted == 0x10) goto EndOfLine; // Check if 16 characters were printed
      }
      bytesPrinted += 1;
      snprintf(outputBuffer, sizeof(outputBuffer), " "); // Append a space if less than 16 bytes
      LogMessage(2, outputBuffer, 0);
    } while (bytesPrinted != 0x10);
    
EndOfLine:
    snprintf(outputBuffer, sizeof(outputBuffer), "\n"); // Print a newline
    startAddress += 0x10; // Move to the next line
    LogMessage(2, outputBuffer, 0);
    
    if ((int)numberOfBytes < 0x11) {
      return; // If less than 17 bytes left to print, return
    }
    
    numberOfBytes -= 0x10; // Decrement bytes by 16 for the next group
  } while(true);
}
