int CheckAndInitializePIC(uint chainIndex)
{
  int status;
  char *errorMsg;
  int result;
  undefined4 i2cRegisterValue;
  undefined4 commandToSend;
  undefined2 commandStatus;
  undefined4 firstCommandPart;
  undefined4 secondCommandPart;
  undefined4 thirdCommandPart;
  char logBuffer[2052];
  
  // Verify that the chain index is within valid range (0 to 15)
  if (chainIndex > 15) {
    snprintf(logBuffer, 0x800, "%s: Bad PIC parameter, input chain is %d\n", PIC_ERROR_PREFIX, chainIndex);
    LogError(0, logBuffer);
    return -EINVAL;
  }
  
  // Check if PIC is already initialized, if not, attempt to open EEPROM
  if ((*(int *)(EEPROM_STATUS_OFFSET + chainIndex * 8) == 0) && (status = OpenEEPROM(), status < 0)) {
    snprintf(logBuffer, 0x800, "%s: auto exec open_pic, but chain %d open EEPROM failed\n",
             PIC_ERROR_PREFIX, chainIndex);
    LogError(0, logBuffer, 0);
    return status;
  }
  
  // Prepare the I2C register value
  i2cRegisterValue = *(undefined4 *)(I2C_REGISTER_OFFSET + chainIndex * 8);
  result = 0;
  
  // Acquire the I2C mutex before communicating
  pthread_mutex_lock((pthread_mutex_t *)&I2C_MUTEX);
  
  // Setup the I2C command to initialize the PIC
  firstCommandPart = PIC_INIT_CMD_HEADER;
  secondCommandPart = 0;
  thirdCommandPart = 0;
  commandToSend = 0;
  commandStatus = 0;
  firstCommandPart = PIC_INIT_CMD_CODE;
  
  // Send the PIC initialization command
  status = SendI2CCommand(i2cRegisterValue, &firstCommandPart);
  if (status == PIC_CMD_LENGTH) {
    usleep(300000); // Wait for PIC to process the command
  
    // Read the response from the PIC
    status = ReadI2CResponse(i2cRegisterValue, &commandToSend, PIC_CMD_LENGTH);
    if (status != PIC_CMD_LENGTH) {
      errorMsg = "%s read I2C error\n";
      goto LogErrorAndExit;
    }

    // Check if the response is as expected
    if ((commandToSend._1_1_ == PIC_RESPONSE_VERIFY_BYTE1) && (commandToSend._2_1_ == PIC_RESPONSE_VERIFY_BYTE2)) goto OperationSuccess;
    
    // If the response is not as expected, log the error
    snprintf(logBuffer, 0x800, "%s failed!\n", PIC_INIT_FAIL_PREFIX);
    LogError(0, logBuffer);
    PrintHex(&commandToSend, PIC_CMD_LENGTH);
  }
  else {
    // If the command was not successful, log the write error
    errorMsg = "%s write I2C error\n";
LogErrorAndExit:
    snprintf(logBuffer, 0x800, errorMsg);
    LogError(0, logBuffer, 0);
  }
  
  // Set error result code
  result = -ENXIO;
OperationSuccess:
  // Release the I2C mutex after communication
  pthread_mutex_unlock((pthread_mutex_t *)&I2C_MUTEX);
  
  return result;
}
