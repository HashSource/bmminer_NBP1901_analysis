```c
#include <stdint.h>

// Function prototype declarations
int32_t GetOptimizedPerformance();
int32_t GetBasicPerformance();
int32_t ReadASICFrequency();
int32_t ReadASICVoltage();

// Define constants for operation modes
#define MODE_PERFORMANCE_OPT 2
#define MODE_PERFORMANCE_BASIC 1
#define MODE_READ_FREQUENCY 0x100
#define MODE_READ_VOLTAGE 0x101

/**
 * GetSystemParameter - Retrieves various system parameters based on the input mode
 * @mode: The operation mode, which determines what parameter to fetch
 * 
 * This function retrieves the system parameter based on the provided mode. It calls different
 * functions that handle specific operations dependent on the parameter being queried.
 * 
 * Returns the result of the operation or 0 if the mode is invalid or the operation fails.
 */
uint32_t GetSystemParameter(uint32_t mode) {
    int32_t result;

    switch (mode) {
        case MODE_PERFORMANCE_OPT:
            result = GetOptimizedPerformance();
            // If fetching the optimized performance is successful, return the result
            if (result >= 0) {
                return GetOptimizedPerformance();
            }
            break;

        case MODE_PERFORMANCE_BASIC:
            return GetBasicPerformance();

        case MODE_READ_FREQUENCY:
            result = ReadASICFrequency();
            // If reading the ASIC frequency is successful, return the result
            if (result >= 0) {
                return ReadASICFrequency();
            }
            break;

        case MODE_READ_VOLTAGE:
            result = ReadASICVoltage();
            // If reading the ASIC voltage is successful, return the result
            if (result >= 0) {
                return ReadASICVoltage();
            }
            break;

        default:
            // If mode is not valid, return 0
            return 0;
    }

    return 0; // Return 0 if any of the operations fail
}
```
