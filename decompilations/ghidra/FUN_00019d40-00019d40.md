int CompareTemperatures(int sensor1_ptr, int sensor2_ptr)
{
  // This function takes pointers to two sensor data structures and
  // subtracts the temperature value (an integer) located at an offset of 4 bytes 
  // in the second sensor's data structure from the temperature value in the first sensor's data structure.
  // It returns the result of the subtraction, which could be used for sorting or comparing temperatures.

  // Casting the pointer to an integer pointer and moving 4 bytes offset to get the temperature value.
  int temperature1 = *(int *)(sensor1_ptr + 4);  
  int temperature2 = *(int *)(sensor2_ptr + 4);

  // Returns the difference between the two temperatures.
  return temperature1 - temperature2;
}
