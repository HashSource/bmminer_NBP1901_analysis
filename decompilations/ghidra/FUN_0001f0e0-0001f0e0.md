void SendCommandToMinerASICs(uint targetASIC, int commandType, undefined4 param1, undefined4 param2)
{
  int isASICActive;
  uint currentASIC;
  
  currentASIC = 0;
  do {
    isASICActive = CheckASICStatus(currentASIC); // Check if the ASIC is active/responding
    if ((isASICActive != 0) && (targetASIC == 0xff || currentASIC == targetASIC)) {
      if (commandType == 0) {
        SendASICConfiguration(currentASIC & 0xff, param1, param2, 0); // Configure ASIC with given parameters
      }
      else {
        SendASICCommand(currentASIC & 0xff, param2, 0); // Send a control command to ASIC
      }
    }
    currentASIC = currentASIC + 1;
  } while (currentASIC != 4); // Assuming there are 4 ASICs in the system
  return;
}
