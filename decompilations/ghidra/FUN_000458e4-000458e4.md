void SendMiningConfiguration(uint hashBoardIndex)
{
  // Send the configuration to the hash board 
  // 0x80808000 is a control flag added to the hash board index being configured.
  ConfigureHashBoard(0x12, hashBoardIndex | 0x80808000);
  
  // Update the last known good configuration with the current one.
  LastKnownGoodConfig = CurrentConfig;
  
  // Apply the mining settings to the hash board.
  ApplyMiningSettings(HashBoardSettings);
  
  return;
}
