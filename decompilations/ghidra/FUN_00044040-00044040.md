void updateMemoryBlock(void **memoryBlockPointer, void *newData, size_t newSize)
{
  void *allocatedMemory;
  
  // Free the memory previously pointed to by the memory block pointer
  free(*memoryBlockPointer);

  // Allocate new memory of the required size
  allocatedMemory = malloc(newSize);

  // Update the memory block pointer to point to the newly allocated memory
  *memoryBlockPointer = allocatedMemory;

  // Copy the new data into the newly allocated memory
  memcpy(allocatedMemory, newData, newSize);

  // End of the function
  return;
}
