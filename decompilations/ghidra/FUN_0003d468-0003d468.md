#include <errno.h>
#include <stdlib.h>
#include <stdio.h>

// Error printing function to display error messages
void printErrorMessage(const char *errorMessage, const char *inputString) {
  fprintf(stderr, errorMessage, inputString);
  fprintf(stderr, "\n");
}

// Function to convert a string to a long integer
void convertStringToLong(char *inputString, long *outputNumber) {
  char *endPtr;
  int *errnoPtr;
  
  // Store the current value of errno
  errnoPtr = __errno_location();
  *errnoPtr = 0;
  
  // Attempt to convert the input string to a long integer
  long convertedNumber = strtol(inputString, &endPtr, 0);
  *outputNumber = convertedNumber;
  
  // Check if the entire string was a valid number and not out of range
  if ((*endPtr == '\0') && (*inputString != '\0')) {
    if (*errnoPtr != 0) {
      printErrorMessage("\'%s\' is out of range", inputString);
      return;
    }
  } else {
    printErrorMessage("\'%s\' is not a number", inputString);
  }
}
