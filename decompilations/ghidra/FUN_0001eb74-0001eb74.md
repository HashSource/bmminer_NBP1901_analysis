void checkAndAdjustUartRelay(void) {
  int minerCount;
  int totalPorts;
  int baudRate;
  int uartChannel;
  int uartRelay;
  uint uartIterator;
  uint uartIndex;
  char logMessage[1024];     // Buffer to store log message, using local_820 and neighboring variables
  uint minerRelayThreshold = 10; // Threshold to decide whether to adjust UART relay
  
  minerCount = getMinerCount();
  
  // If the miner count is less than the relay threshold, and the debug level is high enough, logs a message
  if (minerCount < minerRelayThreshold) {
    if (4 < DEBUG_LEVEL) {
      // Copy a predefined log message from a global variable into the local logMessage array
      memcpy(logMessage, "no_need_to_set_uart_relay", 24);
      // Send the log message to a log function (function at address 0x0002e584 in this case)
      sendDebugLog(4, logMessage, 0);
      return;
    }
  }
  else {
    // If miner count reaches or exceeds threshold, proceed to adjust the UART relays
    minerCount = getMinerCount();
    minerCount = minerCount - 3;
    if (minerCount >= 0) {
      do {
        totalPorts = getTotalPorts();
        uartIterator = 0;
        baudRate = getBaudRate();
        do {
          uartIndex = uartIterator & 0xff; // Ensuring to iterate over UART index within range
          uartChannel = checkUartChannel();
          uartIterator = uartIterator + 1;
          if (uartChannel != 0) {
            uartChannel = getMinerCount();
            uartRelay = getTotalPorts();
            setUartRelayMode(uartIndex, baudRate * (minerCount * totalPorts + totalPorts - 1),
                             uartRelay * (uartChannel - minerCount) + 0xe, 1, 1);
          }
        } while (uartIterator != 4); // Check up to 4 UART channels
        minerCount = minerCount - 3; // Decrement the miner count by an offset of 3
      } while (minerCount >= 0); // Repeat if the adjusted miner count is non-negative
      return;
    }
  }
  return; // If none of the conditions above are met, simply returns
}
