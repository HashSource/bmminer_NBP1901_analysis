uint ReadSingleByteFromDevice(int *deviceFileDescriptor)
{
  ssize_t bytesRead;
  uint readValue;
  byte singleByteBuffer[9]; // Buffer to store the byte read from the device.
  
  // Read a single byte from the device file descriptor.
  bytesRead = read(*deviceFileDescriptor, singleByteBuffer, 1);
  if (bytesRead == 1) {
    // If read was successful, return the byte as an unsigned integer.
    readValue = (uint)singleByteBuffer[0];
  }
  else {
    // If read failed, return an error code.
    readValue = 0xffffffff;
  }
  return readValue;
}
