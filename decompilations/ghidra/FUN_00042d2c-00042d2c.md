/* Write data to LCD screen buffer and flush it to the hardware */
int LcdBufferWriteAndFlush(int lcdAddress, uint bufferIndex, int dataPointer, uint dataSize) {
  int mutexLockResult;
  int bytesCopied;
  uint chunkSize;
  size_t copySize;
  bool isOutOfBound;
  bool isEqualBoundary;

  if (_isLcdInitialized == 0) {
    return -2; // LCD is not initialized
  } else if (_lcdControllerFileDescriptor == lcdAddress) {
    isOutOfBound = dataSize > 0x40; // Check if dataSize is out of allowed bounds
    isEqualBoundary = dataSize == 0x40;
    if (!isOutOfBound || isEqualBoundary) {
      mutexLockResult = pthread_mutex_lock(&_lcdControllerMutex);
      if (mutexLockResult == 0) { // Successfully locked the mutex
        bytesCopied = 0;
        do {
          bufferIndex++; // Move to the next buffer segment
          chunkSize = dataSize;
          if (dataSize > 0x10) {
            chunkSize = 0x10; // Limit copy size to 16 bytes for each segment
          }
          // Calculate the source pointer based on the amount of data already copied
          void* srcPointer = (void*)(dataPointer + bytesCopied);
          bytesCopied += chunkSize;
          // Copy the data chunk to the write buffer
          memcpy(&_lcdWriteBuffer + bufferIndex * 0x10, srcPointer, chunkSize);
          dataSize -= chunkSize; // Decrease remaining data size
          bufferIndex = bufferIndex & 0xff; // Keep index in bounds
        } while (bufferIndex < 4 && dataSize != 0);
        // Flush the write buffer to the LCD via the file descriptor
        write(_lcdControllerFileDescriptor, _lcdWriteBuffer, 0x40);
        pthread_mutex_unlock(&_lcdControllerMutex);
        return 0; // Success
      }
      // Mutex lock failed, log error
      LogError("failed to lock LCD mutex");
      return -4;
    }
  }

  LogError("bad parameter");
  return -3;
}
