int GetErrorLevelMapping(int *errorMappingTable, int errorCode)
{
  int mappedLevel;
  
  // Check if the errorCode is within the valid range and that it has a corresponding error level
  if ((errorCode < errorMappingTable[1]) && 
      (mappedLevel = *(int *)(*errorMappingTable + errorCode * 4), mappedLevel != 0)) {
    return mappedLevel;
  }
  
  // Log an error if the errorCode is out of bounds or doesn't map to a known error level
  LogError(2,"src/level_list.c",0x77,
           "l[%d] not in (0,254), or has no level defined, see configure file define, set to UNKNOWN"
           ,errorCode);
  
  if (errorMappingTable[1] < 0xff) {
    mappedLevel = 0; // Use 0 if there's no default error level at the upper bound
  } else {
    // Use the last error level mapping as a default if errorCode is equal to or exceeds 0xff
    mappedLevel = *(int *)(*errorMappingTable + 0x3f8);
  }
  return mappedLevel;
}
