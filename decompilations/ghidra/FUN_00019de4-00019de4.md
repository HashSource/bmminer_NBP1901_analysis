int CalculateHashRate(int minerIndex)
{
  int hashRateMultiplier;
  int significantHashRate;
  int baseHashRate;
  int calculatedHashRate = 0;
  
  // Check if a specific mode is set, mode 2 indicates special calculation
  if (currentOperatingMode != 2) {
    UpdateHashRateBasedOnTemperature(minerIndex, temperatureAdjustmentFactor, &calculatedHashRate);
    return calculatedHashRate;
  }

  // Access the base hash rate for the specific miner
  baseHashRate = baseHashRateValues[minerIndex];

  // Fetch significant hash rate part based on current conditions
  significantHashRate = GetSignificantHashRateComponent();

  // Fetch hash rate multiplier based on additional conditions
  hashRateMultiplier = GetHashRateMultiplier();

  // Calculate the final hash rate by multiplying components and normalize it by a factor of 1000
  return (hashRateMultiplier * significantHashRate * baseHashRate) / 1000 + calculatedHashRate;
}
