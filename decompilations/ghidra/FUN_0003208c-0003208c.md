float UpdateChipStatus(void)
{
  uint currentStatus;
  int statusCheckResult;
  uint *statusArrayPointer;
  int index;
  uint *statusArrayIterator;
  
  if (chipStatusFlag == 2) { // Check if the chip status update flag is set
    index = 0;
    do {
      statusCheckResult = CheckChipStatus(index); // Check status for each chip
      if ((statusCheckResult != 0) && (statusCheckResult = ResetChip(), currentStatus = globalChipStatus, 0 < statusCheckResult)) {
        // If chip status needs update, initialize the statuses
        statusArrayPointer = &chipStatusArray + index * 0x100; // Offset into status array
        do {
          statusArrayIterator = statusArrayPointer + 1;
          *statusArrayPointer = currentStatus; // Update the status
          statusArrayPointer = statusArrayIterator; 
        } while (statusArrayIterator != &chipStatusArray + statusCheckResult + index * 0x100);
      }
      index += 1;
    } while (index != 4); // Assume 4 chips for this context

    // If the status flag has an invalid value, attempt to correct it
    if (statusCorrectionFlag != 0.0 && statusCorrectionFlag < 0.0 == NAN(statusCorrectionFlag)) {
      AttemptStatusCorrection(globalCorrectionParameter, statusCorrectionFlag, globalChipStatus);
      // Check if the correction was successful
      if (statusCorrectionFlag == 0.0 || statusCorrectionFlag < 0.0 != NAN(statusCorrectionFlag)) goto ContinueExecution;
      goto ReturnCurrentStatus;
    }
  }
  // If status flag is not invalid, proceed
  if (statusCorrectionFlag == 0.0 || statusCorrectionFlag < 0.0 != NAN(statusCorrectionFlag)) {
ContinueExecution:
    globalChipStatus = RetrieveNewStatus(statusAdjustmentParameter, &chipStatusArray, globalCorrectionParameter, 1);
    return (float)(unsigned long long)globalChipStatus; // Cast and return the new chip status
  }
ReturnCurrentStatus:
  return (float)(unsigned long long)globalChipStatus; // Return the current status if no update occurred
}
