void ConcatenateTwoLinkedLists(int *sourceList, int *destinationList, undefined4 errorMsg1, 
                               undefined4 errorMsg2, undefined4 errorMsg3) {
  int listNodePointer;
  char logBuffer[2048]; // Buffer for log messages
  
  // Check that the types of the two lists are the same
  if (*sourceList != *destinationList) {
    snprintf(logBuffer, sizeof(logBuffer),
             "List %s can't append() to a %s list - from %s %s() line %d in %s %s():%d", *sourceList,
             DAT_0002e3d0, *destinationList, errorMsg1, errorMsg2, errorMsg3, "klist.c", 
             DAT_0002e3d0, 0x170);
    LogError(3, logBuffer, 1); // Call logging function with severity level 3
    TerminateProcess(1); // Terminate with exit status 1
  }
   
  // Check that the destination list supports tail operations
  if (*(char *)(sourceList + 0xb) == '\0') {
    snprintf(logBuffer, sizeof(logBuffer),
             "List %s can't append() - do_tail is false - from %s %s() line %d in %s %s():%d", 
             *sourceList, DAT_0002e3d0, errorMsg1, errorMsg2, errorMsg3, "klist.c", DAT_0002e3d0, 
             0x175);
    LogError(3, logBuffer, 1); // Log error with severity level 3
    TerminateProcess(1); // Terminate with exit status 1
  }

  listNodePointer = sourceList[3]; // Get the head pointer of the source list
  if (listNodePointer != 0) {
    if (destinationList[4] == 0) {
      // If the destination list is empty, set its head to the source list's head
      destinationList[3] = listNodePointer;
      listNodePointer = 0;
    } else {
      // If the destination list has elements, link its tail to the source list's head
      *(int *)(destinationList[4] + 8) = listNodePointer;
      listNodePointer = destinationList[4];
    }
    // Set the previous pointer of the new head of the destination list
    *(int *)(sourceList[3] + 4) = listNodePointer;
    destinationList[4] = sourceList[4]; // Set the tail of the destination list to the source's tail
    sourceList[4] = 0; // Reset the source list's tail pointer
    sourceList[3] = 0; // Reset the source list's head pointer
    // Merge the list lengths and element counts
    destinationList[7] = destinationList[7] + sourceList[7];
    sourceList[7] = 0;
    destinationList[8] = destinationList[8] + sourceList[8];
    sourceList[8] = 0;
  }
  return;
}
