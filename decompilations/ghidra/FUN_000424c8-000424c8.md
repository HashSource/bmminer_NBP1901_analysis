void HandleASICEvent(undefined4 minerId,int resetFlag) // Function to handle ASIC (Application Specific Integrated Circuit) events
{
  int eventCode;
  int flag;
  HandlerFunction **eventHandlerPointer;
  HandlerFunction **eventHandlerEnd;
  HandlerFunction **nextEventHandlerPointer;
  
  eventCode = GetASICEventCode(); // Get the event code from ASIC
  if (eventCode != 0x102) { // Check if the event is not equal to a specific event code (0x102)
    eventHandlerEnd = ASICEventHandlersArray + 5; // End pointer of the event handlers array
    eventHandlerPointer = ASICEventHandlersArray; // Pointer to the start of the event handlers array
    do {
      nextEventHandlerPointer = eventHandlerPointer + 1; // Get the next event handler's pointer
      if (*eventHandlerPointer != (HandlerFunction *)0x0) { // If the current handler is not NULL,
        flag = resetFlag; // Use the reset flag as is
        if (resetFlag != 0) { // If the reset flag is anything but 0,
          flag = 1; // Set the flag to 1
        }
        (**eventHandlerPointer)(eventCode, flag); // Call the handler function with the event code and flag
      }
      eventHandlerPointer = nextEventHandlerPointer; // Move to the next handler in the array
    } while (nextEventHandlerPointer != eventHandlerEnd); // Continue until all handlers have been processed
    return;
  }
  PerformASICReset(minerId); // Reset the ASIC if the event code is 0x102
  return;
}
