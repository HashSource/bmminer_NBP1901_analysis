// Initializes a file-based mutex and returns a pointer to the mutex object.
// The file corresponding to `fileName` is used in conjunction with the mutex
// to provide a locking mechanism.
// If the file cannot be opened or the mutex cannot be initialized, NULL is returned.
pthread_mutex_t* InitializeFileMutex(char* fileName) {
  pthread_mutex_t* fileMutex;
  int fileDescriptor;
  int* errnoPtr;
  
  // Check if the file name is provided.
  if (fileName == NULL) {
    LogError(2, "src/rotater.c", 0x56, "fileName is null");
    fileMutex = NULL;
  } else {
    // Allocate memory for the mutex object.
    fileMutex = (pthread_mutex_t*)calloc(1, sizeof(pthread_mutex_t) + sizeof(int) + sizeof(char*));
    if (fileMutex == NULL) {
      errnoPtr = __errno_location();
      LogError(2, "src/rotater.c", 0x5a, "calloc failed, errno[%d]", *errnoPtr);
    } else {
      // Initialize the mutex.
      int initResult = pthread_mutex_init(fileMutex, NULL);
      if (initResult == 0) {
        // Try to open the file, flags: O_CREAT | O_TRUNC | O_RDWR, mode: 0666
        fileDescriptor = open64(fileName, O_CREAT | O_TRUNC | O_RDWR, 0666);
        if (fileDescriptor < 0) {
          errnoPtr = __errno_location();
          LogError(2, "src/rotater.c", 0x6c, "Failed to open file[%s], errno[%d]", fileName, *errnoPtr);
          DestroyMutex(fileMutex);  // Releases the mutex resources.
          fileMutex = NULL;
        } else {
          // Store the file descriptor at an offset after the mutex structure in memory.
          *((int*)((char*)fileMutex + sizeof(pthread_mutex_t))) = fileDescriptor;
          // Store the file name at the same location.
          *((char**)fileMutex + 1) = fileName; // Assuming that the mutex structure is followed by a pointer.
        }
      } else {
        errnoPtr = __errno_location();
        LogError(2, "src/rotater.c", 0x5f, "pthread_mutex_init failed, errno[%d]", *errnoPtr);
        free(fileMutex);  // Free allocated memory if mutex init failed.
        fileMutex = NULL;
      }
    }
  }
  return fileMutex;
}
