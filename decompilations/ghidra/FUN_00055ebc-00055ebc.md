void ProcessCategoryEntries(int categoryTablePtr)
{
  int entryPtr;
  
  // Check if the category table pointer is null or zero
  if (categoryTablePtr == 0) {
    LogError(2, "src/category_table.c", 0x4e, "categories is null or 0");
    return;
  }
  
  // Get the first entry in the category table
  entryPtr = GetFirstCategoryEntry();
  
  // If the entry pointer is null, return immediately
  if (entryPtr == 0) {
    return;
  }
  
  // Iterate over the category entries
  do {
    // Process the current category entry
    ProcessCategory(*(undefined4 *)(entryPtr + 8)); // Assuming the category itself is at offset 8 of the entry structure
    // Get the next category entry
    entryPtr = GetNextCategoryEntry(categoryTablePtr, entryPtr);
  } while (entryPtr != 0); // Continue until there are no more entries
  
  return;
}
