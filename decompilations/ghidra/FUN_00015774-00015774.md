int GetMinimumChipTemperature(void)
{
  short currentTemperature;
  int chipIndex;
  int minTemperature;
  
  chipIndex = 0;
  minTemperature = 0xff; // Initialize the minimum temperature to the highest possible value (255)
  do {
    // Read the temperature of the current chip
    // The first parameter is the chip index, and the second parameter seems to indicate a read operation (1)
    currentTemperature = ReadChipTemperature(chipIndex, 1);
    chipIndex = chipIndex + 1; // Move to the next chip index
    
    // If the current chip temperature is less than the minimum found, update minTemperature
    if (currentTemperature < minTemperature) {
      minTemperature = (int)currentTemperature;
    }
  } while (chipIndex != 4); // There are 4 chips to check
  
  return minTemperature; // Return the minimum temperature found among all chips
}
