void ** allocateAndCopyData(void *sourceData, void *dataSize)
{
  void **dataStructurePointer;
  void *dataCopy;
  
  // Allocate memory for the data structure (two pointers)
  dataStructurePointer = (void **)malloc(2 * sizeof(void *));
  if (dataStructurePointer != NULL) {
    // Store the data size in the second element of the data structure
    dataStructurePointer[1] = dataSize;
    // Allocate memory to copy the actual data
    dataCopy = malloc((size_t)dataSize);
    // Store the address of the copied data in the first element of the data structure
    *dataStructurePointer = dataCopy;
    if (dataCopy == NULL) {
      // Failed to allocate memory for data copy, free the data structure
      free(dataStructurePointer);
      dataStructurePointer = NULL;
    }
    else {
      // Copy the source data into newly allocated memory
      memcpy(dataCopy, sourceData, (size_t)dataSize);
    }
  }
  return dataStructurePointer;
}
