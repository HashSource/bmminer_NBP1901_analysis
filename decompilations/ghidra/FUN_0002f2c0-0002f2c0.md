#include <pthread.h>
#include <stdio.h>
#include <errno.h>

// UnlockMutex logs an error message if the mutex unlock operation fails
// param_1: pointer to the mutex to unlock
// param_2: file name where the operation is attempted (util.c is assumed)
// param_3: line number where the operation is attempted
void UnlockMutex(pthread_mutex_t *mutex, const char *file_name, int line_number) {
  int unlockResult;
  int *errorNumberPtr;
  char errorMessageBuffer[2052];
  
  // Attempt to unlock the mutex
  unlockResult = pthread_mutex_unlock(mutex);
  
  // Check if the mutex unlock operation was successful
  if (unlockResult == 0) {
    // Success, nothing more to do
    return;
  }

  // Mutex unlock failed, get the error number
  errorNumberPtr = __errno_location();

  // Format an error message including file name and line number information
  snprintf(errorMessageBuffer, sizeof(errorMessageBuffer),
           "WTF MUTEX ERROR ON UNLOCK! errno=%d in %s %s():%d",
           *errorNumberPtr, "util.c", file_name, line_number);

  // Log the formatted error message with log level 3 (likely ERROR level)
  LogError(3, errorMessageBuffer, 1);

  // Terminate the program due to the error
  TerminateProgram(1);
  return;
}
