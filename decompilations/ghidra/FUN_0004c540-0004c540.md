uint getErrorStatus(int minerStatusPtr)
{
  uint errorStatus;

  errorStatus = *(uint *)(minerStatusPtr + 0x14); // Read error status from the miner status structure
  if (errorStatus == 0) {
    errorStatus = getDefaultErrorStatus(); // Get the default error status if current is 0
  }
  // Check if errorStatus + 2 exceeds 1, i.e., errorStatus is greater than or equal to 0
  if (1 < errorStatus + 2) {
    updateErrorDisplay(minerStatusPtr + 0x28, errorStatus & 0xff); // Update the display with the lower byte of errorStatus
  }
  return errorStatus; // Return the error status
}
