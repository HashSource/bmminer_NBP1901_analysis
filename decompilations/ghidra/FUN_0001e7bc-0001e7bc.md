int GetMinerInfo(uint minerIndex)
{
  int minerInfo;
  
  // Check if the minerIndex is valid, preventing buffer overflows
  if (((uint)(maxMinersAllowed <= (int)minerIndex) | minerIndex >> 0x1f) != 0) {
    return -0x7ffffbff; // Error code for invalid miner index
  }
  
  // If miners data not initialized, attempt to initialize
  if ((minersDataInitializedFlag == 0) && (minerInfo = InitializeMinersData(), minerInfo < 0)) {
    return minerInfo; // Return the error code from InitializeMinersData
  }
  
  // Return miner information at the specified index
  // Each miner has a data block of 0x210 bytes; 
  // offset by 4 to get to the specific information field
  return *(int *)(minersInfoBaseAddr + minerIndex * 0x210 + 4);
}
