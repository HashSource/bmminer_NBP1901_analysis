int findDeviceInList(int baseAddress, int *headPointer, char *deviceName, int deviceType)
{
  int currentElement;
  int comparisonResult;
  
  currentElement = *headPointer; // First element in the list
  // Check if the list is not empty (head and next are not pointing to the baseAddress)
  if ((currentElement != baseAddress + 0xc) || (currentElement != headPointer[1])) {
    while (true) {
      // Check if current device type matches and device name matches
      if ((*(int *)(currentElement + 0x10) == deviceType) &&
          (comparisonResult = strcmp((char *)(currentElement + 0x18), deviceName), comparisonResult == 0)) {
        return currentElement; // Return the address of the found device element
      }
      // Check if reached the end of the list
      if (headPointer[1] == currentElement) break;
      
      // Move to the next element in the list
      currentElement = *(int *)(currentElement + 4);
    }
  }
  return 0; // Return 0 if no matching device is found
}
