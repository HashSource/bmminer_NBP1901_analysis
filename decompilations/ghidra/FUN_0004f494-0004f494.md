// Function to handle reference decrementing or resource release
unsigned int HandleReferenceOrRelease(int *destReference, int *sourceReference) {
  bool exclusiveAccess;
  int currentCount;
  int *referenceCounter;

  // Check if sourceReference is not null
  if (sourceReference != NULL) {
    // If destReference is null, or not initialized, or the same as sourceReference
    if (destReference == NULL || (*destReference != 1) || (sourceReference == destReference)) {
      // If the reference count is -1, it is invalid and cannot be decremented
      if (sourceReference[1] == -1) {
        return 0xFFFFFFFF;
      }
      referenceCounter = sourceReference + 1; // Get the reference count from the source
      DataMemoryBarrier(0xB); // Ensure memory operations complete before proceeding
      do {
        currentCount = *referenceCounter - 1; // Decrement the reference count
        exclusiveAccess = (bool)hasExclusiveAccess(referenceCounter); // Check for exclusive access to the counter
      } while (!exclusiveAccess);
      *referenceCounter = currentCount; // Set the new reference count
    } else {
      // If destReference is different, try to add to its queue
      currentCount = DecrementReferenceCountIfNotEqual(destReference, 1);
      if (currentCount != 0) {
        currentCount = destReference[3]; // Get the current queue index
        // Update the destination's queue with the source reference
        *(int **)(destReference[4] + currentCount * 4) = sourceReference;
        destReference[3] = currentCount + 1; // Increment the index for the next entry
        return 0;
      }
      // If queue addition failed and source reference count is -1, return error
      if (sourceReference[1] == -1) {
        return 0xFFFFFFFF;
      }
      // Otherwise, we decrement the source reference count
      referenceCounter = sourceReference + 1;
      DataMemoryBarrier(0xB);
      do {
        currentCount = *referenceCounter - 1;
        exclusiveAccess = (bool)hasExclusiveAccess(referenceCounter);
      } while (!exclusiveAccess);
      *referenceCounter = currentCount;
    }
    // If after decrementing, the reference count is 0, release the associated resource
    if (currentCount == 0) {
      ReleaseResource(sourceReference);
      return 0xFFFFFFFF;
    }
  }
  return 0xFFFFFFFF; // Return error if source reference is not valid
}
