int GetCurrentTimeMicrosecondsSinceEpoch(__time_t *outMicrosecondsSinceEpoch)
{
  struct timespec currentTime;
  
  // Get the current time from the clock, with a resolution up to nanoseconds 
  clock_gettime(CLOCK_MONOTONIC, &currentTime);
  
  // Set the first element of the output array to the number of seconds
  outMicrosecondsSinceEpoch[0] = currentTime.tv_sec;
  
  // Convert nanoseconds to microseconds and set that to the second element
  outMicrosecondsSinceEpoch[1] = currentTime.tv_nsec / 1000;
  
  // Return a transformed nanoseconds value
  // (Not entirely clear why this transformation is being done)
  return currentTime.tv_nsec * 0x10624dd3;
}
