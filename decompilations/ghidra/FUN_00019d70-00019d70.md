void CalculateStatistics(uint *data_array, uint *min_value, uint *max_value, uint *sum)
{
  int data_count;
  uint *current_data;
  uint current_value;
  uint min_value_temp;
  uint max_value_temp;
  uint sum_temp;
  uint *next_data;
  
  // Retrieve the number of data points from another function
  data_count = GetNumberOfDataPoints();
  
  // If there are no data points, set all statistics to zero or default values
  if (data_count < 1) {
    sum_temp = 0;
    min_value_temp = 0xffffff; // Assuming 0xffffff is the default "max" for min_value
    max_value_temp = sum_temp;
  }
  else {
    // Initialize temporary variables for calculation
    sum_temp = 0;
    max_value_temp = 0;
    min_value_temp = 0xffffff; // Assuming 0xffffff is the default "max" for min_value
    
    // Begin iterating through the data_array
    current_data = data_array;
    do {
      // Advance to the next data point
      next_data = current_data + 1;
      
      // Get the current value from the data array
      current_value = *current_data;
      
      // Add the current value to the running sum
      sum_temp += current_value;
      
      // Determine if the current_value is a new maximum
      if (max_value_temp < current_value) {
        max_value_temp = current_value;
      }
      
      // Determine if the current_value is a new minimum
      if (current_value <= min_value_temp) {
        min_value_temp = current_value;
      }
      
      // Move to the next data point
      current_data = next_data;
      
    } while (next_data != data_array + data_count); // Continue until all data points are processed
  }
  
  // Assign calculated statistics to the output parameters
  *min_value = min_value_temp;
  *max_value = max_value_temp;
  *sum = sum_temp;
  return;
}
