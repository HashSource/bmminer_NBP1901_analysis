#include <stdint.h>

// Function Prototype Decalarations
int32_t CheckMinerStatus(int minerIndex);
int32_t ResetMinerBoard(int minerIndex);

// Verifies the status of all miner boards and resets them if necessary
uint32_t VerifyAndResetAllMinerBoards(void)
{
  int minerIndex;
  int minerStatus;
  
  minerIndex = 0;
  while (true) {
    minerStatus = CheckMinerStatus(minerIndex);
    // If miner status check failed or the miner board reset succeeds, continue to next
    if (minerStatus == 0 || ResetMinerBoard(minerIndex) != 0) {
      minerIndex++;
      // If all 4 miner boards have been checked, return 1 indicating success
      if (minerIndex == 4) {
        return 1;
      }
    } else {
      // If the miner board fails to reset, return 0 indicating failure
      return 0;
    }
  }
}
