// Defines an error code to represent failure in this context
#define ERROR_CODE 0xffffffff

// Function to update the event associated with an existing thread
// param_1: Pointer to the thread object (must not be null)
// param_2: The event object to associate with the thread
// Returns 0 on success, ERROR_CODE on failure
uint32_t UpdateThreadEvent(int threadPointer, uint32_t newEvent)
{
  int newEventId;

  // If the provided pointer to the thread object is NULL, log an error and return an error code
  if (threadPointer == 0) {
    LogError(2, "src/thread.c", 0xa7, "threadPointer is null or 0");
    return ERROR_CODE;
  }

  // Attempt to create or retrieve a new event ID
  newEventId = CreateOrGetEventId(newEvent);
  
  // If creating or getting the event ID was successful, update the thread's event
  if (newEventId != 0) {
    FreeThreadEvent(*(uint32_t *)(threadPointer + 8));  // Free the old event associated with the thread
    *(int *)(threadPointer + 8) = newEventId;  // Update the thread object with the new event ID
    return 0; // Return success
  }

  // If creating or getting the new event ID failed, log an error and return an error code
  LogError(2, "src/thread.c", 0xab, "CreateOrGetEventId failed");
  return ERROR_CODE;
}
