void InitializeHardware(void)
{
  uint chipCount;
  uint hashboardCount;
  uint *localHashboardData;
  uint *tempLocalData;
  uint localDataIndex;
  uint configurationData[74];
  
  CollectConfigurationData(configurationData); // Read the configuration data from the hardware
  ApplyHardwareConfigurations(configurationData); // Apply the configurations to the hardware
  if (configurationData[0] != 0) { // If there are configured elements (hashboards, chips, etc.)
    tempLocalData = configurationData; // Point to the beginning of the configuration data array
    localDataIndex = 0; // Initialize index for traversing configuration data
    hashboardCount = configurationData[0]; // The first element is the number of hashboards
    do {
      if (tempLocalData[2] != 0) { // If there are chips to configure on this hashboard
        localHashboardData = tempLocalData + 2; // Point to the chip count for the current hashboard
        chipCount = 0; // Initialize chip counter
        do {
          localHashboardData++; // Move to the next chip configuration data
          EnableChip(*localHashboardData, true); // Enable the chip with the given ID
          chipCount++; // Increment the chip counter
          hashboardCount = configurationData[0]; // Update hashboard count from configuration data
        } while (chipCount < tempLocalData[2]); // Loop until all chips on this hashboard are enabled
      }
      localDataIndex++; // Move to the next hashboard data block
      tempLocalData += 0x12; // Move the data pointer to the next set (0x12 is the size of the data block per hashboard)
    } while (localDataIndex < hashboardCount); // Loop until all hashboards are processed
  }
  return;
}
