// Function to set frequency for a mining chip on the Antminer device
// param_1: The index (0 to 3) indicating which part of the frequency to set
// param_2: The frequency value to be used
// Returns: Always returns 0 (success)
uint32_t SetChipFrequency(int frequency_index, uint32_t frequency_value)
{
  uint32_t maskedFrequency;
  uint32_t frequencyToSet;
  
  // Get the masked frequency value by applying a mask operation on the input frequency
  maskedFrequency = MaskFrequencyValue(frequency_value);
  
  // Activate chip for frequency setting
  ActivateChipForFrequencySetting(0xf);
  
  frequencyToSet = 0;
  
  // Depending on the frequency index, shift the masked frequency to
  // the appropriate position for a 32-bit register setting
  if (frequency_index == 0) {
    frequencyToSet = maskedFrequency & 0x3f;
  }
  else if (frequency_index == 1) {
    frequencyToSet = (maskedFrequency & 0x3f) << 8;
  }
  else if (frequency_index == 2) {
    frequencyToSet = (maskedFrequency & 0x3f) << 0x10;
  }
  else if (frequency_index == 3) {
    frequencyToSet = (maskedFrequency & 0x3f) << 0x18;
  }
  
  // Set the frequency for the activated chip
  SetFrequencyForActiveChip(0xf, frequencyToSet);
  
  // Return success
  return 0;
}
