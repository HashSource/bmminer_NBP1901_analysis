void parse_dash_separated_values(char *input_str, int *first_value, int *second_value) {

  int num_values_scanned;
  
  // sscanf reads the input string for two integer values separated by a dash
  num_values_scanned = sscanf(input_str, "%d-%d", first_value, second_value);

  // If only one integer was read, copy it to second_value as well
  if (num_values_scanned == 1) {
    *second_value = *first_value;
  }

  return;
}
