int getMinTemperatureReading(uint32_t sensorType, uint sensorIndex) {
  int initResult;
  uint sensorCount;
  int tempReading;
  uint sensorIter;
  int minTempReading;
  
  initResult = initializeTemperatureSensors();
  // Check if the sensorIndex is within a valid range (less than 3) and sensors initialized successfully
  if (sensorIndex < 3 && initResult != 0) {
    // Get the count of sensors based on the sensorIndex
    if (sensorIndex == 0) {
      sensorCount = getCountOutdoorSensors();
    }
    else {
      sensorCount = getCountIndoorSensors();
    }
    // If there are sensors available
    if (sensorCount > 0) {
      sensorIter = 0;
      minTempReading = 0xff; // Initialize with max value (0xff is 255 in decimal)
      do {
        // Get temperature reading based on sensor type
        if (sensorIndex == 0) {
          // Get temperature for outdoor sensor
          tempReading = readOutdoorTemperatureSensor();
        }
        else if (sensorIndex == 1) {
          // Get temperature for indoor sensor at a specific sensor number
          tempReading = readIndoorTemperatureSensor(sensorType, sensorIter & 0xff);
        }
        else {
          // Get temperature for hypothetical third type of sensor
          tempReading = readOtherTemperatureSensor();
        }
        // Store temporary minimum reading
        minTempReading = (tempReading != 0xff && (minTempReading == 0xff || tempReading <= minTempReading)) 
                         ? tempReading : minTempReading;
        sensorIter++;
      } while (sensorIter != sensorCount); // Loop until all sensors have been read
      return minTempReading;
    }
  }
  // Return max value if no sensors or failed initialization (0xff indicates an error/no reading)
  return 0xff;
}
