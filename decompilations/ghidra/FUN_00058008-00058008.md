int * createHashTable(void)
{
  int *pHashTable;
  int result;
  int *pErrno;
  
  // Allocate memory for the hash table pointer
  pHashTable = (int *)calloc(1, sizeof(int));
  
  // If allocation fails, print an error message
  if (pHashTable == NULL) {
    pErrno = __errno_location();
    logError(2, "src/mdc.c", 0x45, "calloc fail, errno[%d]", *pErrno);
  }
  else {
    // Try to initialize the hash table
    result = initializeHashTable(0x14, &hashTableUpdateCallback, freeHashTableEntry, 0, freeCustomData);
    
    // Save the result (hash table ID or error code) in the allocated memory
    *pHashTable = result;
    
    // If hash table initialization fails, log error and free the allocated memory
    if (result == 0) {
      logError(2, "src/mdc.c", 0x4e, "hashtable_new fail");
      freeMemory(pHashTable);
      pHashTable = NULL; // Set pointer to NULL after freeing it to avoid dangling pointer
    }
  }
  return pHashTable; // Return the pointer to the newly created hash table
}
