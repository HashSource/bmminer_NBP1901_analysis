#include <errno.h>
#include <stdlib.h>

// Forward declaration for error logging function
// LogError function signature
void LogError(int logLevel, const char* sourceFile, int lineNum, const char* message, int errorCode);

// Create and initialize a new ArrayList structure
void** CreateArrayList(void *initialCapacityCallback)
{
  void **arrayList;
  void *arrayDataStorage;
  int *errnoPtr;
  
  // Allocate memory for the ArrayList structure, with space for 4 pointers
  arrayList = (void **)calloc(1, sizeof(void*) * 4);
  if (arrayList == NULL) {
    // Retrieve the error number and log the calloc failure
    errnoPtr = __errno_location();
    LogError(2, "src/zc_arraylist.c", 22, "calloc fail for ArrayList structure, errno[%d]", *errnoPtr);
  }
  else {
    // Initialize the ArrayList with the given callback, default capacity and size
    arrayList[3] = initialCapacityCallback;  // Callback to determine initial capacity
    arrayList[2] = (void *)32;               // Default capacity of 32 elements
    arrayList[1] = (void *)0;                // Current size (number of elements) is 0

    // Allocate memory for the array that will actually store the data, with space for 32 elements
    arrayDataStorage = calloc(32, sizeof(void*));
    arrayList[0] = arrayDataStorage;  // Pointer to the data storage
    if (arrayDataStorage == NULL) {
      // If data storage allocation fails, retrieve the error number and log it
      errnoPtr = __errno_location();
      LogError(2, "src/zc_arraylist.c", 32, "calloc fail for ArrayList data storage, errno[%d]", *errnoPtr);
      
      // Free the previously allocated ArrayList structure, as we can't proceed
      free(arrayList);
      arrayList = NULL;
    }
  }
  return arrayList; // Return the pointer to the newly created ArrayList or NULL if any allocation failed
}
