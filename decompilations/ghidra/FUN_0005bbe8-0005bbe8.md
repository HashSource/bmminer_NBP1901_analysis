#include <stdio.h>
#include <string.h>

// Error logger function declaration
void LogError(int level, const char* fileName, int line, const char* message);

uint CheckRuleMatchesCategory(char *rule, char *category) {
  int comparisonResult;
  size_t ruleLength;
  size_t categoryLength;

  // Validate rule pointer
  if (rule == NULL) {
    LogError(2, "src/rule.c", 0x3f6, "rule is null or 0");
    return 0xffffffff;
  }
  // Validate category pointer
  if (category != NULL) {
    // Check if the rule is a wildcard or exact match
    if (((*rule != '*') || (rule[1] != '\0')) && (comparisonResult = strcmp(rule, category), comparisonResult != 0)) {
      ruleLength = strlen(rule);
      // If rule does not end with '_', it's not a prefix so no match
      if (rule[ruleLength - 1] != '_') {
        return 0;
      }
      // Compare rule as a prefix
      categoryLength = strlen(category);
      // Prefix length must match category length or be one less if rule ends with '_'
      if (ruleLength - 1 != categoryLength) {
        categoryLength = ruleLength;
      }
      comparisonResult = strncmp(rule, category, categoryLength);
      // If strncmp result is 0 then we have a match, return 1 to indicate match
      return (uint)(comparisonResult == 0);
    }
    // If we have a wildcard '*' or exact match, return 1
    return 1;
  }
  // Log error if category pointer is null
  LogError(2, "src/rule.c", 0x3f7, "category is null or 0");
  return 0xffffffff;
}
