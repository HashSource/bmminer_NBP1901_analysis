void SwapEndianAndCopyData(int destinationAddress, uint *sourceData, uint dataLengthBytes)
{
  uint *sourceCurrentWord;
  uint *destinationCurrentWord;
  uint wordValue;
  uint *sourceEndWord;
  
  // If data length is less than 4 bytes, there's nothing to do
  if (dataLengthBytes >> 2 == 0) {
    return;
  }
  
  destinationCurrentWord = (uint *)(destinationAddress + -4);
  sourceCurrentWord = sourceData;
  
  // Loop through the data by words (4 bytes each)
  do {
    sourceEndWord = sourceCurrentWord + 1; // Move to the next word
    wordValue = *sourceCurrentWord; // Read the current word
    destinationCurrentWord = destinationCurrentWord + 1; // Move the destination pointer
    
    // Rearrange the bytes to swap the endianness and write to the destination
    *destinationCurrentWord = wordValue << 0x18 | (wordValue >> 8 & 0xff) << 0x10 | 
                              (wordValue >> 0x10 & 0xff) << 8 | wordValue >> 0x18;
    
    sourceCurrentWord = sourceEndWord; // Move to the next word in the source
  } while (sourceEndWord != (uint *)((int)sourceData + (dataLengthBytes & 0xfffffffc)));
  
  // Function is complete
  return;
}
