uint getMaxVoltageFromEEPROM(void)
{
  int deviceStatus;                      // Variable to hold the status of the device
  int deviceIndex;                       // Index to iterate through devices
  uint maxVoltage;                       // Variable to store the maximum voltage
  uint voltage;                          // Temporary variable to store voltage read from EEPROM
  char logBuffer[2048];                  // Buffer for logging messages
  
  maxVoltage = 0;                        // Initialize the maxVoltage to 0
  deviceIndex = 0;                       // Start from the first device
  do {
    deviceStatus = checkDeviceStatus(deviceIndex); // Check the status of the device
    if ((deviceStatus != 0) &&
        (deviceStatus = readVoltageFromEEPROM(deviceIndex, &voltage), deviceStatus == 0) && // Read voltage from EEPROM
       (maxVoltage < voltage)) {
      maxVoltage = voltage;              // Update maxVoltage if the current voltage is higher
    }
    deviceIndex += 1;
  } while (deviceIndex != 4);            // Loop for all 4 devices
  
  if (EEPROM_LOGGING_LEVEL < ruleLoggingLevel) { // Some kind of logging condition check
    snprintf(logBuffer, 0x800, "max voltage in eeprom = %d\n", maxVoltage); // Prepare log message
    logMessage(4, logBuffer, 0);         // Log the maximum voltage
  }
  
  if (maxVoltage == 0) {
    maxVoltage = 0xffffffff;             // Set maxVoltage to highest possible value if it was not updated
  }
  
  return maxVoltage;                     // Return the maximum voltage found in the EEPROM
}
