#include <stdio.h>  // For snprintf
#include <errno.h>  // For __errno_location

// Forward declarations for functions that are called but not defined in this snippet
extern void LogErrorAndExit(int level, const char *message, int exitCode);
extern void ImmediateExit(int statusCode);

// LogMutexLockError - Logs an error message when a mutex lock operation fails
// param_1: The name of the function where the error occurred (caller function name)
// param_2: The line number in the source file where the error occurred (caller line number)
void LogMutexLockError(const char *callerFuncName, int callerLineNumber)
{
  int *pErrno;                         // Pointer to the error number (errno) set by the last failing call
  char errorMsgBuffer[2052];           // Buffer for the formatted error message

  pErrno = __errno_location();         // Get the location of errno
  // Format the error message including the error number, and source file location where the error occurred
  snprintf(errorMsgBuffer, sizeof(errorMsgBuffer),
           "MUTEX LOCK ERROR! errno=%d in util.c %s():%d",
           *pErrno, callerFuncName, callerLineNumber);

  // Log the error message at error level 3, and indicate it should also exit
  LogErrorAndExit(3, errorMsgBuffer, 1);

  // Perform an immediate exit with status code 1 (this line may not be reached if LogErrorAndExit exits)
  ImmediateExit(1);
}
