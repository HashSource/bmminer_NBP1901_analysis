#include <stdio.h> // Include standard I/O header for snprintf function

// Function prototype declarations
int GetMinerCount(void);
int CalculateAverageFrequency(int totalFreq, int minerCount);
void LogDebugInfo(int level, char *message);

// Define constant value for logging level
#define LOG_LEVEL_DEBUG 4

// Global variable holding a log level threshold
extern int gLogLevelThreshold;

// CalculateChainFrequency computes the average PLL frequency for a specified mining chain.
// param_1: index of the mining chain to calculate the frequency for
// param_2: an offset related to the PLL frequency information
// Returns: the average PLL frequency for the specified mining chain
int CalculateChainFrequency(int chainIndex, int freqOffset)
{
  int minerCount;
  unsigned int totalFreq;
  int *freqArrayPtr;
  int currentMiner;
  unsigned int averageFreq;
  char debugInfoBuffer[2052]; // Buffer to store debug information message
  
  minerCount = GetMinerCount();
  if (minerCount == 0) {
    averageFreq = 0;
  }
  else {
    totalFreq = 0;
    currentMiner = 0;
    freqArrayPtr = (int *)(freqOffset + chainIndex * 0x400 + -4);
    do {
      currentMiner = currentMiner + 1;
      freqArrayPtr = freqArrayPtr + 1;
      totalFreq = totalFreq + *freqArrayPtr;
    } while (minerCount != currentMiner);
  }

  // Compute average frequency
  averageFreq = CalculateAverageFrequency(totalFreq, GetMinerCount());

  // Log average frequency if the log level is set to debug or higher
  if (LOG_LEVEL_DEBUG < gLogLevelThreshold) {
    snprintf(debugInfoBuffer, 0x800, "chain = %d, freq_avg = %d\n", chainIndex, averageFreq);
    LogDebugInfo(LOG_LEVEL_DEBUG, debugInfoBuffer, 0);
  }
  
  return averageFreq;
}
