void GenerateMinerChainInfo(char *outputBuffer)
{
  int chainCount = 0;    // Renamed iVar6 to chainCount
  int outputOffset = 0;  // Renamed iVar5 to outputOffset
  int chainVoltage;      // Removed uVar1 and declared chainVoltage
  int result;            // Renamed iVar2 to result for clarity
  int totalLength;       // Renamed iVar3 to totalLength for accumulated string length tracking
  int currentChain;      // Renamed iVar4 to currentChain
  char *writePtr;        // Renamed __s to writePtr for clearer purpose
  
  chainCount = 0;
  outputOffset = 0;      
  chainVoltage = GetChainVoltage();  // Assumed name for FUN_00019ab4(), which gets the voltage of a miner chain
  currentChain = 0;      // Using as the current chain index
  writePtr = outputBuffer;
  
  do {
    currentChain += 1;
    result = CheckChainPresence(currentChain);  // Assumed name for FUN_0001e29c(), which checks if a chain is present
    if (result != 0) {
      // Generate chain voltage information
      result = sprintf(writePtr, "chain%d_voltage=%d;", currentChain, chainVoltage);
      chainCount += 1;
      
      // Append additional information about the chain
      totalLength = sprintf(outputBuffer + result + outputOffset, "chain%d_voladded=%d;", currentChain, 0);
      totalLength = result + outputOffset + totalLength;
      result = sprintf(outputBuffer + totalLength, "chain%d_basefreq=%d;", currentChain, 300); // Assuming base frequency is always 300
      totalLength += result;
      outputOffset = sprintf(outputBuffer + totalLength, "chain%d_badcore=%d;", currentChain, 0);  // Assuming 0 bad cores by default
      outputOffset += totalLength;
      writePtr = outputBuffer + outputOffset;
    }
    result = currentChain;
  } while (currentChain != 4); // Loop over 4 chains (assuming there are always 4 chains)

  // Append total number of chains and version to the output buffer
  result = sprintf(writePtr, "chainnum=%d;", chainCount);
  
  // Version data assumed to be stored at &DAT_00080ba0
  sprintf(outputBuffer + outputOffset + result, "version=%s;", &DAT_00080ba0);
  return;
}
