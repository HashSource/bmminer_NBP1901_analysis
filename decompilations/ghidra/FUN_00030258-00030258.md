char * getTokenFromString(char **strPointer, char *delimiters)
{
  char *nextToken;
  char *currentString;
  
  currentString = *strPointer; // Get the current position of the string
  if ((currentString != NULL) && (nextToken = strpbrk(currentString, delimiters), nextToken != NULL)) {
    *nextToken = '\0'; // Terminate the current token
    *strPointer = nextToken + 1; // Move the string pointer to the character after the delimiter
    return currentString; // Return the token that has just been processed
  }
  *strPointer = NULL; // No more tokens, set the string pointer to NULL
  return currentString; // Return the current (last) token or NULL if the original string was NULL
}
