char *parseBoolFromString(char *inputStr, undefined *resultBool)
{
  int compareResult;
  size_t inputLen;
  char *errorMsg;
  
  // Compare input string with "yes" or "true" in a case-insensitive manner
  compareResult = strcasecmp(inputStr, "yes");
  if ((compareResult == 0) || (compareResult = strcasecmp(inputStr, "true"), compareResult == 0)) {
    *resultBool = 1; // Set the result to true
    return NULL; // No error message needed when successful
  }

  // Compare input string with "no" or "false" in a case-insensitive manner
  compareResult = strcasecmp(inputStr, "no");
  if ((compareResult != 0) && (compareResult = strcasecmp(inputStr, "false"), compareResult != 0)) {
    // If neither "yes"/"true" nor "no"/"false", return an error message
    inputLen = strlen(inputStr);
    errorMsg = (char *)malloc(inputLen + 22); // Allocate memory for error message string
    sprintf(errorMsg, "Invalid argument \'%s\'", inputStr); // Format the error message
    return errorMsg; // Return the error message
  }

  // If the input string was "no" or "false", set the result to false
  *resultBool = 0;
  return NULL; // No error message needed when successful
}
