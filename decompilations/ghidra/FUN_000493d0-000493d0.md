#include <pthread.h>

/* WARNING: Globals starting with '_I2C_LOCK' overlap smaller symbols at the same address */

// Rename functions for clarity
int InitializeI2C(int parameters_pointer) {
  int lock_result;
  int init_result;
  int ioctl_result;
  char error_buffer[1024];
  
  // Attempt to lock mutex for I2C operations
  lock_result = pthread_mutex_lock((pthread_mutex_t *)&_I2C_LOCK);
  if (lock_result != 0) {
    snprintf(error_buffer, sizeof(error_buffer), "failed to iic lock");
    LogError(0, error_buffer);
    return -4;
  }

  if (parameters_pointer == 0) {
    // If parameters pointer is null, return with an error
    snprintf(error_buffer, sizeof(error_buffer), "bad param");
    LogError(0, error_buffer);
    init_result = -3;
  } else {
    // Try initializing the I2C bus
    init_result = I2CInit();
    if (init_result < 0) {
      snprintf(error_buffer, sizeof(error_buffer), "failed to i2c init");
      LogError(0, error_buffer);
      init_result = -1;
    }
    else {
      I2CSelectorParameters *params = (I2CSelectorParameters *)parameters_pointer;
      // Perform operations on the I2C bus based on the parameters provided
      ioctl_result = SelectI2CDevice(init_result, params->device_address);
      if (ioctl_result != 0) {
        snprintf(error_buffer, sizeof(error_buffer), "failed to i2c select");
        LogError(0, error_buffer);
      }
      else {
        ioctl_result = SetI2CParameters(init_result, params->config_register);
        if (ioctl_result != 0) {
          snprintf(error_buffer, sizeof(error_buffer), "failed to i2c ioctl");
          LogError(0, error_buffer);
        }
        else {
          // Success case
          init_result = 0;
        }
      }

      // Close the I2C device if it was opened
      if (init_result != 0) {
        I2CRelease(init_result);
      }
    }
  }

  // Unlock the mutex now that I2C operations are done
  pthread_mutex_unlock((pthread_mutex_t *)&_I2C_LOCK);
  return init_result;
}
