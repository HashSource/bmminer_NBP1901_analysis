int getMinimumTemperature(void)
{
  short currentTemp;
  int sensorIndex;
  int minimumTemp;
  
  sensorIndex = 0;
  minimumTemp = 0xff; // Initializing with the maximum value for an unsigned char (255 in decimal)
  do {
    currentTemp = readTemperature(sensorIndex, 2); // Reading the temperature from sensor index
    sensorIndex = sensorIndex + 1;
    
    // Check if the current temperature is lower than the previously found minimum
    if (currentTemp < minimumTemp) {
      minimumTemp = (int)currentTemp;
    }
  } while (sensorIndex != 4); // Assuming there are 4 sensors to read from
  
  return minimumTemp; // Returns the minimum temperature found
}
