extern int gHardwareState;

char GetComponentStatus(uint componentIndex)
{
  char status;
  
  // Check if hardware is initialized and componentIndex is within valid range
  if ((gHardwareState != 0) && (componentIndex < 5)) {
    // Access the component's status from the hardware state structure
    // Assuming there's an offset of 0x10 to reach the status bytes in the structure
    status = *(char *)(gHardwareState + componentIndex + 0x10);
    
    // If status is non-zero, return 1 indicating the component is active
    if (status != '\0') {
      status = '\x01';
    }

  return '\0';
}
