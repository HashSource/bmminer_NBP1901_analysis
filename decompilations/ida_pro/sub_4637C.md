int __fastcall resetPic(unsigned int chainIndex)
{
  int status; // r6
  int result; // r4
  __int16 readBackData; // [sp+Ch] [bp-82Ch] BYREF
  int i2cData[4]; // [sp+10h] [bp-828h] BYREF
  char logBuffer[2072]; // [sp+20h] [bp-818h] BYREF

  // Check if chainIndex is more than 15 (0xF) and log an error message if true.
  if (chainIndex > 0xF)
  {
    snprintf(logBuffer, 0x800u, "%s: Bad pic param, input chain is %d\n", "reset_pic", chainIndex);
    result = -2147483135;
    logMessage(0, logBuffer, 0);
    return result;
  }

  // Check if the chain EEPROM has not been opened yet, and attempt to open it.
  if (!openedChainEEPROM[2 * chainIndex + 1])
  {
    result = openChainEEPROM(chainIndex);
    if (result < 0)
    {
      snprintf(logBuffer, 0x800u, "%s: auto exec open_pic, but chain %d open eeprom failed\n", "reset_pic", chainIndex);
      logMessage(0, logBuffer, 0);
      return result;
    }
  }

  // Get the file descriptor for the chain's I2C device
  status = openedChainEEPROM[2 * chainIndex];
  result = 0;

  // Acquire mutex lock before I2C communication
  pthread_mutex_lock(&i2cMutex);

  // Prepare the I2C data to reset the PIC
  i2cData[1] = 2816; // The actual I2C command or data may go here
  i2cData[0] = 117746261; // This value could be a specific command or initialization for the reset
  i2cData[2] = 0; // Additional data or flags if needed
  i2cData[3] = 0; // Additional data or flags if needed
  readBackData = 0;

  // Write I2C data to the PIC
  if (i2cPicWriteCommand(status, (int)i2cData, 6) != 6)
  {
    snprintf(logBuffer, 0x800u, "%s write iic err\n", "_bitmain_pic_reset_common");
    goto logError;
  }

  usleep(0x493E0u); // Delay for PIC to process reset (300000 microseconds?)

  // Read I2C data from the PIC
  if (i2cPicReadCommand(status, &readBackData, 2) != 2)
  {
    snprintf(logBuffer, 0x800u, "%s read iic err\n", "_bitmain_pic_reset_common");
    goto logError;
  }

  // Check if the read-back data from the PIC is as expected
  if ((unsigned __int8)readBackData == 7 && HIBYTE(readBackData) == 1)
    goto unlockAndFinish;

  snprintf(
    logBuffer,
    0x800u,

    "_bitmain_pic_reset_common",
    (unsigned __int8)readBackData,
    HIBYTE(readBackData));
  logMessage(0, logBuffer, 0);

logError:
  result = -2147483136;

unlockAndFinish:
  pthread_mutex_unlock(&i2cMutex);
  usleep(0x7A120u); // Delay after unlock (500000 microseconds?)
  return result;
}
