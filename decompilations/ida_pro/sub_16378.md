int InitializeChainConfigurations()
{
  int chainIndex;              // r4
  int **chainConfigurations;   // r6
  int resultCode;              // r0
  void *chainConfig;           // r8
  int chainCount;              // r0
  int singleConfig;            // r8
  int **chainConfigPtr;        // r4
  _BYTE *configMemoryArea;     // r3
  _BYTE *tempPtr;              // t1
  _BYTE *memoryLocation;       // r3
  char logBuffer[2072];        // [sp+0h] [bp-818h] BYREF

  if ( !globalChainConfigs[0] )
  {
    globalChainConfigs[0] = (int *)calloc(1u, 0x20u);
    if ( !globalChainConfigs[0] )
    {
      if ( globalLogLevel > 3 )
      {
        strcpy(logBuffer, "No memory for configuration.\n");
        LogMessage(3, logBuffer, 0);
        chainConfigurations = (int **)globalChainConfigs[0];
        goto CLEANUP_CONFIGS;
      }
      return -1;
    }
  }
  chainIndex = 0;
  while ( 1 )
  {
    if ( !IsChainIndexValid(chainIndex) )
      goto CHECK_NEXT_CHAIN;
    chainConfigurations = (int **)globalChainConfigs[0];
    if ( chainConfigurations[chainIndex] )
      goto CHECK_NEXT_CHAIN;
    chainConfig = calloc(1u, 0x48u);
    chainConfigurations[chainIndex] = chainConfig;
    if ( !chainConfig )
      break;
    chainCount = GetChainCount();
    chainConfigurations = (int **)globalChainConfigs[0];
    singleConfig = chainConfigurations[chainIndex][0];
    chainConfigurations[chainIndex][15] = calloc(chainCount, 1u);
    if ( !chainConfigurations[chainIndex][15] )
    {
      if ( globalLogLevel <= 3 )
        goto CLEANUP_CONFIGS;
      goto LOG_MEMORY_ERROR;
    }
CHECK_NEXT_CHAIN:
    if ( ++chainIndex == 4 )
      return 0;
  }
  if ( globalLogLevel > 3 )
  {
LOG_MEMORY_ERROR:
    snprintf(logBuffer, 0x800u, "No memory for configuration for chain %d.\n", chainIndex);
    LogMessage(3, logBuffer, 0);
    chainConfigurations = (int **)globalChainConfigs[0];
  }
CLEANUP_CONFIGS:
  if ( !chainConfigurations )
    return -1;
  chainConfigPtr = chainConfigurations - 1;
  do
  {
    tempPtr = (_BYTE *)chainConfigPtr[1];
    ++chainConfigPtr;
    configMemoryArea = tempPtr;
    if ( tempPtr )
    {
      if ( *(_DWORD *)(configMemoryArea + 63) )
      {
        free(*(void **)(configMemoryArea + 63));
        memoryLocation = (_BYTE *)*chainConfigPtr;
        memoryLocation[63] = 0;
        memoryLocation[64] = 0;
        memoryLocation[65] = 0;
        memoryLocation[66] = 0;
        configMemoryArea = (_BYTE *)*chainConfigPtr;
      }
      free(configMemoryArea);
      *chainConfigPtr = 0;
    }
  }
  while ( chainConfigPtr != chainConfigurations + 3 );
  free(chainConfigurations);
  resultCode = -1;
  globalChainConfigs[0] = 0;
  return resultCode;
}
