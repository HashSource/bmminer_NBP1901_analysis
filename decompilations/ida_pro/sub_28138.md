int __fastcall setUartBaudRate(signed int baudRate)
{
  char logBuffer[2056]; // [sp+0h] [bp-808h] BYREF

  if ( baudRate > UART_BAUDRATE_MAX )
  {
    shutdownASIC();
    restartASIC();
  }
  if ( (unsigned int)debugLogLevel > 3 )
  {
    snprintf(logBuffer, 0x800u, "set UART baud to %d\n", baudRate);
    writeToLog(3, logBuffer, 0);
  }
  configureUartBaud(baudRate);
  resetUart();
  return testUartConnection();
}
