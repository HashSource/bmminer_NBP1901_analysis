int __fastcall getASICChipLevel(int *levelListPtr, int chipIndex)
{
  int numberOfLevels; // r3
  bool isIndexInRange; // cc
  int defaultLevelOffset; // r3
  
  numberOfLevels = levelListPtr[1]; // The second element is the number of levels
  if (numberOfLevels > chipIndex && *(_DWORD *)(*levelListPtr + 4 * chipIndex) )
    return *(_DWORD *)(*levelListPtr + 4 * chipIndex); // Return the level at the chipIndex

  // Log an error message when the requested level is not available
  logError(
    2,
    "src/level_list.c",
    119,
    "l[%d] not in (0,254), or has no level defined, see configure file define, set to UNKNOWN",
    chipIndex);

  // Check if the number of levels is within the legal range (0 to 254)
  isIndexInRange = numberOfLevels <= 254;

  if (isIndexInRange)
    defaultLevelOffset = 0; // If it's within the range, set the default level offset to 0
  else
    defaultLevelOffset = *levelListPtr; // If it's not within the range, use the first element of levelListPtr

  // Return the level at the default offset (either 0 or at 254th index, which is *(levelListPtr + 1016))
  if (!isIndexInRange)
    return *(_DWORD *)(defaultLevelOffset + 1016); // Note: 1016 = 4 * 254, because each level is 4 bytes (int size)
    
  return defaultLevelOffset;
}
