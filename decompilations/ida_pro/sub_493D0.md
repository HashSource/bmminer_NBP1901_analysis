int __fastcall initiateI2CTransfer(int i2cTransactionParams)
{
  int i2cHandle; // r0
  int result; // r5
  int unlockResult; // r0
  char errorMessageBuffer[2072]; // [sp+0h] [bp-818h] BYREF

  if ( pthread_mutex_lock(&i2cMutex) )
  {
    result = -4;
    strcpy(errorMessageBuffer, "failed to iic lock\n");
    logMessage(0, errorMessageBuffer, 0);
    return result;
  }
  if ( i2cTransactionParams )
  {
    i2cHandle = initializeI2C();
    result = i2cHandle;
    if ( i2cHandle < 0 )
    {
      result = -1;
      strcpy(errorMessageBuffer, "failed to i2c_init\n");
      logMessage(0, errorMessageBuffer, 0);
    }
    else
    {
      if ( selectI2CSlave(i2cHandle, *(unsigned __int16 *)(i2cTransactionParams + 4)) )
      {
        strcpy(errorMessageBuffer, "failed to i2c_select\n");
        logMessage(0, errorMessageBuffer, 0);
        goto ERROR_HANDLE;
      }
      if ( performI2CIOCTL(i2cHandle, 1795, (2 * *(unsigned __int8 *)(i2cTransactionParams + 7)) | (16 * *(unsigned __int8 *)(i2cTransactionParams + 6))) )
      {
        strcpy(errorMessageBuffer, "failed to i2c_ioctl\n");
        logMessage(0, errorMessageBuffer, 0);
ERROR_HANDLE:
        if ( i2cHandle )
        {
          unlockResult = i2cHandle;
          result = -1;
          finalizeI2C(unlockResult);
        }
        else
        {
          result = -1;
        }
      }
    }
  }
  else
  {
    result = -3;
    strcpy(errorMessageBuffer, "bad param\n");
    logMessage(0, errorMessageBuffer, 0);
  }
  pthread_mutex_unlock(&i2cMutex);
  return result;
}
