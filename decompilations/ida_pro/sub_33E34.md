int InitializeMinerHardware()
{
  int status;     // r4
  int voltage;    // r0
  int minTemp;    // r0
  const char *hwVersion; // r0
  char logBuffer[2064]; // [sp+0h] [bp-810h] BYREF

  InitializeMinerDriver();
  InitializeHardwareMonitor();
  status = CheckASICCommunication(0xFA0u, 1);
  if ( status )
    return status;
  if ( (unsigned int)globalLogLevel > 3 )
  {
    strcpy(logBuffer, "Fan check passed.\n");
    LogMessage(3, logBuffer, 0);
    status = CheckFanStatus();
    if ( !status )
      goto LABEL_FanOK;
  }
  else
  {
    status = CheckFanStatus();
    if ( !status )
      goto LABEL_FanOK;
  }
  if ( !IsMiningAllowed() || !miningEnabledFlag )
    return status;

LABEL_FanOK:
  ReadTemperatureSensor();
  ScanHashBoardPresence();
  if ( hashBoardPresenceFlag )
  {
    hwVersion = GetHardwareVersion();
    if ( !strcmp(hwVersion, "BHB28601") )
      voltage = 1900;
    else
      voltage = 1500;
    goto LABEL_SetVoltage;
  }
  voltage = pwrVersionVoltage;
  if ( pwrVersionVoltage > 0 )
    goto LABEL_SetVoltage;

LABEL_SetVoltage:
  status = ValidatePowerVoltage(voltage);
  if ( status )
  {
    TriggerHardwareError(12, 0);
    LogError(9u, "get power type version failed!");
    return status;
  }
  goto LABEL_LogTemperature;

LABEL_LogTemperature:
  if ( (unsigned int)globalLogLevel > 3 )
  {
    minTemp = GetMinimumStartupTemperature();
    snprintf(logBuffer, 0x800u, "start up min temp by 75a = %d", minTemp);
    LogMessage(3, logBuffer, 0);
  }
  InitializeASICClocks();
  InitializeASICVoltageControl();
  InitializeASICJobControl();
  SetupUARTCommunication();
  SetUARTBaudRate(115200);
  SetupUARTCommunication();
  status = SetupMiningThreads();
  if ( !status || IsMiningAllowed() && miningEnabledFlag )
  {
    StartMiningProcess();
    status = 0;
    ASICResetFlag = 0;
    ASICRebootFlag = 0;
    ASICErrorFlag = 0;
  }
  return status;
}
