int __fastcall GetRegisterCacheValue(int cache_level, unsigned int chain_index, unsigned int asic_index, unsigned int register_address, _DWORD *cached_value)
{
  unsigned int address_offset; // r6
  int error_code; // r10
  unsigned int combined_index_second_level; // r6
  unsigned int combined_index_first_level; // r6
  _DWORD *register_cache_ptr; // r6

  address_offset = (unsigned __int8)(register_address >> 2);
  if ( chain_index > 3 )
    return -1;
  if ( asic_index > 0xFF )
    return -2;
  if ( address_offset > 0x3F )
    return -3;
  pthread_mutex_lock(&mutex_register_cache);
  if ( cache_level )
  {
    if ( cache_level != 1 )
    {
      error_code = -6;
      pthread_mutex_unlock(&mutex_register_cache);
      goto ErrorExit;
    }
    combined_index_second_level = address_offset + ((asic_index + (chain_index << 8)) << 6);
    if ( second_level_cache[2 * combined_index_second_level] != register_address )
    {
      error_code = -5;
      pthread_mutex_unlock(&mutex_register_cache);
      goto ErrorExit;
    }
    register_cache_ptr = &second_level_cache[2 * combined_index_second_level];
  }
  else
  {
    combined_index_first_level = address_offset + (chain_index << 6);
    if ( first_level_cache[2 * combined_index_first_level] != register_address )
    {
      error_code = -4;
      pthread_mutex_unlock(&mutex_register_cache);
      goto ErrorExit;
    }
    register_cache_ptr = &first_level_cache[2 * combined_index_first_level];
  }
  *cached_value = register_cache_ptr[1];
  pthread_mutex_unlock(&mutex_register_cache);
  return 0;

ErrorExit:
  printf(
    "[%s] level::%d, which_chain::%u, which_asic::%u, register_address::%02x failed.\r\n",
    "GetRegisterCacheValue",
    cache_level,
    chain_index,
    asic_index,
    register_address);
  return error_code;
}
