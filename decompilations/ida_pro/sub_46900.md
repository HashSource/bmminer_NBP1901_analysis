int __fastcall disableDCDCConverter(unsigned int chainIndex)
{
  int mutexState; // r6
  int result; // r4
  __int16 readBackData; // [sp+Ch] [bp-82Ch] BYREF
  int i2cBuffer[4]; // [sp+10h] [bp-828h] BYREF
  char logBuffer[2072]; // [sp+20h] [bp-818h] BYREF

  // Validate chainIndex is within range
  if ( chainIndex > 0xF )
  {
    snprintf(logBuffer, 0x800u, "%s: Bad pic param, input chain is %d\n", "disable_dc_dc", chainIndex);
    result = -2147483135;
    logMessage(0, logBuffer, 0);
    return result;
  }

  // Check if the EEPROM is already opened for this chainIndex
  if ( !globalChainStatus[2 * chainIndex + 1] )
  {
    result = openEEPROMForChainIndex(chainIndex);
    if ( result < 0 )
    {
      snprintf(logBuffer, 0x800u, "%s: auto exec open_pic, but chain %d open eeprom failed\n", "disable_dc_dc", chainIndex);
      logMessage(0, logBuffer, 0);
      return result;
    }
  }

  // Retrieve the file descriptor for this chainIndex
  mutexState = globalChainStatus[2 * chainIndex];
  result = 0;

  // Acquire a mutex for I2C communication
  pthread_mutex_lock(&i2cMutex);

  // Prepare the I2C command buffer
  i2cBuffer[1] = 1703936; // Command data part
  i2cBuffer[0] = 352692821; // Command header part
  i2cBuffer[2] = 0; // Possible padding or additional data
  i2cBuffer[3] = 0; // Possible padding or additional data
  readBackData = 0;

  // Send the I2C command to disable DCDC converter
  if ( i2cWrite(mutexState, (int)i2cBuffer, 7) != 7 )
  {
    snprintf(logBuffer, 0x800u, "%s write iic err\n", "disableDCDCConverter");
    goto LogError;
  }

  usleep(0x493E0u); // Delay after sending the command

  // Read the response from I2C
  if ( i2cRead(mutexState, &readBackData, 2) != 2 )
  {
    snprintf(logBuffer, 0x800u, "%s read iic err\n", "disableDCDCConverter");
    goto LogError;
  }

  // Validate the response received from I2C read
  if ( (unsigned __int8)readBackData == 21 && HIBYTE(readBackData) == 1 )
    goto ExitFunction;

  // If received data does not match expected values, log an error
  snprintf(
    logBuffer,
    0x800u,
    "%s failed! read_back_data[0] = 0x%02x, read_back_data[1] = 0x%02x\n\n",
    "disableDCDCConverter",
    (unsigned __int8)readBackData,
    HIBYTE(readBackData));
  logMessage(0, logBuffer, 0);
  result = -2147483136;

LogError:
  logMessage(0, logBuffer, 0);

ExitFunction:
  pthread_mutex_unlock(&i2cMutex);
  return result;
}
