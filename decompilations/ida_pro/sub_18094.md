int __fastcall getPCBVersionForChain(int chainIndex)
{
  char logBuffer[2056]; // [sp-808h] [bp-808h] BYREF

  // Check if globalChainConfigs is initialized and the chain has a PCB version set
  if (globalChainConfigs[0] && *(_BYTE *)(globalChainConfigs[0] + 28))
    // Return the PCB version for the specified chain
    // Note: Each entry in globalChainConfigs array seems to be a struct with PCB version at +28/+27 offsets
    return *(unsigned __int8 *)(*(_DWORD *)(globalChainConfigs[0] + 4 * chainIndex) + 28) |
           (*(unsigned __int8 *)(*(_DWORD *)(globalChainConfigs[0] + 4 * chainIndex) + 27) << 8);

  // If globalChainConfigs is not set properly or no PCB version is available, log a message if globalLogLevel allows
  if ((unsigned int)globalLogLevel > 3)
  {
    snprintf(logBuffer, 0x800u, "No pcb version, chain = %d.\n", chainIndex);
    logMessage(3, logBuffer, 0);
  }

  // If PCB version information is not available, return 0xFFFF to indicate an error or unknown version
  return 0xFFFF;
}
