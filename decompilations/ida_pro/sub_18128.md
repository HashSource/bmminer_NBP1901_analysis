int __fastcall getChainBomVersion(int chainIndex)
{
  char logBuffer[2056]; // [sp-808h] [bp-808h] BYREF

  // Check if the globalChainConfigs pointer is valid and if the 'has_version' flag is set
  if (globalChainConfigs[0] && *(_BYTE *)(globalChainConfigs[0] + 28))
    // Return the combined 'version_major' and 'version_minor' as a 16-bit value
    return *(unsigned __int8 *)(*(_DWORD *)(globalChainConfigs[0] + 4 * chainIndex) + 30) |
           (*(unsigned __int8 *)(*(_DWORD *)(globalChainConfigs[0] + 4 * chainIndex) + 29) << 8);
  
  // Log a message if the global log level is higher than 3 (debug level)
  if ((unsigned int)globalLogLevel > 3)
  {
    snprintf(logBuffer, 0x800u, "No bom version, chain = %d.\n", chainIndex);
    logMessage(3, logBuffer, 0);
  }
  
  // Return an error code (0xFFFF) if the BOM version is not available
  return 0xFFFF;
}
