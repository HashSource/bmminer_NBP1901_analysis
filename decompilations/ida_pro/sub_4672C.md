int __fastcall enableDcDc(unsigned int chainId)
{
  int mutexStatus; // r6
  int statusCode; // r4
  __int16 iicReadData; // [sp+Ch] [bp-82Ch] BYREF
  int iicWriteData[4]; // [sp+10h] [bp-828h] BYREF
  char logBuffer[2072]; // [sp+20h] [bp-818h] BYREF

  if ( chainId > 0xF )
  {
    snprintf(logBuffer, 0x800u, "%s: Bad pic param, input chain is %d\n", "enable_dc_dc", chainId);
    statusCode = -2147483135;
    logMessage(0, logBuffer, 0);
    return statusCode;
  }
  if ( !picEepromOpenedFlags[2 * chainId + 1] )
  {
    statusCode = openEepromForChain(chainId);
    if ( statusCode < 0 )
    {
      snprintf(logBuffer, 0x800u, "%s: auto exec open_pic, but chain %d open eeprom failed\n", "enable_dc_dc", chainId);
      logMessage(0, logBuffer, 0);
      return statusCode;
    }
  }
  mutexStatus = picEepromOpenedFlags[2 * chainId];
  statusCode = 0;
  pthread_mutex_lock(&picIicMutex);
  iicWriteData[1] = 1769473;
  iicWriteData[0] = 352692821;
  iicWriteData[2] = 0;
  iicWriteData[3] = 0;
  iicReadData = 0;
  if ( writeToPicIic(mutexStatus, (int)iicWriteData, 7) != 7 )
  {
    snprintf(logBuffer, 0x800u, "%s write iic err\n", "_bitmain_pic_enable_dc_dc_common");
    goto LABEL_logError;
  }
  usleep(0x493E0u);
  if ( readFromPicIic(mutexStatus, &iicReadData, 2) != 2 )
  {
    snprintf(logBuffer, 0x800u, "%s read iic err\n", "_bitmain_pic_enable_dc_dc_common");
    goto LABEL_logError;
  }
  if ( (unsigned __int8)iicReadData == 21 && HIBYTE(iicReadData) == 1 )
    goto LABEL_unlockMutex;
  snprintf(
    logBuffer,
    0x800u,
    "%s failed! read_back_data[0] = 0x%02x, read_back_data[1] = 0x%02x\n\n",
    "_bitmain_pic_enable_dc_dc_common",
    (unsigned __int8)iicReadData,
    HIBYTE(iicReadData));
  logMessage(0, logBuffer, 0);
LABEL_logError:
  statusCode = -2147483136;
LABEL_unlockMutex:
  pthread_mutex_unlock(&picIicMutex);
  return statusCode;
}
