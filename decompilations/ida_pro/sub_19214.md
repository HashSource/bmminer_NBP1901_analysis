int __fastcall initiatePICJumpToAppAndVerifyFirmware(int chainIndex)
{
  int firmwareVersion; // r0
  char logBuffer[2064]; // [sp+8h] [bp-810h] BYREF

  // Log the chain index if the global log level is greater than 3
  if ((unsigned int)globalLogLevel > 3)
  {
    snprintf(logBuffer, 0x800u, "chain[%d] PIC jump to app\n", chainIndex);
    logMessage(3, logBuffer, 0);
  }
  
  // Initiate jump of PIC to the application
  sendJumpToAppCommand((unsigned __int8)chainIndex);
  
  // Sleep for 500,000 microseconds (0.5 seconds) to allow for PIC processing
  usleep(0x7A120u);
  
  // Verify the firmware version of the PIC
  firmwareVersion = verifyPICFirmware((unsigned __int8)chainIndex);
  
  // Log the firmware version if the global log level is greater than 3
  if ((unsigned int)globalLogLevel > 3)
  {
    snprintf(logBuffer, 0x800u, "Check chain[%d] PIC fw version=0x%02x\n", chainIndex, firmwareVersion);
    return logMessage(3, logBuffer, 0);
  }
  
  return firmwareVersion;
}
