int __fastcall readASICVoltage(int chainIndex)
{
  int status; // r0
  unsigned int logLevel; // r2
  double voltage; // d8
  int rawVoltageData[3]; // [sp+Ch] [bp-83Ch] BYREF
  char logBuffer[2096]; // [sp+18h] [bp-830h] BYREF

  memset(rawVoltageData, 0, sizeof(rawVoltageData));
  status = readVoltageSensorData((unsigned __int8)chainIndex, rawVoltageData, 1); // originally sub_47158
  if ( (BYTE)status )
  {
    logLevel = globalLogLevel;
    voltage = (double)rawVoltageData[0] * 3.3 * 0.0009765625 * 7.6; // Convert raw data to actual voltage
    if ( (unsigned int)globalLogLevel > 5 )
    {
      snprintf(logBuffer, 0x800u, "an0 = %f.\n", voltage); // an0 might represent analog input 0
      status = logMessage(5, logBuffer, 0); // originally result
      logLevel = globalLogLevel;
      voltage = (double)rawVoltageData[0] * 3.3 * 0.0009765625 * 7.6; // Redundant calculation
    }
  }
  else
  {
    voltage = 0.0;
    logLevel = globalLogLevel;
  }
  if ( logLevel > 4 )
  {
    snprintf(logBuffer, 0x800u, "chain = %d, voltage = %f\n", chainIndex, voltage);
    return logMessage(4, logBuffer, 0); // Log voltage for the corresponding chain
  }
  return status;
}
