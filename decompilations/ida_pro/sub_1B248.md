int __fastcall setChainFrequency(int chainIndex, unsigned int command, int freqMHz)
{
  char logBuffer[4]; // [sp+14h] [bp-81Ch] BYREF
  char messageBuffer[4]; // [sp+18h] [bp-818h] BYREF
  int parameterValue; // [sp+1Ch] [bp-814h]

  if ( (unsigned int)globalLogLevel > 3 )
  {
    snprintf(messageBuffer, 0x800u, "chain = %d, freq = %d\n", chainIndex, freqMHz);
    logMessage(3, messageBuffer, 0);
  }
  *(_DWORD *)messageBuffer = 0;
  parameterValue = 0;
  prepareFrequencySettingParameters((int)messageBuffer, logBuffer, 0);
  BYTE2(parameterValue) = logBuffer[0];
  return applyFrequencySettings(chainIndex, 1, 0, command, *(int *)messageBuffer, parameterValue);
}
